## 1. Acceptance Criteria
1. **All HTTP requests** processed by the gateway must contain a unique `X-Request-ID` header (or generate one if absent).  
2. **All HTTP responses** must echo the same `X-Request-ID` header back to the client.  
3. **Log entries** for each request/response must include:
   - `timestamp` (ISO‑8601, UTC)
   - `provider_id` (string)
   - `request_id` (the value from the header)
   - `translation_status` (`queued`, `in_progress`, `completed`, `failed`)
4. **Log level** (e.g., `DEBUG`, `INFO`, `WARN`, `ERROR`) is settable via an environment variable `LOG_LEVEL`.  
5. If `LOG_LEVEL` is unset, default to `INFO`.  
6. The middleware must not alter other request/response headers or body content.  
7. The implementation must be thread‑safe and not introduce race conditions in concurrent request handling.

package middleware_test

import (
    "net/http"
    "net/http/httptest"
    "os"
    "testing"

    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"

    "axentx/thai-gateway/internal/api"

func TestRequestIDMiddleware_GeneratesID(t *testing.T) {
    req := httptest.NewRequest(http.MethodGet, "/test", nil)
    // No X-Request-ID header
    rr := httptest.NewRecorder()

    handler := api.RequestIDMiddleware(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("X-Request-ID", r.Header.Get("X-Request-ID"))
        w.WriteHeader(http.StatusOK)
    }))

    handler.ServeHTTP(rr, req)

    resp := rr.Result()
    defer resp.Body.Close()

    // Header should be present and non-empty
    id := resp.Header.Get("X-Request-ID")
    assert.NotEmpty(t, id, "Generated X-Request-ID should not be empty")
    // ID should be a UUID v4 (simple regex check)
    assert.Regexp(t, `^[a-f0-9-]{36}$`, id)
}

func TestRequestIDMiddleware_PropagatesExistingID(t *testing.T) {
    req := httptest.NewRequest(http.MethodPost, "/echo", nil)
    req.Header.Set("X-Request-ID", "12345")
    rr := httptest.NewRecorder()

    handler := api.RequestIDMiddleware(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("X-Request-ID", r.Header.Get("X-Request-ID"))
        w.WriteHeader(http.StatusOK)
    }))

    handler.ServeHTTP(rr, req)

    resp := rr.Result()
    defer resp.Body.Close()

    assert.Equal(t, "12345", resp.Header.Get("X-Request-ID"))
}

func TestLogLevelEnvVar(t *testing.T) {
    os.Setenv("LOG_LEVEL", "DEBUG")
    defer os.Unsetenv("LOG_LEVEL")

    logger := api.NewLogger() // assumes logger reads env on init
    assert.Equal(t, "DEBUG", logger.Level())
}

func TestLogLevelDefault(t *testing.T) {
    os.Unsetenv("LOG_LEVEL")
    logger := api.NewLogger()
    assert.Equal(t, "INFO", logger.Level())
}
