	return func(ctx context.Context, req *Request) (*Response, error) {

#### 3. Logger Wrapper (`internal/logger/logger.go`)

A thin wrapper around the standard library logger that respects the configured log level.

```go
package logger

import (
	"log"
	"os"
	"strings"
	"sync"
)

type Level int

const (
	DEBUG Level = iota
	INFO
	WARN
	ERROR
)

var (
	levelMap = map[string]Level{
		"DEBUG": DEBUG,
		"INFO":  INFO,
		"WARN":  WARN,
		"ERROR": ERROR,
	}
)

// Logger is a simple level‑aware logger.
type Logger struct {
	mu    sync.Mutex
	level Level
}

// New creates a Logger using the supplied log level string.
func New(levelStr string) *Logger {
	lvl, ok := levelMap[strings.ToUpper(levelStr)]
	if !ok {
		lvl = INFO
	}
	return &Logger{level: lvl}
}

// SetLevel updates the logger's level at runtime.
func (l *Logger) SetLevel(levelStr string) {
	l.mu.Lock()
	defer l.mu.Unlock()
	if lvl, ok := levelMap[strings.ToUpper(levelStr)]; ok {
		l.level = lvl
	}
}

// internal helper – prints only if the message level is >= configured level.
func (l *Logger) logf(minLevel Level, format string, v ...any) {
	l.mu.Lock()
	defer l.mu.Unlock()
	if l.level <= minLevel {
		log.Printf(format, v...)
	}
}

// Debug logs at DEBUG level.
func (l *Logger) Debugf(format string, v ...any) { l.logf(DEBUG, format, v...) }

// Info logs at INFO level.
func (l *Logger) Infof(format string, v ...any) { l.logf(INFO, format, v...) }

// Warn logs at WARN level.
func (l *Logger) Warnf(format string, v ...any) { l.logf(WARN, format, v...) }

// Error logs at ERROR level.
func (l *Logger) Errorf(format string, v ...any) { l.logf(ERROR, format, v...) }

// InitStdLogger redirects the standard library's output to the same destination
// (useful for third‑party packages that call log.Printf directly).
func InitStdLogger() {
	log.SetOutput(os.Stdout)
}
```

## 1. Acceptance Criteria
1. **All HTTP requests** processed by the gateway must contain a unique `X-Request-ID` header (or generate one if absent).  
2. **All HTTP responses** must echo the same `X-Request-ID` header back to the client.  
3. **Log entries** for each request/response must include:
   - `timestamp` (ISO‑8601, UTC)
   - `provider_id` (string)
   - `request_id` (the value from the header)
   - `translation_status` (`queued`, `in_progress`, `completed`, `failed`)
4. **Log level** (e.g., `DEBUG`, `INFO`, `WARN`, `ERROR`) is settable via an environment variable `LOG_LEVEL`.  
5. If `LOG_LEVEL` is unset, default to `INFO`.  
6. The middleware must not alter other request/response headers or body content.  
7. The implementation must be thread‑safe and not introduce race conditions in concurrent request handling.
