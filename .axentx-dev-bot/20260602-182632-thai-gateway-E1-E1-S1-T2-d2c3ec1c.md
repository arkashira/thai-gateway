// router.go
package handler

import (
	"fmt"
	"log"
	"net/http"

	"github.com/axentx/thai-gateway/config"
)

// Router is a basic request router
type Router struct {
	endpoints map[string]string
}

// NewRouter returns a new instance of the Router
func NewRouter() *Router {
	return &Router{
		endpoints: make(map[string]string),
	}
}

// RegisterEndpoint registers a new endpoint
func (r *Router) RegisterEndpoint(provider string, endpoint string) {
	r.endpoints[provider] = endpoint
}

// ServeHTTP serves the request
func (r *Router) ServeHTTP(w http.ResponseWriter, req *http.Request) {
	provider := req.URL.Query().Get("provider")
	if provider == "" {
		http.Error(w, "missing provider query parameter", http.StatusBadRequest)
		return
	}

	endpoint, ok := r.endpoints[provider]
	if !ok {
		http.Error(w, fmt.Sprintf("unknown provider: %s", provider), http.StatusNotFound)
		return
	}

	http.Redirect(w, req, endpoint, http.StatusFound)
}

func init() {
	log.Println("starting router")
}
```
