### `/opt/axentx/thai-gateway/src/metrics.rs`

```diff
+use prometheus::{
+    Encoder, TextEncoder, Registry,
+    CounterVec, HistogramVec, Opts,
+};
+use std::sync::Arc;
+
+/// Centralised Prometheus metrics for the gateway.
+pub struct Metrics {
+    pub registry: Registry,
+    pub usage: CounterVec,
+    pub latency: HistogramVec,
+    pub errors: CounterVec,
+}
+
+impl Metrics {
+    /// Initialise all metrics and register them with the registry.
+    pub fn new() -> Self {
+        let registry = Registry::new();
+
+        // Counter for total provider usage.
+        let usage_opts = Opts::new("provider_usage_total", "Total provider usage");
+        let usage = CounterVec::new(usage_opts, &["provider_id"]).unwrap();
+
+        // Histogram for latency in ms.
+        let latency_opts = Opts::new("provider_latency_ms", "Provider latency in ms");
+        let latency = HistogramVec::new(latency_opts, &["provider_id"]).unwrap();
+
+        // Counter for errors, labelled by provider and status code.
+        let errors_opts = Opts::new("provider_error_total", "Total provider errors");
+        let errors = CounterVec::new(errors_opts, &["provider_id", "status_code"]).unwrap();
+
+        // Register metrics.
+        registry.register(Box::new(usage.clone())).unwrap();
+        registry.register(Box::new(latency.clone())).unwrap();
+        registry.register(Box::new(errors.clone())).unwrap();
+
+        Self {
+            registry,
+            usage,
+            latency,
+            errors,
+        }
+    }
+
+    /// Record a single provider invocation.
+    /// `status_code` is the HTTP status returned by the provider.
+    /// `latency_ms` is the round‑trip time in milliseconds.
+    pub fn record(&self, provider_id: &str, status_code: u16, latency_ms: u64) {
+        self.usage.with_label_values(&[provider_id]).inc();
+        self.latency.with_label_values(&[provider_id]).observe(latency_ms as f64);
+        if status_code >= 400 {
+            self.errors
+                .with_label_values(&[provider_id, &status_code.to_string()])
+                .inc();
+        }
+    }
+
+    /// Return the current metrics in Prometheus text format.
+    pub fn gather(&self) -> String {
+        let encoder = TextEncoder::new();
+        let metric_families = self.registry.gather();
+        let mut buffer = Vec::new();
+        encoder.encode(&metric_families, &mut buffer).unwrap();
+        String::from_utf8(buffer).unwrap()
+    }
+}
```
