## Diagnosis
The PRD requires a **provider status endpoint** that:

1. Exposes each AI provider’s *active* / *inactive* state.
2. Is reachable via a dedicated route (e.g. `/providers/status`).
3. Uses the same configuration source that the gateway already consumes (environment variables or a config file).
4. Is registered in the main Express app so that it can be hit by health‑check probes.

The current code base has no such route and the gateway’s provider configuration is only read during startup.  
We’ll create a lightweight route that reads the current provider list from the shared configuration module and returns a JSON payload of provider names and their status.

## Proposed change

```
/opt/axentx/thai-gateway/src/routes/providerStatus.js
```

Register it in the main server file:

```
/opt/axentx/thai-gateway/src/index.js
```

The route will:

* Import the shared `config` module that already parses `process.env` / config file.
* Map each provider to an object `{ name, status }`.
* Return a 200 JSON response with the array.

---

## Implementation

### 1. Create `providerStatus.js`

```diff
+// src/routes/providerStatus.js
+const express = require('express');
+const router = express.Router();
+
+// The config module holds the parsed provider list.
+const { providers } = require('../config');
+
+/**
+ * GET /providers/status
+ * Returns the active/inactive status of each configured AI provider.
+ */
+router.get('/providers/status', (req, res) => {
+  // providers is expected to be an array of objects:
+  // [{ name: 'openai', active: true }, ...]
+  const status = providers.map(p => ({
+    name: p.name,
+    status: p.active ? 'active' : 'inactive',
+  }));
+
+  res.json({ providers: status });
+});
+
+module.exports = router;
```

```diff
// src/index.js
const express = require('express');
const app = express();

...

...
+// Register provider status endpoint
+app.use('/', providerStatusRouter);
...
module.exports = app;
```

```js
// tests/providerStatus.test.js
const request = require('supertest');
const app = require('../src/app'); // Express instance
const config = require('../src/config');

describe('GET /provider/status', () => {
  beforeEach(() => {
    // Mock configuration with two providers
    config.set({
      providers: [
        { name: 'openai', active: true },
        { name: 'anthropic', active: false }
      ]
    });
  });

  test('returns 200 and JSON array', async () => {
    const res = await request(app).get('/provider/status');
    expect(res.status).toBe(200);
    expect(Array.isArray(res.body)).toBe(true);
  });

  test('includes active provider with status "active"', async () => {
    const res = await request(app).get('/provider/status');
    const openai = res.body.find(p => p.name === 'openai');
    expect(openai).toBeDefined();
    expect(openai.status).toBe('active');
    expect(openai).toHaveProperty('lastChecked');
  });

  test('includes inactive provider with status "inactive"', async () => {
    const res = await request(app).get('/provider/status');
    const anthro = res.body.find(p => p.name === 'anthropic');
    expect(anthro).toBeDefined();
    expect(anthro.status).toBe('inactive');
  });

  test('omits providers with missing name', async () => {
    config.set({
      providers: [
        { active: true } // invalid entry
      ]
    });
    const res = await request(app).get('/provider/status');
    expect(res.body.length).toBe(0);
  });

  test('updates status after config reload', async () => {
    // Initial state
    let res = await request(app).get('/provider/status');
    expect(res.body.find(p => p.name === 'openai').status).toBe('active');

    // Reload config to deactivate
    config.set({
      providers: [
        { name: 'openai', active: false }
      ]
    });

    res = await request(app).get('/provider/status');
    expect(res.body.find(p => p.name === 'openai').status).toBe('inactive');
  });
});
```
