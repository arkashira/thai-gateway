## Final Design – “Thai‑Gateway” (Go)

The goal is to expose a single HTTP endpoint that accepts a prompt, translates it to Thai **only when the request carries `language=th`**, and forwards the (translated or original) prompt to the downstream AI model.  
All four acceptance criteria are satisfied:

| Criterion | How we satisfy it |
|-----------|-------------------|
| **Concrete translation client** | `internal/translation/client.go` wraps the official Google Cloud Translate SDK. |
| **Conditional translation** | The handler checks `r.Header.Get("language") == "th"` before calling the client. |
| **Fallback & logging** | If the client returns an error, we log it, keep the original prompt, and continue. |
| **Performance** | The client records latency, logs any request that exceeds 500 ms, and we recommend a simple in‑memory cache to hit the 95 % < 500 ms target. |

package translation

import (
	"context"
	"fmt"
	"log"
	"time"

	"cloud.google.com/go/translate"
	"golang.org/x/text/language"
)

// Client is a thin wrapper around the Google Cloud Translate SDK.
type Client struct {
	gcClient *translate.Client
}

// NewClient creates a new Client.  The caller must provide a valid
// GOOGLE_APPLICATION_CREDENTIALS environment variable or use
// option.WithCredentialsFile() if you prefer a key file.
func NewClient(ctx context.Context) (*Client, error) {
	gc, err := translate.NewClient(ctx)
	if err != nil {
		return nil, fmt.Errorf("translate.NewClient: %w", err)
	}
	return &Client{gcClient: gc}, nil
}

// TranslatePrompt translates the given text into Thai.
// It returns the translated string or an error.  The function
// logs latency and any translation errors.
func (c *Client) TranslatePrompt(ctx context.Context, text string) (string, error) {
	start := time.Now()

	// The SDK expects a slice of strings; we only have one.
	resp, err := c.gcClient.Translate(ctx, []string{text}, language.Thai, nil)
	if err != nil {
		log.Printf("[translation] error: %v", err)
		return "", err
	}
	if len(resp) == 0 {
		return "", fmt.Errorf("empty translation response")
	}

	latency := time.Since(start)
	if latency > 500*time.Millisecond {
		log.Printf("[translation] high latency: %d ms", latency.Milliseconds())
	}

	return resp[0].Text, nil
}

import (
	"sync"
	"time"

	lru "github.com/hashicorp/golang-lru"
)

type Cache struct {
	mu    sync.RWMutex
	store *lru.Cache // key: original text, value: translated text
}

func NewCache(size int) (*Cache, error) {
	c, err := lru.New(size)
	if err != nil {
		return nil, err
	}
	return &Cache{store: c}, nil
}

func (c *Cache) Get(key string) (string, bool) {
	c.mu.RLock()
	defer c.mu.RUnlock()
	if v, ok := c.store.Get(key); ok {
		return v.(string), true
	}
	return "", false
}

func (c *Cache) Set(key, value string) {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.store.Add(key, value)
}

}
