const { validationResult, check } = require('express-validator');
const logger = require('../utils/logger'); // Assuming a logger utility is available

// Validation schema for OpenAI
const openAIValidation = [
  check('model').isString().withMessage('Model must be a string'),
  check('messages').isArray().withMessage('Messages must be an array'),
  check('messages.*.role').isIn(['user', 'assistant']).withMessage('Role must be user or assistant'),
  check('messages.*.content').isString().withMessage('Content must be a string'),
];

// Validation schema for Anthropic
const anthroValidation = [
  check('prompt').isString().withMessage('Prompt must be a string'),
  check('max_tokens').isInt({ min: 1 }).withMessage('Max tokens must be a positive integer'),
];

// Middleware to validate requests
const validateRequest = (req, res, next) => {
  const provider = req.headers['x-provider']; // Assuming provider is specified in headers
  let validationSchema;

  if (provider === 'openai') {
    validationSchema = openAIValidation;
  } else if (provider === 'anthropic') {
    validationSchema = anthroValidation;
  } else {
    return res.status(400).json({ error: 'Unsupported provider' });
  }

  // Validate the request payload
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    logger.error('Malformed request:', { errors: errors.array(), body: req.body });
    return res.status(400).json({ errors: errors.array() });
  }

  // Proceed to the next middleware if validation passes
  next();
};

module.exports = {
  validateRequest,
};

### 1. Acceptance Criteria
1. **Provider‑specific schema compliance** – The middleware must output a payload that validates against the official JSON‑Schema of each supported provider (OpenAI *chat/completions* and Anthropic *messages*).  
2. **Pre‑send validation** – If the transformed payload does not satisfy the provider schema, the middleware returns **HTTP 400** with a JSON body `{ error: "<description>" }`.  
3. **Malformed‑request logging** – Every rejected request is appended to `logs/validation_errors.log` with ISO‑timestamp, `provider`, and the validation error details.  
4. **Successful pass‑through** – For a valid request, the middleware calls `next()` (or the downstream provider client) with the transformed payload unchanged and returns the provider’s response unchanged to the caller.  
5. **Extensible provider registry** – Adding a new provider only requires a new schema file and an entry in `providersConfig`; the middleware automatically picks it up without code changes.  
6. **Performance bound** – Validation + transformation must complete in **≤ 15 ms** for payloads up to **5 KB** (measured on CI hardware).

describe('validateAndTransform – unit', () => {
  const baseReq = {
    provider: 'openai',
    model: 'gpt-4o-mini',

    temperature: 0.7,
  };

  test('returns transformed payload that matches OpenAI schema', () => {
    const { payload, error } = validateAndTransform(baseReq);
    expect(error).toBeNull();

    const validate = ajv.compile(providersConfig.openai.schema);
    expect(validate(payload)).toBe(true);
  });

  test('returns transformed payload that matches Anthropic schema', () => {
    const anthropicReq = { ...baseReq, provider: 'anthropic' };
    const { payload, error } = validateAndTransform(anthropicReq);
    expect(error).toBeNull();

    const validate = ajv.compile(providersConfig.anthropic.schema);
    expect(validate(payload)).toBe(true);
  });

  test('rejects request with missing required field', () => {
    const badReq = { ...baseReq, messages: undefined };
    const { payload, error } = validateAndTransform(badReq);
    expect(payload).toBeNull();
    expect(error).toMatch(/messages.*required/);
    expect(logger.error).toHaveBeenCalledWith(

    );
  });

  test('rejects request with unsupported provider', () => {

  });
