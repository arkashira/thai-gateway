```python
import os
import time
import logging
from typing import Optional, Dict

import yaml
from google.cloud import translate_v2 as translate
from cachetools import TTLCache, cached

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #
CONFIG_PATH = os.getenv("TRANSLATION_CONFIG", "translation.yaml")

with open(CONFIG_PATH, "r", encoding="utf-8") as f:
    cfg = yaml.safe_load(f)

API_KEY: str = cfg.get("api_key", "")
TIMEOUT: int = cfg.get("timeout", 5)          # seconds
CACHE_TTL: int = cfg.get("cache_ttl", 300)    # seconds
CACHE_MAXSIZE: int = cfg.get("cache_maxsize", 1024)

# --------------------------------------------------------------------------- #
# Logger
# --------------------------------------------------------------------------- #
logger = logging.getLogger("translator")
logger.setLevel(logging.INFO)
handler = logging.StreamHandler()
handler.setFormatter(
    logging.Formatter("[%(asctime)s] %(levelname)s %(name)s: %(message)s")
)
logger.addHandler(handler)

# --------------------------------------------------------------------------- #
# Google Translate client
# --------------------------------------------------------------------------- #
translate_client = translate.Client()

# --------------------------------------------------------------------------- #
# Cache for translation results
# --------------------------------------------------------------------------- #
translation_cache: TTLCache = TTLCache(maxsize=CACHE_MAXSIZE, ttl=CACHE_TTL)


def _translate_text(text: str, glossary_id: Optional[str] = None) -> str:
    """
    Internal helper that calls Google Translate API.
    If glossary_id is provided, the glossary is used.
    """
    params: Dict = {
        "q": text,
        "source": "en",
        "target": "th",
        "format": "text",
        "model": "nmt",
        "timeout": TIMEOUT,
    }

    if glossary_id:
        params["glossary"] = glossary_id

    try:
        start = time.time()
        result = translate_client.translate(**params)
        latency = time.time() - start
        logger.info(
            f"Translated in {latency:.3f}s: '{text[:30]}...' -> '{result['translatedText'][:30]}...'"
        )
        return result["translatedText"]
    except Exception as exc:
        logger.error(f"Translation failed for '{text[:30]}...': {exc}")
        raise


# --------------------------------------------------------------------------- #
# Public API
# --------------------------------------------------------------------------- #
@cached(translation_cache)
def translate_prompt(prompt: str, glossary_id: Optional[str] = None) -> str:
    """
    Translate a prompt from English to Thai.

    Parameters
    ----------
    prompt : str
        The English prompt to translate.
    glossary_id : str, optional
        Google Cloud Glossary ID to preserve technical terms.

    Returns
    -------
    str
        Translated Thai prompt. If translation fails, returns the original prompt.
    """
    try:
        translated = _translate_text(prompt, glossary_id=glossary_id)
        return translated
    except Exception:
        # Fallback to original prompt
        logger.warning("Fallback to original prompt due to translation error.")
        return prompt


# --------------------------------------------------------------------------- #
# Example usage (remove or guard with __name__ == "__main__" in production)
# --------------------------------------------------------------------------- #
if __name__ == "__main__":
    sample = "Order the latest iPhone 14 Pro Max with 256GB storage."
    print(translate_prompt(sample, glossary_id=cfg.get("glossary_id")))
```

## 1. Acceptance Criteria
1. **Translation Execution** – Every incoming user prompt is sent to the configured Thai translation API and the translated text is used when forwarding the request to the downstream AI provider.  
2. **Intent Preservation** – For a curated validation set of 200 real‑world e‑commerce queries, ≥ 90 % of translated prompts retain key entities (product names, SKUs, technical terms) as verified by a deterministic entity‑extraction script.  
3. **Latency SLA** – 90 % of translation calls complete in ≤ 500 ms (measured from receipt of the original prompt to receipt of the translated text).  
4. **Graceful Fallback** – If the translation API returns an error, times‑out, or yields an empty string, the system forwards the original (raw) prompt to the AI provider and logs a structured error entry containing request ID, error type, and timestamp.  
5. **Observability** – Successful translations, fallbacks, and latency metrics are emitted as Prometheus counters/gauges (`thai_gateway_translation_success_total`, `thai_gateway_translation_fallback_total`, `thai_gateway_translation_latency_seconds`).  
6. **Security Hardening** – All outbound calls to the translation API use HTTPS, the API key is read only from environment variables or a secrets manager, and no request payload is logged in plain text. Unit tests must assert that no secret appears in logs.
