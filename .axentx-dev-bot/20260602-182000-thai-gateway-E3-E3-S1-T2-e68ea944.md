## 1. Acceptance Criteria  
- **AC1**: Whenever the gateway receives a response from an AI provider that contains Thai text, the gateway must invoke the DeepL API to translate that text to English.  
- **AC2**: The translated response must preserve the original JSON structure, including any nested tool calls, metadata, or other fields.  
- **AC3**: If the DeepL API call fails (network error, non‑200 status, or malformed response), the failure must be logged with sufficient context (request ID, provider, error message) but the original (Thai) response must still be returned to the client.  
- **AC4**: The logger used for translation failures must be the same logger instance located at `/opt/axentx/thai-gateway/internal/logger/logger.go`.  
- **AC5**: Unit tests must cover successful translation, failure scenarios, and structural preservation.  
- **AC6**: Integration tests must verify that a client request receives an English response when the provider returns Thai, and that the original Thai response is still returned when translation fails.

type MockDeepLClient struct{ mock.Mock }

func (m *MockDeepLClient) Translate(ctx context.Context, src, tgt, text string) (string, error) {
    args := m.Called(ctx, src, tgt, text)
    return args.String(0), args.Error(1)
}

func TestTranslate_Success(t *testing.T) {
    mockDL := new(MockDeepLClient)
    mockDL.On("Translate", mock.Anything, "th", "en", "สวัสดี").Return("Hello", nil)

    lgr := logger.NewMockLogger()
    tr := translator.New(mockDL, lgr)

    out, err := tr.Translate(context.Background(), "สวัสดี")
    assert.NoError(t, err)
    assert.Equal(t, "Hello", out)
    lgr.AssertNotCalled(t, "Error")
}

func TestTranslate_Failure_LogsError(t *testing.T) {

    tr := translator.New(mockDL, lgr)

    assert.NoError(t, err) // translation failure should not propagate
    assert.Equal(t, "สวัสดี", out) // original Thai returned

    lgr.AssertCalled(t, "Error", mock.Anything, mock.AnythingOfType("string"))
}
