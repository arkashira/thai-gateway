// selector.go
package router

import (
	"context"
	"fmt"
	"sync"
	"time"

	"github.com/axentx/thai-gateway/pkg/provider"
	"github.com/axentx/thai-gateway/pkg/util"
	"github.com/gin-gonic/gin"
)

// selector is responsible for selecting the provider with the lowest current load.
type selector struct {
	providers map[string]*provider.Provider
	mu        sync.RWMutex
}

// NewSelector returns a new selector instance.
func NewSelector(providers map[string]*provider.Provider) *selector {
	return &selector{
		providers: providers,
	}
}

// SelectProvider returns the provider with the lowest current load.
func (s *selector) SelectProvider(ctx context.Context) (*provider.Provider, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()

	var lowestLoadedProvider *provider.Provider
	var lowestLoadedCount int

	for _, provider := range s.providers {
		count, err := s.getProviderCount(provider.Name)
		if err != nil {
			return nil, err
		}

		if count < lowestLoadedCount || lowestLoadedProvider == nil {
			lowestLoadedProvider = provider
			lowestLoadedCount = count
		}
	}

	if lowestLoadedProvider == nil {
		return nil, fmt.Errorf("all providers are saturated")
	}

	return lowestLoadedProvider, nil
}

// getProviderCount returns the current request count per minute for the given provider.
func (s *selector) getProviderCount(providerName string) (int, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()

	count, ok := s.providers[providerName].Count
	if !ok {
		return 0, fmt.Errorf("provider %s not found", providerName)
	}

	return count, nil
}

// UpdateProviderCount updates the request count per minute for the given provider.
func (s *selector) UpdateProviderCount(providerName string) {
	s.mu.Lock()
	defer s.mu.Unlock()

	s.providers[providerName].Count++
}

// HandleRequest handles incoming requests and routes them to the selected provider.
func (s *selector) HandleRequest(c *gin.Context) {
	provider, err := s.SelectProvider(c.Request.Context())
	if err != nil {
		if err.Error() == "all providers are saturated" {
			c.JSON(429, gin.H{
				"error": "all providers are saturated",
				"retry-after": time.Now().Add(1 * time.Minute).Unix(),
			})
			return
		}

		c.JSON(500, gin.H{
			"error": err.Error(),
		})
		return
	}

	// Update the request count per minute for the selected provider.
	s.UpdateProviderCount(provider.Name)

	// Route the request to the selected provider.
	provider.HandleRequest(c)
}
