package router

import (
	"context"
	"net/http"
	"sync"
	"time"
)

// ---------------------------------------------------------------------
// Public model – unchanged from the existing code base
// ---------------------------------------------------------------------
type Provider struct {
	Name                 string
	MaxRequestsPerMinute int
	Endpoint             string // where the request will be forwarded
}

// ---------------------------------------------------------------------
// Selector – the new, thread‑safe load‑balancer
// ---------------------------------------------------------------------
type Selector struct {
	mu        sync.Mutex          // protects counters & lastReset
	counters  map[string]int      // provider name → requests in current minute
	lastReset time.Time           // timestamp of the last bucket clear
}

// NewSelector creates a selector ready for use.
func NewSelector() *Selector {
	return &Selector{
		counters:  make(map[string]int),
		lastReset: time.Now(),
	}
}

// resetIfNeeded clears the per‑minute counters when a minute has elapsed.
// It is called at the start of every SelectProvider call *and* by a background
// ticker (see selector_init.go) to guarantee timely cleanup.
func (s *Selector) resetIfNeeded() {
	s.mu.Lock()
	defer s.mu.Unlock()
	if time.Since(s.lastReset) >= time.Minute {
		s.counters = make(map[string]int)
		s.lastReset = time.Now()
	}
}

// SelectProvider chooses the least‑loaded provider that is still under its
// limit.  It returns nil when **all** providers are saturated.
//
// The function also increments the chosen provider’s counter so the caller
// does **not** need a separate Increment step.
func (s *Selector) SelectProvider(ctx context.Context, providers []Provider) *Provider {
	// Ensure we are working with a fresh bucket.
	s.resetIfNeeded()

	var chosen *Provider
	minCount := int(^uint(0) >> 1) // Max int

	// Scan providers once – O(N) where N is the number of providers.
	for i := range providers {
		p := &providers[i]

		s.mu.Lock()
		count := s.counters[p.Name]
		s.mu.Unlock()

		// Skip saturated providers.
		if count >= p.MaxRequestsPerMinute {
			continue
		}
		// Keep the provider with the smallest count.
		if count < minCount {

			chosen = p

		return nil

## 1. Acceptance Criteria
1. **Per‑provider request counting** – The selector maintains a rolling count of requests received for each configured provider, reset automatically every minute.
2. **Load‑aware selection** – When `SelectProvider([]Provider)` is called, it returns the provider that (a) has the **lowest** current request count **and** (b) is **below** its `MaxRequestsPerMinute`.  
3. **Saturation handling** – If **all** supplied providers have `requestCount ≥ MaxRequestsPerMinute`, `SelectProvider` returns an error indicating saturation.
4. **HTTP 429 response** – The gateway layer that uses the selector translates the saturation error into an HTTP 429 response with a `Retry-After` header set to the number of seconds until the next minute tick.
5. **Thread‑safety** – The selector can be called concurrently from multiple goroutines without race conditions or lost updates.
6. **Metrics exposure** – The current per‑provider counts are exposed via a `GetMetrics()` method for observability (optional but verified for correctness).

All criteria are measurable via unit and integration tests.
