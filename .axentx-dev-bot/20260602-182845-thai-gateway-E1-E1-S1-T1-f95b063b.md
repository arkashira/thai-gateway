// src/config/validator.go

package config

import (
	"encoding/json"
	"errors"
	"fmt"
	"net/http"
	"net/url"
	"strings"
)

// ProviderConfig represents the configuration for an AI provider
type ProviderConfig struct {
	Name               string            `json:"name"`
	Endpoint           string            `json:"endpoint"`
	APIKey             string            `json:"api_key"`
	MaxRequestsPerMin  int               `json:"max_requests_per_minute"`
}

// ValidateProviderConfig validates the provider configuration
func ValidateProviderConfig(config []byte) ([]ProviderConfig, error) {
	var providers []ProviderConfig
	if err := json.Unmarshal(config, &providers); err != nil {
		return nil, fmt.Errorf("invalid JSON: %v", err)
	}

	for i, provider := range providers {
		if provider.Name == "" {
			return nil, fmt.Errorf("provider %d: name is required", i)
		}
		if provider.Endpoint == "" {
			return nil, fmt.Errorf("provider %d: endpoint is required", i)
		}
		if _, err := url.Parse(provider.Endpoint); err != nil {
			return nil, fmt.Errorf("provider %d: invalid endpoint URL: %v", i, err)
		}
		if provider.APIKey == "" {
			return nil, fmt.Errorf("provider %d: api_key is required", i)
		}
		if provider.MaxRequestsPerMin <= 0 {
			return nil, fmt.Errorf("provider %d: max_requests_per_minute must be positive", i)
		}
	}

	return providers, nil
}

// CheckProviderHealth checks if a provider is reachable
func CheckProviderHealth(provider ProviderConfig) (bool, error) {
	// Simple health check - just try to connect to the endpoint
	// In a real implementation, this would make a request to a health endpoint
	// or check for basic connectivity
	_, err := http.Get(provider.Endpoint)
	if err != nil {
		return false, fmt.Errorf("provider %s is unreachable: %v", provider.Name, err)
	}
	return true, nil
}

// ValidateAllProviders checks all providers are reachable
func ValidateAllProviders(providers []ProviderConfig) error {
	for _, provider := range providers {
		if _, err := CheckProviderHealth(provider); err != nil {
			return fmt.Errorf("health check failed for %s: %v", provider.Name, err)
		}
	}
	return nil
}
```
