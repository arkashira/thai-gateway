package health

import (
	"log"
	"net/http"
	"sync"
	"time"
)

// Provider represents an AI provider
type Provider struct {
	Name     string
	URL      string
	Healthy  bool
	LastCheck time.Time
	mu       sync.Mutex
}

// HealthCheck performs a health check on a provider
func (p *Provider) HealthCheck() {
	p.mu.Lock()
	defer p.mu.Unlock()

	resp, err := http.Get(p.URL + "/health")
	if err != nil {
		log.Printf("Health check failed for %s: %v", p.Name, err)
		p.Healthy = false
		p.LastCheck = time.Now()
		return
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		log.Printf("Health check failed for %s: status code %d", p.Name, resp.StatusCode)
		p.Healthy = false
		p.LastCheck = time.Now()
		return
	}

	log.Printf("Health check passed for %s", p.Name)
	p.Healthy = true
	p.LastCheck = time.Now()
}

// HealthCheckRoutine starts a goroutine to periodically check the health of providers
func HealthCheckRoutine(providers []*Provider, interval time.Duration) {
	ticker := time.NewTicker(interval)
	defer ticker.Stop()

	for range ticker.C {
		for _, provider := range providers {
			go provider.HealthCheck()
		}
	}
}

// GetHealthMetrics returns the health status of all providers
func GetHealthMetrics(providers []*Provider) map[string]bool {
	metrics := make(map[string]bool)
	for _, provider := range providers {
		provider.mu.Lock()
		metrics[provider.Name] = provider.Healthy
		provider.mu.Unlock()
	}
	return metrics
}
