### 1. Add a new endpoint to reload provider configurations dynamically

```python
# /opt/axentx/thai-gateway/main.py

from fastapi import FastAPI, HTTPException
from config import load_providers, validate_providers
import os

app = FastAPI()

PROVIDERS = load_providers()

@app.post("/reload-providers")
async def reload_providers():
    global PROVIDERS
    try:
        new_providers = load_providers()
        if not validate_providers(new_providers):
            raise HTTPException(status_code=400, detail="Invalid provider configuration")
        PROVIDERS = new_providers
        return {"message": "Providers reloaded successfully"}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.on_event("startup")
async def startup_event():
    if not validate_providers(PROVIDERS):
        raise HTTPException(status_code=500, detail="Invalid provider configuration on startup")
```

### 2. Implement validation for provider configurations

```python
# /opt/axentx/thai-gateway/validators.py

def validate_providers(providers):
    if not providers:
        return False
    for provider in providers:
        if "name" not in provider or "api_key" not in provider:
            return False
    return True
```

### 3. Update the configuration handling

```python
# /opt/axentx/thai-gateway/config.py

import os
from validators import validate_providers

def load_providers():
    providers_env = os.getenv("PROVIDERS", "")
    if not providers_env:
        return []
    providers = providers_env.split(",")
    return [{"name": provider, "api_key": os.getenv(f"{provider.upper()}_API_KEY")} for provider in providers]
```
