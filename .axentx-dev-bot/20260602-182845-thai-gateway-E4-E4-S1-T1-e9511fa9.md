package handlers

import (
	"encoding/json"
	"errors"
	"log"
	"net/http"

	"github.com/axentx/thai-gateway/src/providers"
	"github.com/axentx/thai-gateway/src/translate"
)

// AIRequest represents the JSON payload for the /ai endpoint
type AIRequest struct {
	Prompt string `json:"prompt"`
	Model  string `json:"model,omitempty"`
}

// handleAI handles the POST /ai endpoint
func handleAI(w http.ResponseWriter, r *http.Request) {
	// Validate the request method
	if r.Method != http.MethodPost {
		http.Error(w, "Invalid request method", http.StatusMethodNotAllowed)
		return
	}

	// Decode the JSON payload
	var req AIRequest
	err := json.NewDecoder(r.Body).Decode(&req)
	if err != nil {
		http.Error(w, "Invalid JSON payload", http.StatusBadRequest)
		return
	}

	// Validate the payload
	if req.Prompt == "" {
		http.Error(w, "Prompt is required", http.StatusBadRequest)
		return
	}

	// Translate the prompt
	translatedPrompt, err := translate.Translate(req.Prompt)
	if err != nil {
		http.Error(w, "Failed to translate prompt", http.StatusInternalServerError)
		return
	}

	// Select the AI provider
	provider, err := providers.SelectProvider(req.Model)
	if err != nil {
		http.Error(w, "Failed to select provider", http.StatusInternalServerError)
		return
	}

	// Forward the request to the provider
	resp, err := provider.Request(translatedPrompt)
	if err != nil {
		http.Error(w, "Failed to forward request", http.StatusInternalServerError)
		return
	}

	// Return the provider's response
	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(resp)
}

// Register the /ai endpoint
func init() {
	http.HandleFunc("/ai", handleAI)
}
```
