// Before
const { translate } = require('google-translate-api');

// After
const { translate } = require('google-translate-api');
const { Cache } = require('lru-cache');
const cache = new Cache({
  max: 1000, // max size of the cache
  maxAge: 1000 * 60 * 60, // cache expires after 1 hour
});

// Update the translate function to use the cache
async function translateText(text) {
  const cachedTranslation = cache.get(text);
  if (cachedTranslation) {
    return cachedTranslation;
  }

  try {
    const translation = await translate(text, { from: 'en', to: 'th' });
    cache.set(text, translation);
    return translation;
  } catch (error) {
    // Log the translation failure and retry once
    console.error('Translation failure:', error);
    await new Promise((resolve) => setTimeout(resolve, 1000));
    return await translateText(text);
  }
}

module.exports = { translateText };
