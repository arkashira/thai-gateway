// decodeChatRequest reads the request body, unmarshals known fields into ChatRequest,
// and preserves any unknown fields in RawPayload for downstream providers.
func decodeChatRequest(r *http.Request) (*ChatRequest, error) {
	// Limit body size to avoid abuse (1 MiB is ample for typical prompts)
	const maxBodySize = 1 << 20 // 1 MiB
	limitedReader := io.LimitReader(r.Body, maxBodySize)
	defer r.Body.Close()

	// First unmarshal into a map to capture all keys
	var raw map[string]any
	if err := json.NewDecoder(limitedReader).Decode(&raw); err != nil {
		return nil, err
	}

	// Extract known fields
	req := &ChatRequest{}
	if v, ok := raw["prompt"]; ok {
		if s, ok := v.(string); ok {
			req.Prompt = s
		}
	}
	if v, ok := raw["model"]; ok {
		if s, ok := v.(string); ok {
			req.Model = s
		}
	}
	if v, ok := raw["temperature"]; ok {
		if f, ok := v.(float64); ok {
			req.Temperature = &f
		}
	}
	if v, ok := raw["max_tokens"]; ok {
		// JSON numbers are float64; convert safely
		if f, ok := v.(float64); ok {
			i := int(f)
			req.MaxTokens = &i
		}
	}

	// Remove known keys so RawPayload contains only the rest
	delete(raw, "prompt")
	delete(raw, "model")
	delete(raw, "temperature")
	delete(raw, "max_tokens")
	if len(raw) > 0 {
		req.RawPayload = raw
	}

	return req, nil
}
