// decodeChatRequest reads the request body, unmarshals known fields into ChatRequest,
// and preserves any unknown fields in RawPayload for downstream providers.
func decodeChatRequest(r *http.Request) (*ChatRequest, error) {
	// Limit body size to avoid abuse (1 MiB is ample for typical prompts)
	const maxBodySize = 1 << 20 // 1 MiB
	limitedReader := io.LimitReader(r.Body, maxBodySize)
	defer r.Body.Close()

	// First unmarshal into a map to capture all keys
	var raw map[string]any
	if err := json.NewDecoder(limitedReader).Decode(&raw); err != nil {
		return nil, err
	}

	// Extract known fields
	req := &ChatRequest{}
	if v, ok := raw["prompt"]; ok {
		if s, ok := v.(string); ok {
			req.Prompt = s
		}
	}
	if v, ok := raw["model"]; ok {
		if s, ok := v.(string); ok {
			req.Model = s
		}
	}
	if v, ok := raw["temperature"]; ok {
		if f, ok := v.(float64); ok {
			req.Temperature = &f
		}
	}
	if v, ok := raw["max_tokens"]; ok {
		// JSON numbers are float64; convert safely
		if f, ok := v.(float64); ok {
			i := int(f)
			req.MaxTokens = &i
		}
	}

	// Remove known keys so RawPayload contains only the rest
	delete(raw, "prompt")
	delete(raw, "model")
	delete(raw, "temperature")
	delete(raw, "max_tokens")
	if len(raw) > 0 {
		req.RawPayload = raw
	}

	return req, nil
}

- **AC1**: A POST request to `/chat` with a JSON body containing `prompt` (string) and `model` (string) is accepted and returns HTTP 200.  
- **AC2**: The handler forwards the request to the translation middleware, which enriches the payload with a `translated_prompt` field.  
- **AC3**: The routing middleware selects the correct provider based on the `model` value and forwards the request.  
- **AC4**: The provider’s JSON response is returned to the client unchanged, preserving status code and headers.  
- **AC5**: Missing required fields (`prompt` or `model`) result in HTTP 400 with a clear error message.  
- **AC6**: Unsupported `model` values result in HTTP 404 with an appropriate error message.  
- **AC7**: The endpoint is protected against common security issues (e.g., JSON injection, request size limits).
