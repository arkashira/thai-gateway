```diff
+import { Provider } from '../types';
+import { fetch as nodeFetch } from 'node-fetch';
+
+interface Provider {
+  url: string;
+  status: 'healthy' | 'unhealthy';
+  lastChecked: Date;
+}
+
+export class HealthChecker {
+  private providers: Provider[] = [
+    { url: 'https://provider1.com/health', status: 'healthy', lastChecked: new Date() },
+    { url: 'https://provider2.com/health', status: 'healthy', lastChecked: new Date() },
+  ];
+
+  constructor() {
+    this.start();
+  }
+
+  private async checkProvider(provider: Provider): Promise<void> {
+    try {
+      const response = await nodeFetch(provider.url);
+      if (response.status === 200) {
+        provider.status = 'healthy';
+      } else {
+        provider.status = 'unhealthy';
+      }
+      provider.lastChecked = new Date();
+    } catch (error) {
+      provider.status = 'unhealthy';
+      provider.lastChecked = new Date();
+    }
+  }
+
+  private start(): void {
+    setInterval(async () => {
+      for (const provider of this.providers) {
+        await this.checkProvider(provider);
+      }
+    }, 30000); // 30 seconds
+  }
+
+  public getHealthStatus(): Provider[] {
+    return this.providers;
+  }
+}
+
+// Expose health status via endpoint (adjust as needed for your framework)
+export const healthEndpoint = (app: any) => {
+  app.get('/health', (req, res) => {
+    const healthChecker = new HealthChecker();
+    res.json(healthChecker.getHealthStatus());
+  });
+};
```

- **HC-03**: The routing layer excludes any provider whose status flag is `unhealthy` from selection.
- **HC-04**: The `/health` HTTP endpoint returns a JSON payload listing all providers with their current health status and the timestamp of the last check (HTTP 200).
- **HC-05**: If a provider transitions from unhealthy to healthy (or vice‑versa), the change is reflected in the `/health` response within the next scheduled check.
- **HC-06**: The health‑check scheduler does not crash the gateway process even if a provider’s health‑check request times out or throws an exception.
