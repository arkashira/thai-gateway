```go
// health/monitor.go
package health

import (
    "context"
    "fmt"
    "log"
    "sync"
    "time"

    "github.com/axentx/thai-gateway/pkg/provider"
)

// HealthMonitor is responsible for monitoring the health of providers.
type HealthMonitor struct {
    mu    sync.Mutex
    providers map[string]provider.Provider
    lastPing map[string]time.Time
}

// NewHealthMonitor returns a new instance of HealthMonitor.
func NewHealthMonitor(providers map[string]provider.Provider) *HealthMonitor {
    return &HealthMonitor{
        providers: providers,
        lastPing:  make(map[string]time.Time),
    }
}

// Run starts the health check goroutine.
func (hm *HealthMonitor) Run(ctx context.Context) {
    ticker := time.NewTicker(10 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-ticker.C:
            hm.healthCheck(ctx)
        case <-ctx.Done():
            return
        }
    }
}

// healthCheck checks the health of all providers and updates the last ping timestamp.
func (hm *HealthMonitor) healthCheck(ctx context.Context) {
    hm.mu.Lock()
    defer hm.mu.Unlock()

    for name, provider := range hm.providers {
        status := "healthy"
        if time.Since(hm.lastPing[name]).Seconds() > 30 {
            status = "unhealthy"
        }

        hm.lastPing[name] = time.Now()

        log.Printf("Provider %s is %s", name, status)
    }
}

// GetHealth returns the health status of all providers as a JSON array.
func (hm *HealthMonitor) GetHealth() ([]HealthStatus, error) {
    hm.mu.Lock()
    defer hm.mu.Unlock()

    healthStatus := make([]HealthStatus, 0, len(hm.providers))

    for name, provider := range hm.providers {
        status := "healthy"
        if time.Since(hm.lastPing[name]).Seconds() > 30 {
            status = "unhealthy"
        }

        healthStatus = append(healthStatus, HealthStatus{
            Name:  name,
            Status: status,
            LastPing: hm.lastPing[name].Unix(),
        })
    }

    return healthStatus, nil
}

type HealthStatus struct {
    Name  string `json:"name"`
    Status string `json:"status"`
    LastPing int64 `json:"last_ping"`
}
```

## 1. Acceptance Criteria  

- **Endpoint Response**  
  - `GET /health/providers` returns **HTTP 200** with a JSON array.  
  - Each array element contains:  
    - `providerName` (string)  
    - `status` (`"healthy"` or `"unhealthy"`)  
    - `lastPingTimestamp` (RFC3339 string).  

- **Health Determination**  
  - A provider is **healthy** if its `lastPingTimestamp` is **strictly less than 30 seconds** older than the current server time; otherwise it is **unhealthy**.  

- **Background Scheduler**  
  - A goroutine runs automatically **every 10 seconds**, updating an in‑memory map of provider health without external triggers.  

- **Graceful Shutdown**  
  - The goroutine listens for context cancellation and exits cleanly.  

- **Deterministic Time**  
  - Tests must use a mockable clock interface; no reliance on real wall‑clock time.

```go
package health

import (
	"context"

	"time"

// --- Clock abstraction -------------------------------------------------------

type Clock interface {
	Now() time.Time
}

type mockClock struct{ now time.Time }

func (m *mockClock) Now() time.Time { return m.now }
func (m *mockClock) Advance(d time.Duration) { m.now = m.now.Add(d) }

// --- Health status logic ----------------------------------------------------

func TestDetermineHealthStatus(t *testing.T) {
	now := time.Now()
	tests := []struct {
		name     string
		lastPing time.Time
		want     string
	}{
		{"just now", now, "healthy"},
		{"within 30s", now.Add(-29 * time.Second), "healthy"},
		{"exactly 30s", now.Add(-30 * time.Second), "unhealthy"},
		{"past 30s", now.Add(-31 * time.Second), "unhealthy"},
	}

	for _, tc := range tests {
		t.Run(tc.name, func(t *testing.T) {
			got := determineHealthStatus(tc.lastPing, now)
			if got != tc.want {
				t.Fatalf("expected %s, got %s", tc.want, got)
			}
		})
	}
}

// --- Background routine timing ----------------------------------------------

func TestHealthCheckRoutineFrequency(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	mc := &mockClock{now: time.Now()}
	updateCh := make(chan struct{}, 1)

	// Start the routine with the mock clock.
	go healthCheckRoutine(ctx, mc, updateCh)

	// First tick after 10s.
	mc.Advance(10 * time.Second)
	select {
	case <-updateCh:
	case <-time.After(100 * time.Millisecond):
		t.Fatalf("expected first tick after 10s, but none received")
	}

	// Second tick after another 10s.
	mc.Advance(10 * time.Second)
	select {
	case <-updateCh:
	case <-time.After(100 * time.Millisecond):
		t.Fatalf("expected second tick after 20s, but none received")
	}
}

// --- Graceful shutdown -------------------------------------------------------

func TestHealthCheckRoutineShutdown(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	mc := &mockClock{now: time.Now()}
	updateCh := make(chan struct{}, 1)

	go healthCheckRoutine(ctx, mc, updateCh)

	// Let one tick happen.
	mc.Advance(10 * time.Second)
	select {
	case <-updateCh:
	default:
		t.Fatalf("expected tick before shutdown")
	}

	// Cancel context and ensure goroutine exits.
	cancel()
	time.Sleep(50 * time.Millisecond) // give goroutine time to exit

	select {
	case <-updateCh:
		t.Fatalf("no more ticks should be sent after shutdown")
	default:
	}
}
```
