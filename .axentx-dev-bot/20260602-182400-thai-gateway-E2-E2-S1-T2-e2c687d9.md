// /opt/axentx/thai-gateway/src/middleware/promptParser.js

const { parse } = require('node-html-parser');
const axios = require('axios');
const { TranslateClient } = require('@google-cloud/translate');
const { CacheManager } = require('cache-manager');
const cache = new CacheManager({
  store: 'memory',
  maxAge: 1000 * 60 * 60, // 1 hour
});

const translateClient = new TranslateClient({
  projectId: 'your-project-id',
  keyFile: 'path/to/your/key.json',
});

// Function to extract text content from incoming prompts
function extractText(prompt) {
  const textRegex = /<[^>]*>(.*?)<\/[^>]*>/g;
  const text = prompt.match(textRegex);
  return text ? text[0] : prompt;
}

// Function to translate prompts to Thai
async function translatePrompt(prompt) {
  const text = extractText(prompt);
  const cacheKey = `translated-text-${text}`;
  const cachedTranslation = await cache.get(cacheKey);
  if (cachedTranslation) {
    return cachedTranslation;
  }

  try {
    const [result] = await translateClient.translate(text, 'th');
    const translatedText = result.translatedText;
    await cache.set(cacheKey, translatedText, 1000 * 60 * 60); // cache for 1 hour
    return translatedText;
  } catch (error) {
    console.error('Translation error:', error);
    // retry translation after 1 second
    await new Promise((resolve) => setTimeout(resolve, 1000));
    return translatePrompt(prompt);
  }
}

// Function to parse incoming prompts and translate them to Thai
async function parsePrompt(prompt) {
  const translatedPrompt = await translatePrompt(prompt);
  return translatedPrompt;
}

// Function to log and retry translation failures
async function logAndRetryTranslationFailure(error) {
  console.error('Translation error:', error);
  // retry translation after 1 second
  await new Promise((resolve) => setTimeout(resolve, 1000));
  return logAndRetryTranslationFailure(error);
}

// Function to implement a translation cache to reduce redundant API calls
function implementTranslationCache(prompt) {
  const cacheKey = `translated-text-${prompt}`;
  const cachedTranslation = cache.get(cacheKey);
  if (cachedTranslation) {
    return cachedTranslation;
  }
  return translatePrompt(prompt);
}

module.exports = { parsePrompt, logAndRetryTranslationFailure, implementTranslationCache };
