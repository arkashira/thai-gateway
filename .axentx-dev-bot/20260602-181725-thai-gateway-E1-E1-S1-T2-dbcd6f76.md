## 1. Acceptance Criteria
1. **Rolling 60‑second usage window** – The manager maintains a per‑provider request count for the last 60 seconds; `GetProviderUsage()` returns the exact count for each configured provider.  
2. **Least‑congested routing** – For every incoming request, `SelectProvider()` returns the provider with the smallest usage count **that has not exceeded its `QuotaPerMinute`**.  
3. **Queue on full quota** – If *all* providers have reached their quota, the request is placed in an in‑memory FIFO queue and automatically retried every 30 seconds until a provider becomes available.  
4. **Decision logging** – Each routing decision (including “queued” and “de‑queued/retried”) is logged with fields: `request_id`, `selected_provider_id` (or `queued`), and an ISO‑8601 timestamp.  
5. **Concurrency safety** – The manager can be accessed concurrently by ≥ 100 goroutines without data races or lost updates (verified by `go test -race`).  
6. **Queue size limit** – The queue has a configurable `MaxQueueSize` (default 10 000). When the limit is hit, the oldest entry is dropped and a warning is emitted.  
7. **Graceful shutdown** – On `Shutdown()`, the manager stops the retry ticker, flushes pending logs, and returns an error if any queued request could not be processed.

package ratelimit

import (
    "testing"
    "time"

    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"
)

// -------------------------------------------------------------------
// Helpers
// -------------------------------------------------------------------
func testConfig() Config {
    return Config{
        Providers: map[string]ProviderConfig{
            "thai-openai":  {QuotaPerMinute: 5},
            "thai-gemini":  {QuotaPerMinute: 5},
            "thai-azure":   {QuotaPerMinute: 5},
        },
        RetryInterval: 30 * time.Second,
        MaxQueueSize:  100,
    }
}

func newManager(t *testing.T) *Manager {
    m, err := NewManager(testConfig())
    require.NoError(t, err)
    return m
}

// -------------------------------------------------------------------
// Acceptance Criterion 1 – rolling window
// -------------------------------------------------------------------
func TestRecordAndCleanupUsageWindow(t *testing.T) {
    m := newManager(t)
    now := time.Now()

    // Record 3 requests for openai at time now
    for i := 0; i < 3; i++ {
        m.RecordUsage("thai-openai", now)
    }
    // Record 2 requests for gemini at now+10s
    for i := 0; i < 2; i++ {
        m.RecordUsage("thai-gemini", now.Add(10*time.Second))
    }

    usage := m.GetProviderUsage()
    assert.Equal(t, 3, usage["thai-openai"])
    assert.Equal(t, 2, usage["thai-gemini"])
    assert.Equal(t, 0, usage["thai-azure"])

    // Advance time beyond the 60‑second window
    later := now.Add(61 * time.Second)
    m.CleanupOldEntries(later)

    usage = m.GetProviderUsage()
    // All counts should have expired
    assert.Equal(t, 0, usage["thai-openai"])
    assert.Equal(t, 0, usage["thai-gemini"])
    assert.Equal(t, 0, usage["thai-azure"])
}

// -------------------------------------------------------------------
// Acceptance Criterion 2 – least‑congested selection
// -------------------------------------------------------------------
func TestSelectProviderLeastCongested(t *testing.T) {
    m := newManager(t)

    // Simulate usage: openai=4, gemini=2, azure=4 (quota=5)
    now := time.Now()
    for i := 0; i < 4; i++ { m.RecordUsage("thai-openai", now) }
    for i := 0; i < 2; i++ { m.RecordUsage("thai-gemini", now) }
    for i := 0; i < 4; i++ { m.RecordUsage("thai-azure", now) }

    provider, err := m.SelectProvider()
    require.NoError(t, err)
    // gemini has the lowest count (2)
    assert.Equal(t, "thai-gemini", provider.ID)
}
