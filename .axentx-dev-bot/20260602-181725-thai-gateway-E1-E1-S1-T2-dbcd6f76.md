# rate_limiter.py

import time
from config import config

class RateLimiter:
    def __init__(self):
        self.bucket_size = config['rate_limiting']['bucket_size']
        self.refill_rate = config['rate_limiting']['refill_rate']
        self.tokens = self.bucket_size
        self.last_refill = time.time()

    def allow_request(self):
        current_time = time.time()
        elapsed_time = current_time - self.last_refill
        self.tokens = min(self.bucket_size, self.tokens + elapsed_time * self.refill_rate)
        self.last_refill = current_time

        if self.tokens < 1:
            return False
        else:
            self.tokens -= 1
            return True

# Example usage:
rate_limiter = RateLimiter()

def handle_request():
    if rate_limiter.allow_request():
        # Process the request
        print("Request allowed")
    else:
        # Rate limit exceeded, handle the error
        print("Rate limit exceeded")

handle_request()
