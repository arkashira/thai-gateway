	name := os.Getenv("PROVIDER_NAME")

		rateLimit = 100 // Default rate limit
	}

	translationEnabled, err := strconv.ParseBool(translationEnabledStr)
	if err != nil {
		translationEnabled = false // Default translation status
	}

	return &Provider{
		Name:             name,
		Endpoint:         endpoint,
		RateLimit:        rateLimit,
		TranslationEnabled: translationEnabled,
		DeeplAPIKey:      deeplAPIKey,
	}, nil
}
