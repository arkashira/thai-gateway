```go
package middleware

import (
	"log"
	"net/http"
	"strings"
	"time"
)

// RequestLogger is a middleware that logs incoming requests with timestamp and provider routing
type RequestLogger struct {
	next http.Handler
}

// NewRequestLogger creates a new RequestLogger middleware with the given next handler
func NewRequestLogger(next http.Handler) *RequestLogger {
	return &RequestLogger{next: next}
}

// ServeHTTP implements the http.Handler interface
func (rl *RequestLogger) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	// Get provider endpoint from URL path
	parts := strings.Split(r.URL.Path, "/")
	provider := "unknown"
	if len(parts) > 1 && parts[1] != "" {
		provider = parts[1]
	}

	// Log request with timestamp and provider
	log.Printf("%s - [%s] \"%s %s\" from %s → provider=%s",
		time.Now().Format(time.RFC3339),
		r.Method,
		r.URL.Path,
		r.Proto,
		r.RemoteAddr,
		provider,
	)

	// Call the next handler
	rl.next.ServeHTTP(w, r)
}
```
