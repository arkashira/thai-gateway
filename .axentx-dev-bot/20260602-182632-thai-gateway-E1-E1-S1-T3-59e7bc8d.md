```go
package middleware

import (
	"log"
	"net/http"
	"strings"
	"time"
)

// RequestLogger logs incoming requests with timestamps and provider routing
type RequestLogger struct{}

// Next returns the next middleware in the chain
func (rl *RequestLogger) Next() http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		// Get the provider endpoint from the URL path
		endpoint := strings.Split(r.URL.Path, "/")[1]

		// Log the request
		log.Printf("%s %s %s %s\n",
			r.Method,
			r.URL.Path,
			endpoint,
			r.RemoteAddr,
		)

		// Call the next middleware in the chain
		rl.Next()(w, r)
	})
}

// NewRequestLogger returns a new RequestLogger instance
func NewRequestLogger() *RequestLogger {
	return &RequestLogger{}
```
```go
}
```

// middleware/logger_test.go
package middleware

import (
    "bytes"
    "io/ioutil"
    "net/http"
    "net/http/httptest"
    "os"
    "strings"
    "testing"
    "time"
)

// helper to create a dummy handler that returns 200 OK
func dummyHandler() http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.WriteHeader(http.StatusOK)
        w.Write([]byte(`{"status":"ok"}`))
    })
}

// Test that a request generates a log line with required fields
func TestLoggerMiddleware_WritesLogEntry(t *testing.T) {
    // use a temp file for logs
    tmpFile, err := ioutil.TempFile("", "gateway-log-*.log")
    if err != nil {
        t.Fatalf("cannot create temp log file: %v", err)
    }
    defer os.Remove(tmpFile.Name())
    os.Setenv("LOG_FILE", tmpFile.Name())

    // set provider env for routing detection
    os.Setenv("PROVIDER_1_URL", "http://provider1.example.com")
    os.Setenv("PROVIDER_2_URL", "http://provider2.example.com")

    // create router that selects provider based on path prefix
    // (the actual router is outside the scope of this unit test;
    // we simulate by setting a header that the logger reads)
    req := httptest.NewRequest("POST", "/v1/chat", nil)
    req.Header.Set("X-Provider-Id", "provider1") // our logger will read this header

    rr := httptest.NewRecorder()
    logger := NewRequestLogger(dummyHandler())
    logger.ServeHTTP(rr, req)

    // read log file
    data, _ := ioutil.ReadFile(tmpFile.Name())
    logLine := string(data)

    // verify required fields
    if !strings.Contains(logLine, "POST") {
        t.Error("log missing HTTP method")
    }
    if !strings.Contains(logLine, "/v1/chat") {
        t.Error("log missing request path")
    }
    if !strings.Contains(logLine, "provider1") {
        t.Error("log missing provider identifier")
    }
    if !strings.Contains(logLine, time.Now().Format("2006-01-02")) {
        t.Error("log missing timestamp")
    }
}

// Test that the middleware does not modify response body or status
func TestLoggerMiddleware_Passthrough(t *testing.T) {

    rr := httptest.NewRecorder()

    }
}
