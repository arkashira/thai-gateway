```python
# file: coverage_service/collector.py
import xml.etree.ElementTree as ET
import json, pathlib, boto3

def parse_cobertura(xml_path: str) -> dict:
    tree = ET.parse(xml_path)
    root = tree.getroot()
    cov = {}
    for cls in root.findall('.//class'):
        classname = cls.attrib['name']
        lines = {int(l.attrib['number']): l.attrib['hits'] != '0'
                 for l in cls.findall('lines/line')}
        cov[classname] = lines
    return cov

def upload_to_s3(cov_dict: dict, bucket: str, key: str):
    s3 = boto3.client('s3')
    s3.put_object(Body=json.dumps(cov_dict), Bucket=bucket, Key=key)

if __name__ == "__main__":
    xml = pathlib.Path("target/site/cobertura/coverage.xml")
    cov = parse_cobertura(str(xml))
    upload_to_s3(cov, bucket="atcga-coverage", key="build-{{BUILD_ID}}.json")
```

## 1. Acceptance Criteria  
(Each bullet is measurable and can be verified by the test suite)

| # | Criterion | Verification method |
|---|-----------|---------------------|
| 1 | **Quota persistence** – The gateway writes a Redis key `quota:{providerId}:{apiKey}` containing the remaining quota and a TTL. | Unit test checks that `redis.set` is called with correct key, value, and `EX` option. |
| 2 | **TTL enforcement** – The key expires after the configured TTL (default 3600 s). | Integration test uses a mock Redis that tracks TTL; verifies that after TTL the key is absent. |
| 3 | **Provider selection** – When a request would exceed a provider’s remaining quota, the gateway selects the next provider with `remaining > 0`. | Unit test simulates multiple providers; asserts that the returned provider is the first eligible one. |
| 4 | **Quota decrement** – Upon successful request, the remaining quota of the chosen provider is decremented by the request’s usage and saved back to Redis. | Unit test verifies that `redis.set` is called with the updated remaining value. |
| 5 | **All providers exhausted** – If every provider’s remaining quota is 0, the gateway returns HTTP 429 with a `Retry-After` header (value > 0). | Integration test triggers exhaustion and checks status code and header. |
| 6 | **Retry‑After calculation** – The `Retry-After` value equals the minimum TTL among the exhausted providers. | Unit test sets different TTLs and verifies the header value. |
| 7 | **Graceful error handling** – Redis connection errors do not crash the gateway; instead a 503 is returned. | Unit test simulates `redis.get` throwing and checks response. |

```js
// quota.test.js
const { storeQuota, getQuota, decrementQuota, selectProvider } = require('./quota');
const redisMock = require('./__mocks__/redisMock');

describe('Quota Service – unit', () => {
  const provider = { id: 'p1', apiKey: 'key1', limit: 100 };

  beforeEach(() => {
    redisMock.reset();
  });

  test('storeQuota writes key with correct TTL', async () => {
    await storeQuota(redisMock, provider, 80, 3600);

    const key = `quota:${provider.id}:${provider.apiKey}`;
    expect(redisMock.set).toHaveBeenCalledWith(
      key,
      JSON.stringify({ remaining: 80 }),
      'EX',
      3600
    );
  });

  test('getQuota returns parsed value', async () => {
    const key = `quota:${provider.id}:${provider.apiKey}`;
    redisMock.set(key, JSON.stringify({ remaining: 80 }));

    const result = await getQuota(redisMock, provider);
    expect(result).toEqual({ remaining: 80 });
  });

  test('decrementQuota updates remaining correctly', async () => {
    const key = `quota:${provider.id}:${provider.apiKey}`;
    redisMock.set(key, JSON.stringify({ remaining: 80 }));

    await decrementQuota(redisMock, provider, 15);

    const stored = JSON.parse(redisMock.get(key));
    expect(stored.remaining).toBe(65);
  });

  test('selectProvider picks first with remaining > 0', () => {
    const providers = [
      { id: 'p1', apiKey: 'k1', remaining: 0 },
      { id: 'p2', apiKey: 'k2', remaining: 5 },
      { id: 'p3', apiKey: 'k3', remaining: 10 }
    ];

    const chosen = selectProvider(providers, 3);
    expect(chosen.id).toBe('p2');
  });

  test('selectProvider returns null when none available', () => {
    const providers = [
      { id: 'p1', apiKey: 'k1', remaining: 0 },
      { id: 'p2', apiKey: 'k2', remaining: 0 }
    ];

    const chosen = selectProvider(providers, 1);
    expect(chosen).toBeNull();
  });
});
```
