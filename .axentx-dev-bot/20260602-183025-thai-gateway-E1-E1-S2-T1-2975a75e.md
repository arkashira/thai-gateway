use actix_web::{web, App, HttpResponse, HttpServer, Responder};

async fn health() -> impl Responder {
    HttpResponse::Ok().json({"status":"ok"})
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {

    .bind("127.0.0.1:8080")?
