+ const { v4: uuidv4 } = require('uuid');

+ // Initialize an object to store rate limiters per platform
+ const platformRateLimiters = {};

+ function createPlatformRateLimiter(platformId) {
+   return rateLimit({
+     windowMs: 15 * 60 * 1000, // 15 minutes
+     max: 100, // Limit each IP to 100 requests per windowMs
+     message: 'Too many requests from this platform, please try again later.',
+     standardHeaders: true, // Return rate limit info in the `RateLimit-*` headers

+   });
+ }

+ function getPlatformRateLimiter(platformId) {
+   if (!platformRateLimiters[platformId]) {
+     platformRateLimiters[platformId] = createPlatformRateLimiter(platformId);
+   }
+   return platformRateLimiters[platformId];
+ }

## 1. Acceptance Criteria  
(Each bullet is measurable and testable)

1. **API‑Key Generation** – On the first registration request, the service returns a non‑empty, unique API key that is persisted and associated with the requesting platform.  
2. **Key Scope Isolation** – A request authenticated with a platform’s API key can only access resources belonging to that platform; any attempt to use the key on another platform’s resources results in HTTP 401.  
3. **Per‑Platform Rate Limit** – Each platform is limited to a configurable number of requests per minute (default 120 rpm). Exceeding the limit returns HTTP 429 with a `Retry-After` header indicating the remaining time until the window resets.  
4. **Independent Quotas** – Traffic from Platform A does not affect the quota of Platform B; each platform’s counter is isolated.  
5. **Rate‑Limit Reset** – After the time window expires, the platform’s remaining quota resets to the configured value.  
6. **Documentation** – The `README.md` contains a section “Onboarding & Rate Limiting” that describes the registration endpoint, how to retrieve the API key, and the default rate‑limit values.

// tests/unit/platformRateLimiter.test.js
import { PlatformRateLimiter } from '../../src/providers/platformRateLimiter';
import { createPlatform, getPlatformByKey } from '../../src/models/platform';
import { jest } from '@jest/globals';

describe('PlatformRateLimiter', () => {
  let limiter;
  const DEFAULT_RPM = 120;

  beforeEach(() => {
    jest.useFakeTimers();
    limiter = new PlatformRateLimiter({ rpm: DEFAULT_RPM });
    // reset in‑memory store
    limiter.resetStore();
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  test('generates a new API key on first registration', async () => {
    const platform = await createPlatform({ name: 'ShopA' });
    expect(platform.apiKey).toMatch(/^pk_[a-z0-9]{32}$/);
    const stored = await getPlatformByKey(platform.apiKey);
    expect(stored.name).toBe('ShopA');
  });

  test('rejects request with unknown API key', async () => {
    const result = await limiter.check('invalid_key');
    expect(result.allowed).toBe(false);
    expect(result.status).toBe(401);
  });

  test('allows request when under quota', async () => {
    const { apiKey } = await createPlatform({ name: 'ShopB' });
    const result = await limiter.check(apiKey);
    expect(result.allowed).toBe(true);
    expect(result.remaining).toBe(DEFAULT_RPM - 1);
  });

  test('blocks request when quota exceeded', async () => {
    const { apiKey } = await createPlatform({ name: 'ShopC' });
    // consume full quota
    for (let i = 0; i < DEFAULT_RPM; i++) {
      await limiter.check(apiKey);
    }
    const over = await limiter.check(apiKey);
    expect(over.allowed).toBe(false);
    expect(over.status).toBe(429);
    expect(over.retryAfter).toBeGreaterThan(0);
  });

  test('rate‑limit resets after window', async () => {
    const { apiKey } = await createPlatform({ name: 'ShopD' });
    for (let i = 0; i < DEFAULT_RPM; i++) {
      await limiter.check(apiKey);
    }
    // advance time by 61 seconds to cross the 1‑minute window
    jest.advanceTimersByTime(61_000);
    const afterReset = await limiter.check(apiKey);
    expect(afterReset.allowed).toBe(true);
    expect(afterReset.remaining).toBe(DEFAULT_RPM - 1);
  });

  test('platform quotas are isolated', async () => {
    const { apiKey: keyA } = await createPlatform({ name: 'ShopE' });
    const { apiKey: keyB } = await createPlatform({ name: 'ShopF' });

    // exhaust A
    for (let i = 0; i < DEFAULT_RPM; i++) {
      await limiter.check(keyA);
    }
    const overA = await limiter.check(keyA);
    expect(overA.allowed).toBe(false);

    // B should still be allowed
    const okB = await limiter.check(keyB);
    expect(okB.allowed).toBe(true);
  });
});
