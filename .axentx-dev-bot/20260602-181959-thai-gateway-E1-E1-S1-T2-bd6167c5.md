+type ProviderConfig struct {
+    ProviderID string  `json:"provider_id"`
+    BaseURL    string  `json:"base_url"`
+    RateLimit  int     `json:"rate_limit"`
+    Weight     float64 `json:"weight"`
+}
+
+func AddProvider(w http.ResponseWriter, r *http.Request) {
+    var config ProviderConfig
+    err := json.NewDecoder(r.Body).Decode(&config)
+    if err != nil {
+        http.Error(w, "Invalid JSON", http.StatusBadRequest)
+        return
+    }
+
+    // Validate config
+    if config.ProviderID == "" || config.BaseURL == "" || config.RateLimit <= 0 || config.Weight <= 0 {
+        http.Error(w, "Invalid configuration", http.StatusBadRequest)
+        return
+    }
+
+    db.AddProvider(config)
+    w.WriteHeader(http.StatusCreated)
+}

    cases := []struct {

        payload   ProviderConfig
        wantErr   bool
        errMsg    string
    }{

            ProviderID: "openai", BaseURL: "https://api.openai.com", RateLimit: 1200, Weight: 10},
            false, ""},

        {"zero rate_limit", ProviderConfig{
            ProviderID: "anthropic", BaseURL: "https://api.anthropic.com", RateLimit: 0, Weight: 5},
            true, "rate_limit must be > 0"},
        {"negative weight", ProviderConfig{
            ProviderID: "cohere", BaseURL: "https://api.cohere.ai", RateLimit: 500, Weight: -1},
            true, "weight must be >= 0"},
        {"excessive rate_limit", ProviderConfig{
            ProviderID: "gemini", BaseURL: "https://api.gemini.com", RateLimit: 70000, Weight: 1},
            true, "rate_limit exceeds maximum"},
    }

    for _, tc := range cases {
        t.Run(tc.name, func(t *testing.T) {
            err := validateProviderConfig(&tc.payload)
            if tc.wantErr {
                assert.Error(t, err)
                assert.Contains(t, err.Error(), tc.errMsg)
            } else {
                assert.NoError(t, err)
            }
        })
    }
}
