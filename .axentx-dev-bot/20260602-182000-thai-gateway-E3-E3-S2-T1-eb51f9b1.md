// logger.go
package logger

import (
	"context"
	"log"
	"os"
	"time"

	"github.com/sirupsen/logrus"
)

var (
	logLevel = logrus.DebugLevel
	logger   = logrus.New()
)

func init() {
	// Set log level to debug by default, but allow it to be configurable via env vars
	logLevelStr := os.Getenv("LOG_LEVEL")
	if logLevelStr == "" {
		logger.SetLevel(logrus.DebugLevel)
	} else {
		switch logLevelStr {
		case "debug":
			logger.SetLevel(logrus.DebugLevel)
		case "info":
			logger.SetLevel(logrus.InfoLevel)
		case "warn":
			logger.SetLevel(logrus.WarnLevel)
		case "error":
			logger.SetLevel(logrus.ErrorLevel)
		default:
			logger.SetLevel(logrus.DebugLevel)
		}
	}
	logger.SetOutput(os.Stdout)
	logger.SetFormatter(&logrus.TextFormatter{
		TimestampFormat: time.RFC3339Nano,
	})
}

func LogRequest(ctx context.Context, providerID string, requestId string, translationStatus string) {
	logger.WithFields(logrus.Fields{
		"provider_id": providerID,
		"request_id":  requestId,
		"translation_status": translationStatus,
	}).Info("Request received")
}

func LogResponse(ctx context.Context, providerID string, requestId string, translationStatus string) {
	logger.WithFields(logrus.Fields{
		"provider_id": providerID,
		"request_id":  requestId,
		"translation_status": translationStatus,
	}).Info("Response received")
}
```
