+	"strings"

+// SanitizeInput sanitizes the input to prevent injection attacks.
+func SanitizeInput(input string) string {
+	// Trim whitespace
+	input = strings.TrimSpace(input)
+
+	// Remove potentially harmful characters (e.g., <, >, &, ;, ', \)
+	re := regexp.MustCompile(`[<>;&'\\]`)
+	input = re.ReplaceAllString(input, "")
+
+	// Optionally, you can add more validation based on expected input format
+	// For example, if expecting alphanumeric input:
+	reAlphaNum := regexp.MustCompile(`[^a-zA-Z0-9 ]`)
+	input = reAlphaNum.ReplaceAllString(input, "")
+
+	return input
+}
+
+// LogSensitiveInfo masks sensitive information in logs

+	// Mask sensitive information
