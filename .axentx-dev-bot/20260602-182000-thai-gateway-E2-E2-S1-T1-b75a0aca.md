package translator

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"net/http"

	"github.com/huaweicloud/huaweicloud-sdk-go-v5/core/auth/basic"
	"github.com/huaweicloud/huaweicloud-sdk-go-v5/core/auth/impl/http/impl"
)

const (
	deeplAPIKey = "YOUR_DEEPL_API_KEY"
	deeplSecret = "YOUR_DEEPL_SECRET"
)

type DeeplClient struct {
	client *basic.AuthenticatedClient
}

func NewDeeplClient() (*DeeplClient, error) {
	auth := basic.NewCredentials(
		deeplAPIKey,
		deeplSecret,
	)
	client := basic.NewClient(auth, impl.NewHTTPClient())
	return &DeeplClient{client: client}, nil
}

func (d *DeeplClient) Translate(ctx context.Context, text string) (string, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodPost, "https://api.deepl.com/v2/translate", strings.NewReader(`{"text": "`+text+`","target_lang": "TH"}`))
	if err != nil {
		return "", err
	}
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", "Bearer "+deeplAPIKey)

	resp, err := d.client.Do(req)
	if err != nil {
		return "", err
	}
	defer resp.Body.Close()

	var result struct {
		Translations []struct {
			Text string `json:"text"`
		} `json:"translations"`
	}
	err = json.NewDecoder(resp.Body).Decode(&result)
	if err != nil {
		return "", err
	}
	return result.Translations[0].Text, nil
}
```
