```diff
+# src/rate_limiter.py
+"""
+Simple in‑memory rate limiter that limits the number of calls to a function
+within a sliding time window.  It is intentionally lightweight – suitable
+for a single‑process gateway – and is used by the translation service.
+"""
+from __future__ import annotations
+
+from functools import wraps
+from time import time
+from typing import Callable, List
+
+
+class RateLimiter:
+    """
+    Limit *max_requests* calls to a function within *period* seconds.
+    Raises ``RuntimeError`` when the limit is exceeded.
+    """
+
+    def __init__(self, max_requests: int, period: int):
+        self.max_requests = max_requests
+        self.period = period
+        self._timestamps: List[float] = []
+
+    def __call__(self, func: Callable) -> Callable:
+        @wraps(func)
+        def wrapper(*args, **kwargs):
+            now = time()
+            # Keep only timestamps inside the window
+            self._timestamps = [t for t in self._timestamps if now - t < self.period]
+            if len(self._timestamps) >= self.max_requests:
+                raise RuntimeError(
+                    f"Rate limit exceeded: {self.max_requests} requests per {self.period}s"
+                )
+            self._timestamps.append(now)
+            return func(*args, **kwargs)
+
+        return wrapper
+
+__all__ = ["RateLimiter"]
```

## 1. Acceptance Criteria  
1. **POST /ai** accepts a JSON payload that **must contain a string field `prompt`**; any other shape returns **400 Bad Request**.  
2. The handler forwards the `prompt` to the configured AI provider (URL taken from `AI_PROVIDER_URL` env‑var) using an **HTTP POST with JSON `{ "prompt": "..."}**`.  
3. The gateway returns **exactly the body** received from the AI provider **without modification** (including content‑type).  
4. On successful forward the endpoint responds with **200 OK** and the provider’s payload.  
5. If the provider returns a non‑2xx status, the gateway propagates that status and body unchanged.  
6. Network or serialization failures inside the gateway result in **502 Bad Gateway**.  
7. All responses include **`Content-Type: application/json`** (or the provider’s content‑type when propagated).
