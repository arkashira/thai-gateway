// /opt/axentx/thai-gateway/src/services/translator.rs

use log::{debug, info};
use tencent_translation::{Translator, Language};
use tokio::time::{Instant, Duration};
use std::error::Error;

pub struct ThaiTranslator {
    translator: Translator,
}

impl ThaiTranslator {
    pub fn new() -> Self {
        ThaiTranslator {
            translator: Translator::new(Language::English, Language::Thai),
        }
    }

    pub async fn translate(&self, prompt: &str) -> Result<String, Box<dyn Error>> {
        if !self.is_english(prompt) {
            return Ok(prompt.to_string());
        }

        info!("Original English prompt: {}", prompt);

        let start = Instant::now();
        let translation = self.translator.translate(prompt).await?;

        let latency = start.elapsed();
        debug!("Translation latency: {:?}", latency);

        if latency > Duration::from_millis(200) {
            return Err("Translation latency exceeded 200 ms".into());
        }

        Ok(translation)
    }

    fn is_english(&self, prompt: &str) -> bool {
        prompt.chars().all(|c| c.is_ascii())
    }
}
```

Update the `main.rs` file to integrate the translation service:

// /opt/axentx/thai-gateway/src/main.rs

use axentx::gateway::{Gateway, Request, Response};
use axentx::services::translator::*;
use log::info;
use std::time::Instant;
use tokio::time::Duration;

#[tokio::main]
async fn main() -> Result<(), Box<dyn Error>> {
    env_logger::init();

    let mut gateway = Gateway::new();

    let translator = ThaiTranslator::new();
    gateway.add_service("translator", Box::new(translator));

    gateway.add_middleware(Box::new(move |req: Request, next| {
        let start = Instant::now();
        let original = req.body().clone();
        let translated = match translator.translate(&original).await {
            Ok(t) => t,
            Err(e) => {
                info!("Translation failed: {}", e);
                return Response::error("Translation failed");
            }
        };

        let duration = start.elapsed();
        if duration > Duration::from_millis(200) {
            return Response::error("Translation timeout exceeded 200ms");
        }

        next(Request::new(translated)).await
    }));

    gateway.run().await?;
    Ok(())
}
```
