### 1. `src/middleware/metrics.ts`

```ts
// src/middleware/metrics.ts
import { Request, Response, NextFunction } from 'express';
import client from 'prom-client';

/**
 * 1️⃣ Create a dedicated registry so we can expose it later.
 */
export const metricsRegistry = new client.Registry();

/**
 * 2️⃣ Define the metrics we need.
 */
export const totalRequests = new client.Counter({
  name: 'thai_gateway_requests_total',
  help: 'Total number of requests received',
});

export const successRequests = new client.Counter({
  name: 'thai_gateway_requests_success_total',
  help: 'Number of successful (2xx) responses',
});

export const latencyHistogram = new client.Histogram({
  name: 'thai_gateway_requests_latency_seconds',
  help: 'Request latency in seconds',
  // 50 ms → 5 s buckets – tweak if your traffic differs
  buckets: [0.05, 0.1, 0.2, 0.5, 1, 2, 5],
});

export const translationStatus = new client.Counter({
  name: 'thai_gateway_translation_status',
  help: 'Translation attempt status',
  labelNames: ['status'],
});

/**
 * 3️⃣ Register the metrics with the registry.
 */
metricsRegistry.registerMetric(totalRequests);
metricsRegistry.registerMetric(successRequests);
metricsRegistry.registerMetric(latencyHistogram);
metricsRegistry.registerMetric(translationStatus);

/**
 * 4️⃣ Middleware that captures metrics and logs.
 */
export function metricsMiddleware(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const startHrTime = process.hrtime(); // high‑resolution time

  // Increment the total request counter immediately
  totalRequests.inc();

  // Hook into the response finish event
  res.on('finish', () => {
    const elapsedHrTime = process.hrtime(startHrTime);
    const elapsedSeconds =
      elapsedHrTime[0] + elapsedHrTime[1] / 1e9; // seconds

    // Record latency
    latencyHistogram.observe(elapsedSeconds);

    // Record success counter
    if (res.statusCode >= 200 && res.statusCode < 300) {
      successRequests.inc();
    }

    // Log provider, latency, and translation status
    const provider = req.headers['x-provider'] ?? 'unknown';
    const translation = (req as any).translationStatus ?? 'unknown';
    console.log(
      `Provider: ${provider}, Latency: ${elapsedSeconds.toFixed(
        3
      )}s, Translation: ${translation}`
    );
  });

  next();
}
```

> **Why this design?**  
> * `prom-client` v14+ automatically exposes the default `collectDefaultMetrics()` if you want node‑level metrics (CPU, memory, etc.).  
> * We expose a **separate registry** (`metricsRegistry`) so that the `/metrics` route can serve only the gateway‑specific metrics, keeping the default registry clean for other libraries.
