package rate_limit

import (
	"fmt"
	"strings"

	"github.com/gorilla/mux"
)

// ParseXRateLimitHeaders parses X-RateLimit headers from a response.
func ParseXRateLimitHeaders(response *mux.Response) (map[string]int, error) {
	headers := response.HeaderMap

	rateLimitHeaders := make(map[string]int)

	for header, values := range headers {
		if strings.HasPrefix(header, "X-RateLimit-") {
			parts := strings.SplitN(values[0], "-", 2)
			if len(parts) != 2 {
				return nil, fmt.Errorf("invalid X-RateLimit header: %s", header)
			}

			limit, err := strconv.Atoi(parts[1])
			if err != nil {
				return nil, err
			}

			rateLimitHeaders[header] = limit
		}
	}

	return rateLimitHeaders, nil
}
