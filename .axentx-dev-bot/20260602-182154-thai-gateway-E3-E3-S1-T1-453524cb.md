+/**
+ * Prometheus metrics for thai‑gateway.

+  buckets: [0.05, 0.1, 0.25, 0.5, 1, 2, 5],
+});
+
+// Gauge: remaining quota per provider
+const providerQuota = new client.Gauge({
+  name: 'gateway_provider_quota_remaining',
+  help: 'Remaining quota for each AI provider',
+  labelNames: ['provider'],
+});
+
+// ---- Helper functions ---------------------------------------------------
+
+/** Increment the total request counter. */
+function incRequest() {
+  totalRequests.inc();
+}
+
+/** Record latency for a given provider. */
+function observeLatency(provider, seconds) {
+  providerLatency.labels(provider).observe(seconds);
+}
+
+/** Set the remaining quota for a provider. */
+function setQuota(provider, remaining) {
+  providerQuota.labels(provider).set(remaining);
+}
+
+module.exports = {
+  register: client.register,
+  incRequest,
+  observeLatency,
+  setQuota,
+};

## 1. Acceptance Criteria
- **AC‑1**: `GET /metrics` returns HTTP 200 with `Content-Type: text/plain; version=0.0.4`.
- **AC‑2**: The response body contains a Prometheus **counter** named `gateway_requests_total` with a `provider` label for every provider that has handled at least one request.
- **AC‑3**: The response body contains a **histogram** named `gateway_request_latency_seconds` with the same `provider` label and default buckets (`0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5, 10`).
- **AC‑4**: The response body contains a **gauge** named `gateway_quota_remaining` with a `provider` label reflecting the current quota left for that provider.
- **AC‑5**: Metrics are updated in real‑time: a request to any provider increments the counter, records latency, and decrements the quota gauge.
- **AC‑6**: If the Prometheus client throws an unexpected error while gathering metrics, the endpoint returns HTTP 500 with a JSON error payload `{ error: "Metrics collection failed" }`.
- **AC‑7**: The `/metrics` route is mounted **before** any authentication middleware so that Prometheus can scrape without credentials.
