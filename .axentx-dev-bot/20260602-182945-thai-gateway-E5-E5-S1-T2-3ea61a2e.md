+package api
+
+import (
+	"net/http"
+
+	"github.com/prometheus/client_golang/prometheus"
+	"github.com/prometheus/client_golang/prometheus/promhttp"
+)
+
+var (
+	totalRequests = prometheus.NewCounterVec(
+		prometheus.CounterOpts{
+			Name: "total_requests",
+			Help: "Total number of requests processed.",
+		},
+		[]string{"method"},
+	)
+
+	requestLatency = prometheus.NewHistogramVec(
+		prometheus.HistogramOpts{
+			Name:    "request_latency_seconds",
+			Help:    "Request latency in seconds.",
+			Buckets: []float64{0.1, 0.5, 1, 2, 5},
+		},
+		[]string{"method"},
+	)
+
+	providerQuotaRemaining = prometheus.NewGaugeVec(
+		prometheus.GaugeOpts{
+			Name: "provider_quota_remaining",
+			Help: "Remaining quota for the provider.",
+		},
+		[]string{"provider"},
+	)
+
+	healthStatus = prometheus.NewGauge(
+		prometheus.GaugeOpts{
+			Name: "health_status",
+			Help: "Health status of the gateway (1=healthy, 0=unhealthy).",
+		},
+	)
+)
+
+func init() {
+	prometheus.MustRegister(totalRequests, requestLatency, providerQuotaRemaining, healthStatus)
+}
+
+func MetricsHandler() http.HandlerFunc {
+	return promhttp.Handler().ServeHTTP
+}
```
