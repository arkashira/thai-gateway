package services

import (
	"context"
	"fmt"
	"log"
	"time"

	"golang.org/x/text/language"
	"golang.org/x/text/transform"
	"golang.org/x/text/unicode/norm"

	"cloud.google.com/go/translate/apiv3"
	translatepb "cloud.google.com/go/translate/apiv3/protocol"
)

// TranslateService handles translation from English to Thai
type TranslateService struct {
	client *translatepb.TranslateClient
}

// NewTranslateService creates a new TranslateService with Google Translate API client
func NewTranslateService(ctx context.Context) (*TranslateService, error) {
	client, err := translatepb.NewTranslateClient(ctx)
	if err != nil {
		return nil, fmt.Errorf("failed to create translate client: %v", err)
	}
	return &TranslateService{client: client}, nil
}

// TranslateEnglishToThai translates English text to Thai
func (s *TranslateService) TranslateEnglishToThai(ctx context.Context, text string) (string, error) {
	startTime := time.Now()

	// Prepare request
	req := &translatepb.TranslateTextRequest{
		Parent: "projects/your-project-id/locations/global",
		SourceLanguageCode: "en",
		TargetLanguageCode: "th",
		Format: "text",
		SourceText: text,
	}

	// Perform translation
	resp, err := s.client.TranslateText(ctx, req)
	if err != nil {
		return "", fmt.Errorf("translation failed: %v", err)
	}

	// Log translation details
	log.Printf("Translation completed in %v: %s -> %s", time.Since(startTime), text, resp.GetTranslations()[0].GetTranslatedText())

	return resp.GetTranslations()[0].GetTranslatedText(), nil
}

// TranslateChatRequest translates the entire chat request body
func (s *TranslateService) TranslateChatRequest(ctx context.Context, requestBody string) (string, error) {
	startTime := time.Now()

	// Log original request
	log.Printf("Original request body: %s", requestBody)

	// Translate the request
	translated, err := s.TranslateEnglishToThai(ctx, requestBody)
	if err != nil {
		return "", fmt.err
	}

	// Log translation details
	log.Printf("Translation completed in %v: %s", time.Since(startTime), translated)

	return translated, nil
}
```
