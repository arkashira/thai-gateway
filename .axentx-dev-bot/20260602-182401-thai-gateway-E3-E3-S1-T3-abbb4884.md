// /opt/axentx/thai-gateway/src/alerts/capacityAlert.js
const axios = require('axios');

class CapacityAlert {
  constructor(providers) {
    this.providers = providers;
    this.remainingLimits = {};
  }

  async trackRemainingLimits() {
    for (const provider of this.providers) {
      const response = await axios.get(`${provider.url}/rate-limit`);
      this.remainingLimits[provider.name] = response.data.remaining;
    }
  }

  async sendAlerts() {
    for (const provider of this.providers) {
      if (this.remainingLimits[provider.name] <= 0) {
        // Send alert to notification system
        console.log(`Provider ${provider.name} has reached its capacity`);
      }
    }
  }
}

module.exports = CapacityAlert;
```

// /opt/axentx/thai-gateway/src/alerts/providerPool.js
class ProviderPool {
  constructor(providers, config) {
    this.providers = providers;
    this.config = config;
    this.pool = [];
  }

  async initializePool() {
    for (const provider of this.providers) {
      this.pool.push({ provider, available: true });
    }
  }

  async rejectRequest() {
    for (const provider of this.pool) {
      if (provider.available) {
        return false; // Request can be processed by available provider
      }
    }
    return true; // All providers are at capacity, reject request
  }

  async updatePool() {
    for (const provider of this.pool) {
      const response = await axios.get(`${provider.provider.url}/rate-limit`);
      if (response.data.remaining > 0) {
        provider.available = true;
      } else {
        provider.available = false;
      }
    }
  }
}

module.exports = ProviderPool;
```
