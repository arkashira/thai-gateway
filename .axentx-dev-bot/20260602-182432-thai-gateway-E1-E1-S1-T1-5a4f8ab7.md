+    "io"
+    "log"
+    "net/http"
+    "net/http/httputil"
+    "net/url"
+    "os"
+    "sync/atomic"
+    "time"
+
+    "gopkg.in/yaml.v3"
+)
+
+// ---------- Configuration structures ----------
+type providerCfg struct {
+    Name string `yaml:"name"`
+    URL  string `yaml:"url"`
+}
+
+type configFile struct {
+    Providers []providerCfg `yaml:"providers"`
+}
+
+// ---------- Proxy server ----------
+type Proxy struct {
+    // slice of parsed provider URLs (read‑only after init)
+    providers []*url.URL
+    // round‑robin counter (atomic for safety under concurrency)
+    rrCounter uint64
+}
+
+// NewProxy builds a Proxy from a list of provider URLs.
+func NewProxy(providers []*url.URL) *Proxy {
+    return &Proxy{providers: providers}
+}
+
+// nextProvider returns the next healthy provider using simple round‑robin.
+// If the chosen provider does not answer a HEAD request within 200 ms,
+// we skip it and try the next one (up to len(providers) attempts).
+func (p *Proxy) nextProvider() (*url.URL, error) {
+    n := len(p.providers)
+    for i := 0; i < n; i++ {
+        idx := int(atomic.AddUint64(&p.rrCounter, 1) % uint64(n))
+        candidate := p.providers[idx]
+
+        // Quick health‑check – non‑blocking HEAD request with short timeout.
+        client := http.Client{Timeout: 200 * time.Millisecond}
+        if resp, err := client.Head(candidate.String()); err == nil && resp.StatusCode < 500 {
+            resp.Body.Close()
+            return candidate, nil
+        }
+        // otherwise try the next one
+    }
+    return nil, http.ErrServerClosed // generic “no healthy provider”
+}
+
+// ServeHTTP satisfies http.Handler. It creates a reverse‑proxy on‑the‑fly
+// for the selected provider and forwards the request.
+func (p *Proxy) ServeHTTP(w http.ResponseWriter, r *http.Request) {
+    target, err := p.nextProvider()
+    if err != nil {
+        http.Error(w, "Service Unavailable – no healthy AI provider", http.StatusServiceUnavailable)
+        return

+        req.URL.Host = target.Host

+        // keep query string intact
