Create a new file `config/config.py` with the following content:

```python
import yaml
import os

class Config:
    def __init__(self):
        self.provider_endpoints = {}
        self.translation_service_api_key = None

    def load_config(self):
        try:
            with open('config.yaml', 'r') as f:
                config = yaml.safe_load(f)
                self.provider_endpoints = config['provider_endpoints']
                self.translation_service_api_key = config['translation_service_api_key']
        except FileNotFoundError:
            print("Config file not found. Using environment variables instead.")
            self.load_env_vars()

    def load_env_vars(self):
        self.provider_endpoints = {
            'provider1': {
                'endpoint': os.environ.get('PROVIDER1_ENDPOINT'),
                'rate_limit': int(os.environ.get('PROVIDER1_RATE_LIMIT'))
            },
            'provider2': {
                'endpoint': os.environ.get('PROVIDER2_ENDPOINT'),
                'rate_limit': int(os.environ.get('PROVIDER2_RATE_LIMIT'))
            }
        }
        self.translation_service_api_key = os.environ.get('TRANSLATION_SERVICE_API_KEY')
```

// __tests__/config.test.js
const { loadConfig } = require('../src/config');

describe('Configuration Loader', () => {
  const ORIGINAL_ENV = process.env;

  beforeEach(() => {
    jest.resetModules();
    process.env = { ...ORIGINAL_ENV };
  });

  afterAll(() => {
    process.env = ORIGINAL_ENV;
  });

  test('parses valid YAML file', () => {
    const yaml = `
      providers:
        translation: "https://api.trans.example"
        payment: "https://api.pay.example"
      rate_limits:
        translation: 100
        payment: 50
    `;
    const config = loadConfig({ yamlContent: yaml, env: {} });
    expect(config.providers.translation).toBe('https://api.trans.example');
    expect(config.providers.payment).toBe('https://api.pay.example');
    expect(config.rate_limits.translation).toBe(100);
    expect(config.rate_limits.payment).toBe(50);
  });

  test('environment variables override YAML', () => {
    const yaml = `
      providers:
        translation: "https://api.trans.example"
    `;
    const env = { THAI_GATEWAY_TRANSLATION_ENDPOINT: 'https://env.trans' };
    const config = loadConfig({ yamlContent: yaml, env });
    expect(config.providers.translation).toBe('https://env.trans');
  });

  test('falls back to defaults on invalid YAML', () => {
    const yaml = 'invalid: : yaml';
    const config = loadConfig({ yamlContent: yaml, env: {} });
    expect(config.providers.translation).toBe('https://default-translation.com');
    // Expect a warning log (mocked)
  });

  test('provides defaults when YAML missing', () => {
    const config = loadConfig({ yamlContent: null, env: {} });
    expect(config.providers.translation).toBe('https://default-translation.com');
  });
});
