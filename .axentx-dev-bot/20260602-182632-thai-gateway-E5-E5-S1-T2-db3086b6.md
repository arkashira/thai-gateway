Create a new file `config/config.py` with the following content:

```python
import yaml
import os

class Config:
    def __init__(self):
        self.provider_endpoints = {}
        self.translation_service_api_key = None

    def load_config(self):
        try:
            with open('config.yaml', 'r') as f:
                config = yaml.safe_load(f)
                self.provider_endpoints = config['provider_endpoints']
                self.translation_service_api_key = config['translation_service_api_key']
        except FileNotFoundError:
            print("Config file not found. Using environment variables instead.")
            self.load_env_vars()

    def load_env_vars(self):
        self.provider_endpoints = {
            'provider1': {
                'endpoint': os.environ.get('PROVIDER1_ENDPOINT'),
                'rate_limit': int(os.environ.get('PROVIDER1_RATE_LIMIT'))
            },
            'provider2': {
                'endpoint': os.environ.get('PROVIDER2_ENDPOINT'),
                'rate_limit': int(os.environ.get('PROVIDER2_RATE_LIMIT'))
            }
        }
        self.translation_service_api_key = os.environ.get('TRANSLATION_SERVICE_API_KEY')
```
