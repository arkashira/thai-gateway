## 1. Acceptance Criteria  
- **Real‑time tracking**: The system must maintain an accurate count of remaining requests for each provider and update it immediately after each request.  
- **Collective rejection**: A request is rejected **only** when *all* providers have exhausted their remaining quota.  
- **Header propagation**: The `X-RateLimit-Remaining` header returned to the client must reflect the *lowest* remaining quota among all providers (i.e., the effective pool limit).  
- **Configurable limits**: The API endpoint `/api/providers/:id/ratelimit` must allow an admin to set or update the maximum request count and reset window for a provider.  
- **Reset logic**: When the reset window elapses, the provider’s remaining count must reset to the configured maximum.  
- **Edge handling**: Requests arriving exactly at the reset boundary should be treated as within the new window.  
- **Concurrency safety**: Concurrent requests must not cause race conditions that allow the quota to be exceeded.

// rateTracker.test.js
import { RateTracker } from '../src/rateLimiter/rateTracker';
import { jest } from '@jest/globals';

describe('RateTracker', () => {
  let tracker;

  beforeEach(() => {
    tracker = new RateTracker();
  });

  test('initializes provider with correct limits', () => {
    tracker.addProvider('p1', { limit: 100, windowMs: 60000 });
    const state = tracker.getProviderState('p1');
    expect(state.remaining).toBe(100);
    expect(state.limit).toBe(100);
  });

  test('decrements remaining on request', () => {
    tracker.addProvider('p1', { limit: 5, windowMs: 60000 });
    tracker.recordRequest('p1');
    expect(tracker.getProviderState('p1').remaining).toBe(4);
  });

  test('does not allow negative remaining', () => {
    tracker.addProvider('p1', { limit: 1, windowMs: 60000 });
    tracker.recordRequest('p1');
    expect(() => tracker.recordRequest('p1')).toThrow('Rate limit exceeded');
  });

  test('resets after windowMs', async () => {
    jest.useFakeTimers();
    tracker.addProvider('p1', { limit: 2, windowMs: 1000 });
    tracker.recordRequest('p1');
    tracker.recordRequest('p1');
    expect(tracker.getProviderState('p1').remaining).toBe(0);
    jest.advanceTimersByTime(1000);
    expect(tracker.getProviderState('p1').remaining).toBe(2);
    jest.useRealTimers();
  });

  test('handles concurrent requests safely', async () => {
    tracker.addProvider('p1', { limit: 10, windowMs: 60000 });
    const promises = Array.from({ length: 15 }, () => tracker.recordRequest('p1').catch(e => e));
    const results = await Promise.all(promises);
    const successes = results.filter(r => r === undefined).length;
    const failures = results.filter(r => r instanceof Error).length;
    expect(successes).toBe(10);
    expect(failures).toBe(5);
  });
});
