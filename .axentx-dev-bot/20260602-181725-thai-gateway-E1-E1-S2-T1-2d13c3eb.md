// aiHandler handles POST /v1/ai requests
func (h *Handler) aiHandler(w http.ResponseWriter, r *http.Request) {
	// Parse JSON body
	var req models.AiRequest
	err := json.NewDecoder(r.Body).Decode(&req)
	if err != nil {
		http.Error(w, err.Error(), http.StatusBadRequest)
		return
	}

	// Process request and get response
	resp, err := h.processAiRequest(req)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}

	// Return response with required fields
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(http.StatusOK)
	json.NewEncoder(w).Encode(models.AiResponse{
		ID:       resp.ID,
		Model:    resp.Model,
		Provider: resp.Provider,
		Usage:    resp.Usage,
		Response: resp.Response,
	})
}
