package translation

import (
	"context"
	"log"
	"time"

	"cloud.google.com/go/translate"
)

// translateWithFallback translates the input text using a pre-trained Thai model
// and logs errors with a fallback to English.
func translateWithFallback(ctx context.Context, text string) (string, error) {
	// Create a client for the Google Cloud Translation API
	client, err := translate.NewClient(ctx)
	if err != nil {
		log.Printf("Error creating translation client: %v", err)
		return text, err
	}
	defer client.Close()

	// Translate the text using the pre-trained Thai model
	translation, err := client.Translate(ctx, text, "th")
	if err != nil {
		log.Printf("Error translating text: %v", err)
		// Fallback to English
		return text, nil
	}

	return translation.Text, nil
}

// translate translates the input text and measures the translation latency.
func translate(ctx context.Context, text string) (string, time.Duration, error) {
	startTime := time.Now()
	translatedText, err := translateWithFallback(ctx, text)
	translationTime := time.Since(startTime)

	return translatedText, translationTime, err
}

REJECT: The implementation uses a non‑existent `client.Translate(ctx, text, "th")` API signature. The Google Cloud Translate Go client requires a different call (e.g., `client.Translate(ctx, []string{text}, language.Thai, option.WithSourceLanguage(language.English))`). As written, the code will not compile, making the change non‑functional.

- The `translateWithFallback` function calls a made‑up `Translate` method that does not exist in the `cloud.google.com/go/translate` package.
- The test suite invokes the real Google Translate service, which introduces external dependencies and will fail without proper authentication or network access.
- The fallback logic returns the original input text instead of translating to English, contradicting the requirement to fall back to English on error.
- Creating a new `translate.NewClient` on every request adds significant overhead and will likely prevent meeting the <200 ms latency target.
