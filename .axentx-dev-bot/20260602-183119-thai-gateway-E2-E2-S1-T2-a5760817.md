// src/services/providerHealth.ts
/**
 * In-memory store for provider health status.
 * Other parts of the system (e.g. health-check scheduler) should call
 * `setProviderStatus` to update the status.
 */
export type ProviderStatus = 'healthy' | 'unhealthy';

const providerStatusMap: Map<string, ProviderStatus> = new Map();

/**
 * Initialise a provider as healthy by default.
 */
export function initProvider(name: string): void {
  if (!providerStatusMap.has(name)) {
    providerStatusMap.set(name, 'healthy');
  }
}

/**
 * Update the health status for a provider.
 * @param name The provider's name.
 * @param status The new health status.
 */
export function setProviderStatus(name: string, status: ProviderStatus): void {
  providerStatusMap.set(name, status);
}

/**
 * Get the current health status for all providers.
 * @returns A map of provider names to their health status.
 */
export function getAll(): Map<string, ProviderStatus> {
  return providerStatusMap;
}
```

// src/services/providerHealth.ts
/**
 * Simple in‑memory store for provider health status.
 * The periodic health‑check worker should call `setProviderStatus`
 * whenever it obtains a fresh result.
 */

export type ProviderStatus = 'healthy' | 'unhealthy';

/* Map<providerName, status> */
const providerStatusMap: Map<string, ProviderStatus> = new Map();

/**
 * Initialise a provider entry (healthy by default).
 * Idempotent – calling multiple times has no side‑effects.
 */
export function initProvider(name: string): void {
  if (!providerStatusMap.has(name)) {
    providerStatusMap.set(name, 'healthy');
  }
}

/**
 * Update the health status for a given provider.
 */
export function setProviderStatus(name: string, status: ProviderStatus): void {
  providerStatusMap.set(name, status);
}

/**
 * Retrieve a shallow copy of the current status map.
 * Returned as a plain object for easy JSON serialization.
 */
export function getAll(): Record<string, ProviderStatus> {
  const obj: Record<string, ProviderStatus> = {};
  providerStatusMap.forEach((status, name) => {
    obj[name] = status;
  });
  return obj;
}

/**
 * Reset the store – useful for tests.
 */
export function reset(): void {
  providerStatusMap.clear();
}
```

## 1. Acceptance Criteria
1. **Periodic health checks** – The gateway must execute a health‑check request to each configured provider every 30 seconds.  
2. **Status update** – After each check the provider’s status (`healthy` / `unhealthy`) is stored in the in‑memory registry and is reflected in the `/health` response within 1 second.  
3. **Routing exclusion** – The load‑balancer/router must never select a provider marked `unhealthy` for any request.  
4. **/health payload** – `GET /health` returns JSON with:
   ```json
   {
     "timestamp": "<ISO‑8601>",
     "providers": {
       "<providerId>": { "status": "healthy|unhealthy", "lastChecked": "<ISO‑8601>" },
       ...
     }
   }
   ```
   HTTP status 200.  
5. **Security header** – Response includes `Content‑Security‑Policy: default-src 'none'` and `X-Content-Type-Options: nosniff`.  
6. **Graceful degradation** – If *all* providers become unhealthy, `/health` still returns status `degraded` with HTTP 503 and a body indicating no healthy back‑ends.  
7. **Observability** – Each health‑check execution logs a structured entry (`providerId`, `status`, `duration`) at `info` level.

  });

    service.start();
    expect(mockClient.ping).toHaveBeenCalledTimes(2); // immediate first run
    jest.advanceTimersByTime(30_000);
    expect(mockClient.ping).toHaveBeenCalledTimes(4);
  });

  test('marks provider healthy when ping resolves', async () => {
    mockClient.ping.mockResolvedValueOnce('OK');
    await service.checkProvider('p1');
    expect(service.getStatus('p1')).toBe('healthy');
  });

  test('marks provider unhealthy when ping rejects or times out', async () => {
    mockClient.ping.mockRejectedValueOnce(new Error('timeout'));
    await service.checkProvider('p2');
    expect(service.getStatus('p2')).toBe('unhealthy');
  });

  test('exposes current health snapshot', () => {
    service.setStatus('p1', 'healthy');
    service.setStatus('p2', 'unhealthy');
    const snapshot = service.snapshot();
    expect(snapshot.providers.p1.status).toBe('healthy');
    expect(snapshot.providers.p2.status).toBe('unhealthy');
    expect(new Date(snapshot.timestamp)).toBeInstanceOf(Date);
  });

  test('does not leak internal timers when stopped', () => {
    service.start();
    service.stop();

  });
