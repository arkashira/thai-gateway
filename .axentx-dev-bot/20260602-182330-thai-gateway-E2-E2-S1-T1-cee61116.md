# Task: Integrate Thai Translation API

## Diagnosis
The current codebase lacks the functionality to translate prompts into Thai before sending them to AI models. The `translation.js` file needs to be updated to include this functionality. Additionally, the system should handle translation API unavailability gracefully.

## Proposed Change
1. Install `libretranslate` package for Thai translation.
2. Update `translation.js` to use `libretranslate` for Thai translation.
3. Add error handling for translation API unavailability.

## Implementation

### 1. Install `libretranslate` package
```bash
npm install libretranslate
```

### 2. Update `translation.js`
```javascript
// Before
async function translatePrompt(prompt) {
  // No translation logic here
  return prompt;
}

// After
const libretranslate = require('libretranslate');

async function translatePrompt(prompt) {
  try {
    const translation = await libretranslate.translate(prompt, 'en', 'th');
    return translation.translatedText;
  } catch (error) {
    console.error('Translation API error:', error);
    // Return original prompt if translation fails
    return prompt;
  }
}
```

### 3. Add error handling for translation API unavailability
The updated `translatePrompt` function now includes a try-catch block to handle errors from the translation API. If the API is unavailable, the function will log the error and return the original prompt.

## Verification

**1. Acceptance Criteria**
- **AC1:** When a request containing a `prompt` field is received, the service calls the Thai translation API and replaces the original prompt with the translated Thai text before forwarding it to the downstream AI model.
- **AC2:** The translation API must be invoked with the original prompt text and the target language code `th`; the response must be parsed and used as the new prompt.
- **AC3:** If the translation API returns a successful response (HTTP 200 with a non‑empty `translatedText` field), the downstream request contains the translated prompt and the original prompt is not sent.
- **AC4:** If the translation API returns an error (non‑2xx status, timeout, or malformed response), the service returns a 502 Bad Gateway error to the caller with a clear error message, and no request is sent to the AI model.
- **AC5:** The service logs translation attempts and failures at appropriate log levels (info for success, warning/error for failures).
- **AC6:** The translation step adds less than 150 ms of latency on average under normal load (measured in integration tests).

// translation.test.js
const { translateAndForward } = require('./translation');
const axios = require('axios');
jest.mock('axios');
const logger = require('../utils/logger');
jest.mock('../utils/logger');

describe('translateAndForward', () => {
  const mockNext = jest.fn(); // simulates downstream AI call
  const baseReq = { body: { prompt: 'Hello world' }, headers: {} };

  afterEach(() => {
    jest.clearAllMocks();
  });

  test('calls translation API with correct payload and forwards translated prompt', async () => {
    axios.post.mockResolvedValue({
      status: 200,
      data: { translatedText: 'สวัสดีโลก' },
    });

    await translateAndForward(baseReq, mockNext);

    expect(axios.post).toHaveBeenCalledWith(
      expect.stringContaining('/translate'),
      { text: 'Hello world', targetLang: 'th' },
      expect.any(Object)
    );
    expect(mockNext).toHaveBeenCalledWith(
      expect.objectContaining({ body: { prompt: 'สวัสดีโลก' } })
    );
    expect(logger.info).toHaveBeenCalledWith(
      expect.stringContaining('Translation successful')
    );
  });

  test('returns 502 when translation API responds with non‑2xx', async () => {
    axios.post.mockResolvedValue({ status: 500, data: {} });

    const res = await translateAndForward(baseReq, mockNext);

    expect(res.status).toBe(502);
    expect(res.body.message).toMatch(/translation service unavailable/i);
    expect(mockNext).not.toHaveBeenCalled();
    expect(logger.warn).toHaveBeenCalled();
  });

  test('returns 502 on network timeout', async () => {
    axios.post.mockRejectedValue(new Error('ETIMEDOUT'));

    const res = await translateAndForward(baseReq, mockNext);

    expect(res.status).toBe(502);
    expect(res.body.message).toMatch(/translation service unavailable/i);
    expect(mockNext).not.toHaveBeenCalled();
    expect(logger.error).toHaveBeenCalled();
  });

  test('handles malformed translation response gracefully', async () => {
    axios.post.mockResolvedValue({ status: 200, data: {} }); // missing translatedText

    const res = await translateAndForward(baseReq, mockNext);

    expect(res.status).toBe(502);
    expect(res.body.message).toMatch(/invalid translation response/i);
    expect(mockNext).not.toHaveBeenCalled();
  });

  test('does not modify request when prompt field is missing', async () => {
    const req = { body: {}, headers: {} };
    await translateAndForward(req, mockNext);
    expect(mockNext).toHaveBeenCalledWith(req);
    expect(axios.post).not.toHaveBeenCalled();
  });
});
