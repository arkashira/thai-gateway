const express = require('express');
const axios = require('axios');
const app = express();
const PORT = 3000;

// Mock data for AI providers and their rate limits
const providers = [
    { name: 'OpenAI', url: 'https://api.openai.com/v1', rateLimit: 100, lastChecked: Date.now() },
    { name: 'Anthropic', url: 'https://api.anthropic.com/v1', rateLimit: 80, lastChecked: Date.now() },
    { name: 'LocalThaiAI', url: 'https://api.localthai.ai/v1', rateLimit: 120, lastChecked: Date.now() }
];

// Function to check and update rate limits
function updateRateLimits() {
    providers.forEach(provider => {
        // Simulate rate limit check (in real scenario, this would be an actual API call)
        provider.rateLimit = Math.floor(Math.random() * 150);
        provider.lastChecked = Date.now();
    });
}

// Update rate limits every 10 seconds
setInterval(updateRateLimits, 10000);

// Route handler for AI requests
app.get('/ai', async (req, res) => {
    let selectedProvider = null;
    let fastestResponseTime = Infinity;

    for (const provider of providers) {
        try {
            const startTime = Date.now();
            const response = await axios.get(`${provider.url}/status`, { timeout: 500 });
            const endTime = Date.now();
            const responseTime = endTime - startTime;

            if (responseTime < fastestResponseTime && provider.rateLimit > 0) {
                fastestResponseTime = responseTime;
                selectedProvider = provider;
            }
        } catch (error) {
            console.error(`Error checking ${provider.name}:`, error.message);
        }
    }

    if (!selectedProvider) {
        for (const provider of providers) {
            try {
                const response = await axios.get(`${provider.url}/status`, { timeout: 1000 });
                selectedProvider = provider;
                break;
            } catch (error) {
                console.error(`Fallback error checking ${provider.name}:`, error.message);
            }
        }
    }

    if (selectedProvider) {
        selectedProvider.rateLimit--;
        res.send(`Request routed to ${selectedProvider.name}`);
    } else {
        res.status(503).send('All providers are currently unavailable.');
    }
});

app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
```
