```python
import time
import logging
import threading
from collections import deque
from dataclasses import dataclass, field
from typing import Optional

logger = logging.getLogger(__name__)


@dataclass
class ProviderState:
    name: str
    rate_limit_tpm: int
    fallback_threshold: float = 0.90
    _window: deque = field(default_factory=lambda: deque())
    _lock: threading.Lock = field(default_factory=threading.Lock)

    def record_tokens(self, tokens: int) -> None:
        now = time.monotonic()
        with self._lock:
            self._window.append((now, tokens))
            cutoff = now - 60.0
            while self._window and self._window[0][0] < cutoff:
                self._window.popleft()

    def current_usage(self) -> int:
        now = time.monotonic()
        cutoff = now - 60.0
        with self._lock:
            return sum(t for ts, t in self._window if ts >= cutoff)

    def utilization(self) -> float:
        return self.current_usage() / self.rate_limit_tpm

    def is_overloaded(self) -> bool:
        return self.utilization() >= self.fallback_threshold


class RateLimiter:
    def __init__(self, providers_cfg: list[dict]):
        self._providers: dict[str, ProviderState] = {
            p["name"]: ProviderState(
                name=p["name"],
                rate_limit_tpm=p["rate_limit_tpm"],
                fallback_threshold=p.get("fallback_threshold", 0.90),
            )
            for p in providers_cfg
        }

    def record(self, provider: str, tokens: int) -> None:
        self._providers[provider].record_tokens(tokens)

    def select(self) -> Optional[str]:
        """Return least-congested non-overloaded provider, or None if all overloaded."""
        candidates = [
            (name, state)
            for name, state in self._providers.items()
            if not state.is_overloaded()
        ]
        if not candidates:
            logger.warning("routing_decision provider=NONE reason=all_overloaded")
            return None
        chosen = min(candidates, key=lambda x: x[1].utilization())[0]
        util = self._providers[chosen].utilization()
        logger.info(
            "routing_decision provider=%s utilization=%.3f", chosen, util
        )
        return chosen

    def utilization(self, provider: str) -> float:
        return self._providers[provider].utilization()

    def provider_names(self) -> list[str]:
        return list(self._providers.keys())
```

## 1. Acceptance Criteria

- `RateLimiter.select()` returns the provider with the lowest token-per-minute utilization among all non-overloaded providers.
- `select()` completes in ≤ 100 ms under a 100-concurrent-request load (measured end-to-end in integration harness).
- Any provider at ≥ 90% of its `rate_limit_tpm` is excluded from selection; the next-least-congested provider is chosen instead.
- Every call to `select()` emits a structured log line containing `routing_decision`, `provider`, and `utilization`.
- When all providers are overloaded, `select()` returns `None` and logs `reason=all_overloaded`.
- `record()` counts only tokens within the trailing 60-second sliding window (older entries are evicted).
- Thread-safety: concurrent `record()` and `select()` calls on the same `ProviderState` do not corrupt usage counts.
