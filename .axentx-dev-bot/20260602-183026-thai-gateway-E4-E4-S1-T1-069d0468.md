# test_dockerfile.py
import subprocess
import os

def build_image(tmp_path):
    """Builds the Docker image and returns its ID."""
    result = subprocess.run(
        ["docker", "build", "-t", "axentx/thai-gateway:test", "."],
        cwd=str(tmp_path),
        capture_output=True,
        text=True,
        check=True
    )
    return result.stdout.splitlines()[-1].split()[-1]  # image ID

def test_image_size(tmp_path):
    image_id = build_image(tmp_path)
    size_bytes = int(subprocess.check_output(
        ["docker", "image", "inspect", image_id, "--format", "{{.Size}}"]
    ))
    assert size_bytes <= 30 * 1024 * 1024, f"Image too large: {size_bytes} bytes"

def test_exposed_port(tmp_path):
    image_id = build_image(tmp_path)
    ports = subprocess.check_output(
        ["docker", "image", "inspect", image_id, "--format", "{{json .Config.ExposedPorts}}"]
    )
    assert "8080/tcp" in ports.decode()

def test_env_vars(tmp_path):
    image_id = build_image(tmp_path)
    env = subprocess.check_output(
        ["docker", "image", "inspect", image_id, "--format", "{{json .Config.Env}}"]
    )
    env_list = env.decode()
    for var in ["PROVIDER_ID", "PROVIDER_SECRET", "THAI_GATEWAY_ENV"]:
        assert var in env_list

def test_entrypoint(tmp_path):
    image_id = build_image(tmp_path)
    entrypoint = subprocess.check_output(
        ["docker", "image", "inspect", image_id, "--format", "{{json .Config.Entrypoint}}"]
    )
    assert "/usr/local/bin/thai-gateway" in entrypoint.decode()

def test_healthcheck(tmp_path):
    image_id = build_image(tmp_path)
    healthcheck = subprocess.check_output(

    assert "HEALTHCHECK" in healthcheck.decode()
