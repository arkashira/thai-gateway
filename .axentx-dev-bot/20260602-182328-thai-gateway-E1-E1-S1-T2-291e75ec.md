```js
/**
 * endpoint-registry.js
 * In-memory registry of AI provider endpoints, their quota state, and health.
 * Designed to be refreshed periodically (every 5 min) by the routing layer.
 */

class EndpointRegistry {
  constructor() {
    /** @type {Map<string, { quotaLimit: number, quotaUsed: number, available: boolean, lastRefresh: number }>} */
    this.endpoints = new Map();
  }

  /**
   * Register or update an endpoint.
   * @param {string} id            — stable endpoint identifier (e.g. "openai-primary")
   * @param {object} opts
   * @param {number} opts.quotaLimit
   * @param {number} opts.quotaUsed
   * @param {boolean} opts.available
   */
  upsert(id, { quotaLimit = Infinity, quotaUsed = 0, available = true } = {}) {
    this.endpoints.set(id, {
      quotaLimit,
      quotaUsed,
      available,
      lastRefresh: Date.now(),
    });
  }

  /** Mark an endpoint unavailable (e.g. after a 5xx or timeout). */
  markUnavailable(id) {
    const ep = this.endpoints.get(id);
    if (ep) {
      ep.available = false;
      ep.lastRefresh = Date.now();
    }
  }

  /** Mark an endpoint available again. */
  markAvailable(id) {
    const ep = this.endpoints.get(id);
    if (ep) {
      ep.available = true;
      ep.lastRefresh = Date.now();
    }
  }

  /** Record quota consumption after a successful request. */
  recordUsage(id, cost = 1) {
    const ep = this.endpoints.get(id);
    if (ep) ep.quotaUsed += cost;
  }

  /** Return remaining quota for an endpoint (0 when unavailable). */
  remainingQuota(id) {
    const ep = this.endpoints.get(id);
    if (!ep || !ep.available) return 0;
    return Math.max(0, ep.quotaLimit - ep.quotaUsed);
  }

  /** Snapshot of all endpoints for inspection / testing. */
  list() {
    return Array.from(this.endpoints.entries()).map(([id, state]) => ({ id, ...state }));
  }

  /** Remove an endpoint entirely. */
  remove(id) {
    this.endpoints.delete(id);
  }
}

module.exports = { EndpointRegistry };
```

## 1. Acceptance Criteria
1. **Quota‑Based Selection** – For every request the router returns the *available* endpoint whose `remaining_quota` is the smallest positive integer.  
2. **Refresh Interval** – The routing table is automatically refreshed **every 5 minutes (300 000 ms)** regardless of request volume.  
3. **Graceful Degradation** – If no endpoint is both `available` and has `remaining_quota > 0`, the router returns `null` (or throws a controlled `NoAvailableEndpointError`).  
4. **Availability Flag** – Endpoints flagged `unavailable: true` are never considered, even if they have a positive quota.  
5. **Manual Refresh** – A public `refresh()` method forces an immediate re‑evaluation of the endpoint list and updates the internal cache.  
6. **Metrics Exposure** – The router exposes `getCurrentSelection()` (endpoint ID) and `getLastRefreshTimestamp()` for monitoring.  
7. **Thread‑Safety / Concurrency** – Concurrent calls to `chooseEndpoint()` must never observe a partially‑updated routing table; the selection always reflects the most recent full refresh.
