// tests/dockerfile.test.js
const { execSync } = require('child_process');
const fs = require('fs');

describe('Dockerfile', () => {
  const imageTag = 'axentx/thai-gateway:test';

  beforeAll(() => {
    execSync(`docker build -t ${imageTag} .`, { stdio: 'inherit' });
  });

  afterAll(() => {
    execSync(`docker rmi ${imageTag}`);
  });

  test('image size <= 30 MB', () => {
    const sizeBytes = execSync(`docker image inspect ${imageTag} --format='{{.Size}}'`)
      .toString()
      .trim();
    const sizeMB = Number(sizeBytes) / (1024 * 1024);
    expect(sizeMB).toBeLessThanOrEqual(30);
  });

  test('exposes port 8080', () => {
    const inspect = JSON.parse(
      execSync(`docker image inspect ${imageTag}`).toString()
    );
    const exposed = inspect[0].Config.ExposedPorts;
    expect(exposed).toHaveProperty('8080/tcp');
  });

  test('accepts provider credentials via env vars', () => {
    const container = execSync(
      `docker run -d -e PROVIDER_KEY=foo -e PROVIDER_SECRET=bar ${imageTag}`
    ).toString().trim();
    const env = execSync(`docker exec ${container} env`).toString();
    expect(env).toMatch(/PROVIDER_KEY=foo/);
    expect(env).toMatch(/PROVIDER_SECRET=bar/);
    execSync(`docker rm -f ${container}`);
  });

  test('health check returns 200', () => {
    const container = execSync(
      `docker run -d -p 0:8080 ${imageTag}`
    ).toString().trim();
    const res = execSync(`curl -s -o /dev/null -w "%{http_code}" http://localhost:8080/health`);
    expect(res.toString()).toBe('200');
    execSync(`docker rm -f ${container}`);
  });
});
