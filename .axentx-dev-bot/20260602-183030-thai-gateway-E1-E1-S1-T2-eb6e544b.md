```javascript
// healthCheck.js
const axios = require('axios');

const providers = [
    { name: 'OpenAI', url: 'https://api.openai.com/v1/health' },
    { name: 'Anthropic', url: 'https://api.anthropic.com/v1/health' },
    { name: 'LocalThaiAI', url: 'http://localhost:8000/health' }
];

async function checkProviderHealth(provider) {
    try {
        const response = await axios.get(provider.url, { timeout: 1000 });
        return { ...provider, status: 'healthy', responseTime: response.headers['response-time'] };
    } catch (error) {
        return { ...provider, status: 'unhealthy', responseTime: error.response ? error.response.headers['response-time'] : null };
    }
}

async function getProvidersHealth() {
    const healthChecks = await Promise.all(providers.map(checkProviderHealth));
    return healthChecks;
}

module.exports = {
    getProvidersHealth
};
```

```javascript
});
```

## 1. Acceptance Criteria  

1. **Health‑check endpoint exists** – A GET endpoint `/providers/health` returns a JSON payload.  
2. **Provider status** – For each registered provider (OpenAI, Anthropic, Local Thai AI) the payload includes:  
   - `name` (string)  
   - `status` (`"healthy"` or `"unhealthy"`)  
   - `lastChecked` (ISO‑8601 timestamp)  
   - `responseTimeMs` (number, null if unavailable)  
3. **Health check logic** – The endpoint triggers an asynchronous health check that:  
   - Sends a lightweight request (e.g., `/v1/health` or `/ping`) to each provider.  
   - Considers a provider healthy if the response is received within 500 ms and status code 200.  
   - Marks provider unhealthy if the request times out or returns non‑200.  
4. **Rate limiting** – Health checks are throttled to no more than one request per provider per 10 seconds.  
5. **Security** – The endpoint is protected by an API key header (`X-API-KEY`) and returns 401 if missing/invalid.  
6. **Error handling** – Any unexpected exception results in a 500 response with a generic error message.
