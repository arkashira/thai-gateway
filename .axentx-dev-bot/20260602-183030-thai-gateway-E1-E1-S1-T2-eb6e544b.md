// healthCheck.js
const axios = require('axios');

const providers = [
    { name: 'OpenAI', url: 'https://api.openai.com/v1/health' },
    { name: 'Anthropic', url: 'https://api.anthropic.com/v1/health' },
    { name: 'LocalThaiAI', url: 'http://localhost:8000/health' }
];

async function checkProviderHealth(provider) {
    try {
        const response = await axios.get(provider.url, { timeout: 1000 });
        return { ...provider, status: 'healthy', responseTime: response.headers['response-time'] };
    } catch (error) {
        return { ...provider, status: 'unhealthy', responseTime: error.response ? error.response.headers['response-time'] : null };
    }
}

async function getProvidersHealth() {
    const healthChecks = await Promise.all(providers.map(checkProviderHealth));
    return healthChecks;
}

module.exports = {
    getProvidersHealth
};
```

});
