use serde::{Deserialize, Serialize};
use std::fs;
use std::path::Path;

#[derive(Serialize, Deserialize, Debug)]
pub struct ProviderConfig {
    pub urls: Vec<String>,
    pub api_keys: Vec<String>,
}

impl ProviderConfig {
    pub fn new(urls: Vec<String>, api_keys: Vec<String>) -> Self {
        ProviderConfig { urls, api_keys }
    }
}

pub struct Persistence {
    config_path: String,
}

impl Persistence {
    pub fn new(config_path: &str) -> Self {
        Persistence {
            config_path: config_path.to_string(),
        }
    }

    pub fn save_config(&self, config: &ProviderConfig) -> Result<(), Box<dyn std::error::Error>> {
        let config_json = serde_json::to_string(config)?;
        fs::write(&self.config_path, config_json)?;
        Ok(())
    }

    pub fn load_config(&self) -> Result<ProviderConfig, Box<dyn std::error::Error>> {
        if Path::new(&self.config_path).exists() {
            let config_json = fs::read_to_string(&self.config_path)?;
            let config: ProviderConfig = serde_json::from_str(&config_json)?;
            Ok(config)
        } else {
            Ok(ProviderConfig::new(Vec::new(), Vec::new()))
        }
    }
}
```
