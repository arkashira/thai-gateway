use serde::{Serialize, Deserialize};
use once_cell::sync::Lazy;

use std::path::PathBuf;

/// Represents a single AI provider configuration.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub struct ProviderConfig {
    pub url: String,
    pub api_key: String,
}

/// Global in-memory cache for fast reads.
static CONFIG_CACHE: Lazy<Mutex<Option<Vec<ProviderConfig>>>> = Lazy::new(|| Mutex::new(None));

/// Load configuration from disk.
fn load_config() -> io::Result<Vec<ProviderConfig>> {
    let path = PathBuf::from("config.json");
    if !path.exists() {
        return Ok(vec![]);
    }
    let file = File::open(path)?;
    let reader = io::BufReader::new(file);
    serde_json::from_reader(reader).map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))
}

/// Save configuration to disk.
fn save_config(config: &Vec<ProviderConfig>) -> io::Result<()> {
    let path = PathBuf::from("config.json");
    let file = File::create(path)?;
    let writer = io::BufWriter::new(file);
    serde_json::to_writer_pretty(writer, config).map_err(|e| io::Error::new(io::ErrorKind::Other, e))
}

/// Apply and persist configuration changes.
async fn apply_and_persist(providers: Vec<ProviderConfig>) -> io::Result<()> {
    // Verify provider health (assuming a health check function exists).
    for provider in &providers {
        if !verify_provider_health(provider).await {
            eprintln!("Provider {} is not healthy", provider.url);
            return Err(io::Error::new(io::ErrorKind::InvalidInput, "Unhealthy provider"));
        }
    }

    // Update in-memory cache.
    *CONFIG_CACHE.lock().unwrap() = Some(providers.clone());

    // Save to disk.
    save_config(&providers)
}

## 1. Acceptance Criteria
1. **POST /config stores configuration** – A `POST /config` request with a valid JSON body containing an array of provider objects (`url` and `api_key`) returns HTTP 200 and writes the exact JSON payload to the configured persistence file (`config.json` by default).  
2. **File permissions** – The persisted file is created with mode `0o600` (read/write for owner only) on Unix‑like systems; on Windows the file ACL grants read/write to the service user only.  
3. **Load on startup** – On service start, the gateway reads `config.json` (if present) and populates the in‑memory provider list exactly as stored, making the providers immediately available to health‑check logic.  
4. **Health verification timeout** – After loading or receiving a new configuration, each provider’s health endpoint is probed; all probes must complete (success or failure) within **5 seconds** total. The health result (up/down) is stored in memory.  
5. **Overwrite semantics** – Submitting a new configuration overwrites the existing `config.json` atomically (write‑temp‑then‑rename) without leaving a partially‑written file.  
6. **Persistence across restarts** – After a graceful shutdown and subsequent restart, the gateway’s in‑memory provider list matches the last successfully persisted configuration.  
7. **Error handling** – If the JSON payload is malformed, missing required fields, or the file cannot be written (e.g., permission error), the endpoint returns HTTP 400 (validation) or HTTP 500 (IO) and **does not** modify the existing persisted file.
