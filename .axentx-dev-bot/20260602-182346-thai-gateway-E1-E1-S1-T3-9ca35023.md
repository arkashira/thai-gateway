use std::fs::{self, File};
use std::io::{self, Write, Read};
use serde::{Serialize, Deserialize};
use once_cell::sync::Lazy;
use std::sync::Mutex;
use std::path::PathBuf;

/// Represents a single AI provider configuration.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub struct ProviderConfig {
    pub url: String,
    pub api_key: String,
}

/// Global in-memory cache for fast reads.
static CONFIG_CACHE: Lazy<Mutex<Option<Vec<ProviderConfig>>>> = Lazy::new(|| Mutex::new(None));

/// Load configuration from disk.
fn load_config() -> io::Result<Vec<ProviderConfig>> {
    let path = PathBuf::from("config.json");
    if !path.exists() {
        return Ok(vec![]);
    }
    let file = File::open(path)?;
    let reader = io::BufReader::new(file);
    serde_json::from_reader(reader).map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))
}

/// Save configuration to disk.
fn save_config(config: &Vec<ProviderConfig>) -> io::Result<()> {
    let path = PathBuf::from("config.json");
    let file = File::create(path)?;
    let writer = io::BufWriter::new(file);
    serde_json::to_writer_pretty(writer, config).map_err(|e| io::Error::new(io::ErrorKind::Other, e))
}

/// Apply and persist configuration changes.
async fn apply_and_persist(providers: Vec<ProviderConfig>) -> io::Result<()> {
    // Verify provider health (assuming a health check function exists).
    for provider in &providers {
        if !verify_provider_health(provider).await {
            eprintln!("Provider {} is not healthy", provider.url);
            return Err(io::Error::new(io::ErrorKind::InvalidInput, "Unhealthy provider"));
        }
    }

    // Update in-memory cache.
    *CONFIG_CACHE.lock().unwrap() = Some(providers.clone());

    // Save to disk.
    save_config(&providers)
}
```
