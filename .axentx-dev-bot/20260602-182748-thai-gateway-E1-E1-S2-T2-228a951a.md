# Thai-Gateway — Core Gateway Skeleton

## Overview

The goal of the Thai-Gateway project is to create a minimal, functional gateway that can handle OpenAI-compatible chat-completion requests, manage multiple AI providers using a weighted round-robin strategy, and incorporate a Thai translation feature. This document synthesizes the best insights from multiple proposals to create a cohesive implementation plan.

## Key Features

1. **OpenAI-Compatible Requests**: The gateway will accept requests formatted for OpenAI's API.
2. **Weighted Round-Robin Provider Selection**: The system will select an upstream AI provider based on a weighted round-robin algorithm, with in-memory rate-limit tracking using a token-bucket approach.
3. **Thai Translation Hooks**: The gateway will include pluggable pre/post translation hooks for Thai, with a stub implementation for immediate deployment and a plan for a full implementation in future iterations.
4. **Health and Metrics Endpoints**: The gateway will expose endpoints for health checks and statistics.

import "dotenv/config";

export type ProviderConfig = {
  name: string;
  upstream: string;
  weight: number;
  rps: number;
  burst: number;
};

function parseProviders(raw: string | undefined): ProviderConfig[] {
  if (!raw) return [];
  return raw.split(",").map((entry) => {
    const [name, upstream, weight, rps, burst] = entry.split("|");
    return {
      name,
      upstream,
      weight: Number(weight || 1),
      rps: Number(rps || 5),
      burst: Number(burst || rps || 5),
    };
  });
}

export const config = {
  port: Number(process.env.PORT || 8080),
  logLevel: process.env.LOG_LEVEL || "info",
  providers: parseProviders(process.env.PROVIDERS),
  thaiTranslateEnabled: process.env.THAI_TRANSLATE_ENABLED === "true",
};
