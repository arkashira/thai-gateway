+package api
+
+import (
+	"encoding/json"
+	"net/http"
+)
+
+// HealthResponse is the JSON payload returned by the /health endpoint.
+type HealthResponse struct {
+	Status string `json:"status"`
+}
+
+// HealthHandler writes a 200 response with a JSON body `{"status":"ok"}`.
+// It is deliberately tiny – the gateway’s liveness can be inferred from the
+// ability to serve this endpoint.
+func HealthHandler(w http.ResponseWriter, r *http.Request) {
+	// Enforce GET only – other methods get 405.
+	if r.Method != http.MethodGet {
+		w.Header().Set("Allow", http.MethodGet)
+		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
+		return
+	}
+
+	w.Header().Set("Content-Type", "application/json")
+	// Encode never returns an error for a static struct, ignore it.
+	_ = json.NewEncoder(w).Encode(HealthResponse{Status: "ok"})
+}
```
