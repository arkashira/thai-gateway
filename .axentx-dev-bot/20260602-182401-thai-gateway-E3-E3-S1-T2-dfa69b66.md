// /opt/axentx/thai-gateway/src/middleware/rate-limiter.js
const RateLimitPool = require('../models/rate-limit-pool');

class RateLimiterMiddleware {
  constructor(poolConfig) {
    this.pool = new RateLimitPool(poolConfig);
    this.headers = {
      'X-RateLimit-Limit': 0,
      'X-RateLimit-Remaining': 0,
      'X-RateLimit-Reset': 0,
      'X-RateLimit-Pool': 'active',
      'Retry-After': 0
    };
  }

  async processRequest(req, res, next) {
    const provider = req.headers['x-provider'] || 'default';
    const upstreamHeaders = req.headers;
    
    // Check if any provider has capacity
    const hasCapacity = await this.pool.hasCapacity(provider);
    
    if (!hasCapacity) {
      const resetTime = this.pool.getResetTime(provider);
      res.set('Retry-After', Math.ceil(resetTime / 1000).toString());
      res.status(429).json({
        error: 'Rate limit exceeded',
        message: 'All providers are at capacity',
        retry_after: Math.ceil(resetTime / 1000)
      });
      return;
    }
    
    // Update rate limit headers from upstream response
    this.updateHeaders(upstreamHeaders);
    
    // Forward rate limit headers to client
    res.set('X-RateLimit-Limit', this.headers['X-RateLimit-Limit']);
    res.set('X-RateLimit-Remaining', this.headers['X-RateLimit-Remaining']);
    res.set('X-RateLimit-Reset', this.headers['X-RateLimit-Reset']);
    res.set('X-RateLimit-Pool', this.headers['X-RateLimit-Pool']);
    
    next();
  }

  updateHeaders(upstreamHeaders) {
    const limit = parseInt(upstreamHeaders['x-ratelimit-limit'] || '0', 10);
    const remaining = parseInt(upstreamHeaders['x-ratelimit-remaining'] || '0', 10);
    const reset = parseInt(upstreamHeaders['x-ratelimit-reset'] || '0', 10);
    
    this.headers['X-RateLimit-Limit'] = limit;
    this.headers['X-RateLimit-Remaining'] = remaining;
    this.headers['X-RateLimit-Reset'] = reset;
    this.headers['X-RateLimit-Pool'] = 'active';
  }

  getPoolStatus() {
    return this.pool.getStatus();
  }
}

module.exports = RateLimiterMiddleware;
```
