// /opt/axentx/thai-gateway/src/middleware/translatePrompt.js
const axios = require('axios');
const { detectLanguage, translateText } = require('google-translate-api');
const { ThaiTranslator } = require('./ThaiTranslator'); // Import the Thai translator class

module.exports = async (req, res, next) => {
  try {
    // Detect the language of the prompt
    const language = detectLanguage(req.body.prompt);
    if (language === 'en') {
      // Translate the prompt to Thai
      const thaiTranslator = new ThaiTranslator();
      const translatedPrompt = await thaiTranslator.translateText(req.body.prompt);
      // Send the translated prompt to the AI provider
      const response = await axios.post('/ai', { prompt: translatedPrompt });
      // Return the provider's response in Thai
      res.json(response.data);
    } else {
      // If the prompt is not in English, send it to the AI provider as is
      next();
    }
  } catch (error) {
    console.error(error);
    res.status(500).json({ message: 'Internal Server Error' });
  }
};
```
