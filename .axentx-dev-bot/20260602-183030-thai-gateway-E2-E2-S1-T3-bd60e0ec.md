// translator.test.js
const translator = require('../translator');
const assert = require('assert');

describe('Translator', () => {
  it('translates English to Thai with high accuracy', () => {
    const englishPrompt = 'Hello, how are you?';
    const expectedThaiTranslation = 'สวัสดี, คุณสบายดีไหม?';
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('preserves original intent and context', () => {
    const englishPrompt = 'I want to buy a book.';
    const expectedThaiTranslation = 'ฉันอยากซื้อหนังสือ';
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('reduces redundant API calls using cache', () => {
    const englishPrompt = 'What is the weather like today?';
    const expectedThaiTranslation = 'สภาพอากาศวันนี้เป็นอย่างไร?';
    translator.translate(englishPrompt);
    translator.translate(englishPrompt);
    assert.strictEqual(translator.cacheHits, 1);
    assert.strictEqual(translator.cacheMisses, 1);
  });

  it('supports text prompt format', () => {
    const englishPrompt = 'Hello, world!';
    const expectedThaiTranslation = 'สวัสดี, โลก!';
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('supports JSON prompt format', () => {
    const englishPrompt = { text: 'Hello, world!' };
    const expectedThaiTranslation = { text: 'สวัสดี, โลก!' };
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.deepStrictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('handles empty string input', () => {
    const englishPrompt = '';
    const expectedThaiTranslation = '';
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('handles null input', () => {
    const englishPrompt = null;
    const expectedThaiTranslation = null;
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });

  it('handles non-English characters', () => {
    const englishPrompt = 'Hello, ¡hola!';
    const expectedThaiTranslation = 'สวัสดี, ¡hola!';
    const actualThaiTranslation = translator.translate(englishPrompt);
    assert.strictEqual(actualThaiTranslation, expectedThaiTranslation);
  });
});
