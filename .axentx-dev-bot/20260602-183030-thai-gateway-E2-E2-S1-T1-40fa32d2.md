```javascript
const { Translate } = require('@google-cloud/translate').v2;
const NodeCache = require('node-cache');

// Initialize the cache
const translationCache = new NodeCache({ stdTTL: 3600, checkperiod: 1200 });

// Load environment variables
require('dotenv').config();

// Initialize the Google Cloud Translation client
const translate = new Translate({
  projectId: process.env.GOOGLE_CLOUD_PROJECT_ID,
  keyFilename: process.env.GOOGLE_APPLICATION_CREDENTIALS,
});

/**
 * Translates text from English to Thai
 * @param {string} text - The text to translate
 * @returns {Promise<string>} The translated text
 */
async function translateText(text) {
  // Check cache first
  const cachedTranslation = translationCache.get(text);
  if (cachedTranslation) {
    return cachedTranslation;
  }

  try {
    const [translation] = await translate.translate(text, 'th');
    // Cache the translation
    translationCache.set(text, translation);
    return translation;
  } catch (error) {
    console.error('Error translating text:', error);
    throw error;
  }
}

/**
 * Translates a JSON prompt from English to Thai
 * @param {Object} prompt - The JSON prompt to translate
 * @returns {Promise<Object>} The translated JSON prompt
 */
async function translateJsonPrompt(prompt) {
  const translatedPrompt = {};
  for (const [key, value] of Object.entries(prompt)) {
    if (typeof value === 'string') {
      translatedPrompt[key] = await translateText(value);
    } else if (typeof value === 'object') {
      translatedPrompt[key] = await translateJsonPrompt(value);
    } else {
      translatedPrompt[key] = value;
    }
  }
  return translatedPrompt;
}

module.exports = {
  translateText,
  translateJsonPrompt,
};
```
