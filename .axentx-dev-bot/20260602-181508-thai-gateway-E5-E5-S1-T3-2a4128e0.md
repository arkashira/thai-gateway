// cmd/server/health.go
//
// A minimal, self‑contained health‑check endpoint for the gateway.
// The handler verifies that the Prometheus metrics endpoint and the
// structured‑log endpoint are reachable.  If either check fails the
// handler returns 500; otherwise it returns 200.
//
// The code is deliberately simple so it can be unit‑tested easily
// (the HTTP client is injected via a function variable).  It also
// keeps the `main` package separate from the handler so the file
// can be imported by other parts of the application if needed.

package cmd

import (
	"log"
	"net/http"
	"time"
)

// httpGet is a variable that holds the function used to perform
// HTTP GET requests.  It defaults to http.Get but can be replaced
// in tests to avoid real network traffic.
var httpGet = http.Get

// healthHandler checks the health of the gateway.
func healthHandler(w http.ResponseWriter, r *http.Request) {
	// 1. Verify Prometheus metrics are exposed
	if !checkURL("http://localhost:9090/metrics") {
		http.Error(w, "Prometheus metrics not reachable", http.StatusInternalServerError)
		return
	}

	// 2. Verify structured logs are exposed
	if !checkURL("http://localhost:9091/logs") {
		http.Error(w, "Structured logs not reachable", http.StatusInternalServerError)
		return
	}

	// All checks passed
	w.WriteHeader(http.StatusOK)
}

// checkURL performs a GET request to the supplied URL and returns
// true if the request succeeds with a 2xx status code.
func checkURL(url string) bool {
	resp, err := httpGet(url)
	if err != nil {
		log.Printf("health check failed for %s: %v", url, err)
		return false
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		log.Printf("health check returned non‑2xx for %s: %d", url, resp.StatusCode)
		return false
	}
	return true
}
```
