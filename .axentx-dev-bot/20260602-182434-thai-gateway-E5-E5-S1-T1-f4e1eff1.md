package auth

import (
	"html"
	"io/ioutil"
	"log"
	"net/http"
	"strings"

	"github.com/gorilla/mux"
	"gopkg.in/yaml.v3"
)

// ---------------------------------------------------------------------
// Configuration
// ---------------------------------------------------------------------

// APIKeyConfig holds the list of valid keys that the gateway will accept.
type APIKeyConfig struct {
	APIKeys []string `yaml:"api_keys"`
}

// LoadAPIKeyConfig reads the YAML file and unmarshals it into APIKeyConfig.
func LoadAPIKeyConfig(path string) (APIKeyConfig, error) {
	var cfg APIKeyConfig
	data, err := ioutil.ReadFile(path)
	if err != nil {
		return cfg, err
	}
	if err = yaml.Unmarshal(data, &cfg); err != nil {
		return cfg, err
	}
	return cfg, nil
}

// ---------------------------------------------------------------------
// Middleware
// ---------------------------------------------------------------------

// APIKeyAuthMiddleware returns a mux.MiddlewareFunc that checks the
// "X-API-KEY" header against the supplied configuration.
func APIKeyAuthMiddleware(cfg APIKeyConfig) mux.MiddlewareFunc {
	// Build a map for O(1) look‑ups – useful if you have many keys.
	keySet := make(map[string]struct{}, len(cfg.APIKeys))
	for _, k := range cfg.APIKeys {
		keySet[k] = struct{}{}
	}

	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			apiKey := strings.TrimSpace(r.Header.Get("X-API-KEY"))
			if apiKey == "" {
				http.Error(w, "API key missing", http.StatusUnauthorized)
				return
			}
			if _, ok := keySet[apiKey]; !ok {
				http.Error(w, "Invalid API key", http.StatusUnauthorized)
				return
			}
			// Optionally, store the key in the request context for downstream use.
			ctx := r.Context()
			ctx = contextWithAPIKey(ctx, apiKey)
			next.ServeHTTP(w, r.WithContext(ctx))
		})
	}
}

// ---------------------------------------------------------------------
// Input Sanitisation
// ---------------------------------------------------------------------

// SanitizeInput escapes HTML entities and trims whitespace.
// Use this on any user‑supplied string that will be rendered or stored.
func SanitizeInput(raw string) string {
	return html.EscapeString(strings.TrimSpace(raw))
}

// ---------------------------------------------------------------------
// Context helpers (optional but handy)
// ---------------------------------------------------------------------

import "context"

type ctxKeyAPIKey struct{}

// contextWithAPIKey stores the key in the request context.
func contextWithAPIKey(ctx context.Context, key string) context.Context {
	return context.WithValue(ctx, ctxKeyAPIKey{}, key)
}

// APIKeyFromContext retrieves the key from the context; empty string if not set.
func APIKeyFromContext(ctx context.Context) string {
	if v := ctx.Value(ctxKeyAPIKey{}); v != nil {
		if s, ok := v.(string); ok {
			return s
		}
	}
	return ""
}

## 1. Acceptance Criteria  

| # | Criterion | Measurable Target |
|---|-----------|-------------------|
| 1 | **API key presence** | Every incoming request must contain an `X-API-Key` header; missing header → 401 Unauthorized. |
| 2 | **API key validity** | Header value must match a key stored in `auth.yaml`; invalid key → 403 Forbidden. |
| 3 | **Rate‑limit bypass** | Valid key allows request to proceed to next handler; no premature termination. |
| 4 | **Input sanitisation** | All request body fields are scanned for common injection patterns (`<script>`, `; DROP`, `--`, `/*`, `*/`, etc.) and rejected with 400 Bad Request. |
| 5 | **Logging mask** | Any sensitive values (API keys, passwords, tokens) that appear in logs are replaced with `***MASKED***`. |
| 6 | **Performance** | Middleware adds ≤ 5 ms latency under 100 req/s load. |
| 7 | **Configuration reload** | Updating `auth.yaml` while the server is running refreshes the key list without restart. |
