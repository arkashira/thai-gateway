func withEnv(vars map[string]string, fn func()) {
	orig := map[string]string{}
	for k, v := range vars {
		orig[k] = os.Getenv(k)
		os.Setenv(k, v)
	}
	defer func() {
		for k, v := range orig {
			os.Setenv(k, v)
		}
	}()
	fn()
}
