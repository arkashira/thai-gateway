```python
# thai_gateway/translation/fallback.py
import logging
from thai_gateway.translation.translation_service import TranslationService

class FallbackLogic:
    def __init__(self, translation_service):
        self.translation_service = translation_service
        self.logger = logging.getLogger(__name__)

    def translate(self, prompt):
        try:
            translated_prompt = self.translation_service.translate(prompt)
            return translated_prompt
        except Exception as e:
            self.logger.error(f"Translation failed for prompt: {prompt}, error: {str(e)}")
            return prompt
```
