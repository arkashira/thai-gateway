```python
from abc import ABC, abstractmethod
import yaml

class ProviderInterface(ABC):
    @abstractmethod
    def get_rate_limit(self):
        pass

    @abstractmethod
    def get_endpoint(self):
        pass

    @classmethod
    def from_config(cls, config):
        provider_config = config.get(cls.__name__)
        if provider_config:
            return cls(**provider_config)
        else:
            raise ValueError(f"Missing config for {cls.__name__}")
```

**/opt/axentx/thai-gateway/src/providers/config.py**
```python
import yaml

class ProviderConfig:
    def __init__(self, config_file):
        self.config_file = config_file
        self.config = self.load_config()

    def load_config(self):
        with open(self.config_file, 'r') as f:
            return yaml.safe_load(f)

    def get_provider(self, provider_name):
        return ProviderInterface.from_config(self.config.get(provider_name))
```
