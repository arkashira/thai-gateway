```go
package metrics

import (
    "net/http"
    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promhttp"
)

var (
    requestLatency = prometheus.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "request_latency_seconds",
            Help:    "Histogram of request latency in seconds.",
            Buckets: prometheus.DefBuckets,
        },
        []string{"route"},
    )
    errorCount = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "error_count",
            Help: "Total number of errors.",
        },
        []string{"route"},
    )
)

func init() {
    prometheus.MustRegister(requestLatency)
    prometheus.MustRegister(errorCount)
}

func RegisterMetricsHandler() {
    http.Handle("/metrics", promhttp.Handler())
}

func RecordLatency(route string, duration float64) {
    requestLatency.WithLabelValues(route).Observe(duration)
}

func IncrementErrorCount(route string) {
    errorCount.WithLabelValues(route).Inc()
```
```go
}
```

## 1. Acceptance Criteria
1. **Prometheus metrics endpoint**  
   - `GET /metrics` returns `200 OK` with `Content‑Type: text/plain; version=0.0.4` and a non‑empty body containing metric families `gateway_requests_total`, `gateway_request_duration_seconds`, `gateway_errors_total`, and provider‑specific counters (e.g., `gateway_provider_success_total`).  
2. **Metric correctness**  
   - After a successful request to any route, `gateway_requests_total{status="200"}` increments by 1 and `gateway_request_duration_seconds` records a latency > 0.  
   - After a request that triggers a rate‑limit or translation error, the corresponding `gateway_errors_total{type="rate_limit"}` or `gateway_errors_total{type="translation"}` increments.  
3. **Structured JSON logs**  
   - Every incoming HTTP request produces a log line in valid JSON containing at least: `timestamp`, `level`, `request_id`, `method`, `path`, `status`, `duration_ms`, and `error` (null if none).  
   - Errors (e.g., 5xx responses) include an `error.message` field with the error description.  
4. **Health endpoint**  
   - `GET /health` returns `200 OK` with JSON body `{ "status": "ok" }` when all internal components (router, rate‑limiter, translation service client) are operational.  
   - If any component is unhealthy, `/health` returns `503 Service Unavailable` with `{ "status": "degraded", "details": [...] }`.  
5. **Observability performance impact**  
   - Adding metrics and logging must not increase request latency by more than **10 %** compared to a baseline run (measured with a synthetic load test).
