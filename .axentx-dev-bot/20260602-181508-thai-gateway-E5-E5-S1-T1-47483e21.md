```go
// /opt/axentx/thai-gateway/internal/metrics/prometheus.go

package metrics

import (
    "net/http"
    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promhttp"
)

var (
    requestLatency = prometheus.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "request_latency_seconds",
            Help:    "Histogram of request latency in seconds.",
            Buckets: prometheus.DefBuckets,
        },
        []string{"route"},
    )
    errorCount = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "error_count",
            Help: "Total number of errors.",
        },
        []string{"route"},
    )
)

func init() {
    prometheus.MustRegister(requestLatency)
    prometheus.MustRegister(errorCount)
}

func RegisterMetricsHandler() {
    http.Handle("/metrics", promhttp.Handler())
}

func RecordLatency(route string, duration float64) {
    requestLatency.WithLabelValues(route).Observe(duration)
}

func IncrementErrorCount(route string) {
    errorCount.WithLabelValues(route).Inc()
```
```go
}
```

# QA Test Plan — Prometheus Metrics Endpoint (thai-gateway)

## 1. Acceptance Criteria

- **AC1**: `GET /metrics` returns HTTP 200 with `Content-Type: text/plain; version=0.0.4` and a body parseable by the Prometheus exposition format.
- **AC2**: After N routed requests, the body contains `request_latency_seconds_bucket`, `request_latency_seconds_sum`, and `request_latency_seconds_count` series with monotonically non-decreasing values.
- **AC3**: Rate-limit events increment a `rate_limit_hits_total` counter labeled by `route` and `reason` (e.g., `quota_exceeded`, `burst_exceeded`).
- **AC4**: Translation invocations increment a `translations_total` counter labeled by `provider`, `source_lang`, `target_lang`, and `status` (`success`/`error`).
- **AC5**: Every request (success or error) emits exactly one structured JSON log line containing `ts`, `level`, `method`, `path`, `status`, `latency_ms`, `request_id` fields; malformed JSON fails the test.
- **AC6**: `GET /health` returns HTTP 200 with body `{"status":"ok"}` when the gateway is up; returns HTTP 503 with `{"status":"degraded","reason":"..."}` when a declared dependency (e.g., upstream provider unreachable) is unhealthy.
- **AC7**: Error-rate metric `request_errors_total` is labeled by `status_class` (2xx/4xx/5xx) and `route`; 5xx responses increment it.
