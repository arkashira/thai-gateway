import pytest
from services.translation import translate_to_thai

def test_basic_translation():

def test_idempotence():
    prompt = "list features"
    first = translate_to_thai(prompt)
    second = translate_to_thai(prompt)
    assert first == second

def test_empty_prompt():
    assert translate_to_thai("") == ""

def test_whitespace_prompt():
    assert translate_to_thai("   ") == ""

    result = translate_to_thai(prompt)

    assert len(result) > 0
