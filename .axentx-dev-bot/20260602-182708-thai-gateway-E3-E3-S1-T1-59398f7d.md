import googletrans
import logging

class TranslationService:
    def __init__(self):
        self.translator = googletrans.Translator()
        logging.basicConfig(level=logging.INFO)

    def translate_prompt(self, prompt):
        """
        Translate the prompt to Thai.

        Args:
            prompt (str): The prompt to translate.

        Returns:
            str: The translated prompt in Thai.
        """
        try:
            thai_prompt = self.translator.translate(prompt, dest='th')
            logging.info(f"Translated '{prompt}' to Thai: {thai_prompt.text}")
            return thai_prompt.text
        except googletrans.exceptions.LangNotSupportedError:
            logging.warning(f"Language '{prompt}' is not supported. Skipping translation.")
            return None

    def translate_all_prompts(self, prompts):
        """
        Translate all incoming prompts to Thai.

        Args:
            prompts (list[str]): The list of prompts to translate.

        Returns:
            list[str]: The list of translated prompts in Thai.
        """
        thai_prompts = []
        for prompt in prompts:
            thai_prompt = self.translate_prompt(prompt)
            if thai_prompt:
                thai_prompts.append(thai_prompt)
        return thai_prompts

    def send_translated_prompt(self, thai_prompt, ai_provider):
        """
        Send the translated prompt to the AI provider.

        Args:
            thai_prompt (str): The translated prompt in Thai.
            ai_provider (str): The AI provider to send the prompt to.
        """
        # This will be implemented in the AI provider's service
        logging.info(f"Sending translated prompt '{thai_prompt}' to AI provider '{ai_provider}'")
