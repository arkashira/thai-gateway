import pytest
from unittest.mock import MagicMock, patch
from thai_gateway.services.translation import Translator, translate_and_forward

# 1. Happy‑path translation
def test_translate_success():
    translator = Translator(model_path="dummy")
    translator.translate = MagicMock(return_value="สร้างคำอธิบายสินค้า")
    result = translator.translate("generate product description")
    translator.translate.assert_called_once_with("generate product description")
    assert result == "สร้างคำอธิบายสินค้า"

# 2. Empty prompt handling
def test_translate_empty_prompt_raises():
    translator = Translator(model_path="dummy")
    with pytest.raises(ValueError, match="Prompt cannot be empty"):
        translator.translate("   ")

# 3. Model exception propagation
def test_translate_model_exception():
    translator = Translator(model_path="dummy")
    translator.translate = MagicMock(side_effect=RuntimeError("model crash"))
    with pytest.raises(RuntimeError):
        translator.translate("any text")

# 4. translate_and_forward happy path (mock AI provider)
@patch("thai_gateway.services.translation.AIProviderClient")
@patch("thai_gateway.services.translation.Translator")
def test_translate_and_forward_happy(mock_translator_cls, mock_ai_client_cls):
    mock_translator = mock_translator_cls.return_value
    mock_translator.translate.return_value = "สร้างคำอธิบายสินค้า"
    mock_ai_client = mock_ai_client_cls.return_value
    mock_ai_client.send_prompt.return_value = {"response": "OK"}

    resp = translate_and_forward("generate product description")
    mock_translator.translate.assert_called_once_with("generate product description")
    mock_ai_client.send_prompt.assert_called_once_with("สร้างคำอธิบายสินค้า")
    assert resp == {"response": "OK"}

# 5. translate_and_forward empty prompt returns 400
@patch("thai_gateway.services.translation.Translator")
def test_translate_and_forward_empty_prompt(mock_translator_cls):

    mock_translator_cls.assert_not_called()
