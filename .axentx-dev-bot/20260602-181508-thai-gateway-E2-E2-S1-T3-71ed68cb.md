// Translate translates a prompt into Thai
func (t *Translator) Translate(ctx context.Context, prompt string) (string, error) {
	startTime := time.Now()

	// Use the pre-configured translation API (e.g., Google Translate API)
	translatedPrompt, err := t.config.Translate(prompt)
	if err != nil {
		log.Printf("Translation failed: %v (prompt: %s)", err, prompt)
		return prompt, err
	}

	// Log translation latency
	latency := time.Since(startTime)
	if latency > 500*time.Millisecond {
		log.Printf("Translation latency exceeded: %v (prompt: %s, latency: %v)", latency, prompt, latency)
	}

	return translatedPrompt, nil
}
