```go
package main

import (
    "context"
    "log"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"
)

func main() {
    // Initialise configuration
    if err := loadConfig(); err != nil {
        log.Fatalf("config error: %v", err)
    }

    // Build router
    r := newRouter()

    // Create HTTP server
    srv := &http.Server{
        Addr:    ":8080",
        Handler: r,
        // Good defaults
        ReadTimeout:  10 * time.Second,
        WriteTimeout: 30 * time.Second,
        IdleTimeout:  60 * time.Second,
    }

    // Start server in a goroutine
    go func() {
        logger.Printf("thai-gateway listening on %s", srv.Addr)
        if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
            logger.Fatalf("listen: %v", err)
        }
    }()

    // Wait for interrupt signal to gracefully shutdown
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit
    logger.Println("Shutting down server...")

    ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
    defer cancel()
    if err := srv.Shutdown(ctx); err != nil {
        logger.Fatalf("server forced to shutdown: %v", err)
    }
    logger.Println("Server exited")
```
```go
}
```

1. **Configuration Loading** – `loadConfig` must:
   - Parse a valid JSON array from the `AI_PROVIDERS` env var and populate `config.Providers` (slice of structs with `Name` and `URL`).

   - Return an error containing the text *“no AI providers configured”* when neither env var is present or when JSON is malformed.
2. **Round‑Robin Provider Selection** – `nextProvider` must return providers in the exact order they appear in `config.Providers`, looping back to the first after the last, and must be safe for concurrent calls.
3. **HTTP Routing** – The server must expose:
   - `GET /` → returns HTTP 200 with a JSON body `{ "status": "ok" }`.
   - `POST /invoke` → forwards the request body to the provider returned by `nextProvider`, returns the provider’s response unchanged, and sets `X-Provider-Name` header to the selected provider’s name.
4. **Logging** – Every incoming request (method, path, timestamp, selected provider) must be emitted as a structured JSON log line to stdout.
5. **Graceful Shutdown** – On receiving SIGINT or SIGTERM the server must:
   - Stop accepting new connections within 1 second.
   - Allow in‑flight requests up to 15 seconds to complete before exiting with status 0.
6. **Deployment Path** – After `go build`, the binary must be runnable from `/opt/axentx/thai-gateway` without additional runtime configuration files.
