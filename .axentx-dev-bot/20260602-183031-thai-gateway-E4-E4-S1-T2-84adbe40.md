-// Placeholder for metrics implementation
+/**
+ * Prometheus metrics for the Thai gateway.
+ * Exposes:
+ *   - /metrics  : Prometheus exposition format
+ *   - /health   : simple health check
+ *
+ * Metrics:
+ *   - gateway_requests_total

+
+// Counter: total requests

+
+// Counter: total errors

+  help: 'Total number of error responses from the gateway',
+  labelNames: ['method', 'path', 'status'],
+});
+register.registerMetric(errorsTotal);
+
+// Histogram: request latency
+const latencyHistogram = new client.Histogram({
+  name: 'gateway_latency_seconds',
+  help: 'Request latency in seconds',
+  labelNames: ['method', 'path'],
+  buckets: [0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5, 10],
+});
+register.registerMetric(latencyHistogram);
+
+// Counter: rate‑limit exhaustion per provider
+const rateLimitExhausted = new client.Counter({
+  name: 'gateway_rate_limit_exhausted_total',
+  help: 'Total number of times a provider’s rate limit was exhausted',
+  labelNames: ['provider'],
+});
+register.registerMetric(rateLimitExhausted);
+
+/**
+ * Middleware to record metrics for each request.
+ */
+function metricsMiddleware(req, res, next) {
+  const end = latencyHistogram.startTimer({ method: req.method, path: req.path });
+  requestsTotal.inc({ method: req.method, path: req.path });
+
+  res.on('finish', () => {
+    if (res.statusCode >= 400) {
+      errorsTotal.inc({
+        method: req.method,
+        path: req.path,
+        status: res.statusCode,
+      });
+    }
+    end();
+  });
+
+  next();
+}
+
+/**
+ * Express router exposing /metrics and /health.
+ */
+const router = express.Router();
+
+router.get('/metrics', async (req, res) => {
+  try {
+    res.set('Content-Type', register.contentType);
+    res.end(await register.metrics());
+  } catch (ex) {
+    res.status(500).end(ex);
+  }
+});
+
+router.get('/health', (_req, res) => {
+  res.status(200).json({ status: 'ok', timestamp: new Date().toISOString() });
+});
+

## 1. Acceptance Criteria
- **AC1 – Metrics Exposure**  
  The `/metrics` endpoint must return a `text/plain` response containing valid Prometheus metrics in the standard exposition format.
- **AC2 – Rate‑Limit Counter**  
  A counter metric `gateway_rate_limit_exhausted_total` is incremented each time a provider’s rate limit is exhausted.
- **AC3 – Error Counter**  
  A counter metric `gateway_request_errors_total` is incremented for any non‑2xx HTTP response from the provider.
- **AC4 – Latency Histogram**  
  A histogram metric `gateway_request_latency_seconds` records the latency of each request to a provider, with buckets: 0.1, 0.5, 1, 5, 10 seconds.
- **AC5 – Health Check**  
  The `/health` endpoint returns HTTP 200 with JSON `{ "status": "ok" }` when the gateway is healthy.
- **AC6 – Provider Routing Log**  
  Each request logs the provider chosen and the timestamp to the console (or configured logger) in the format `"[YYYY-MM-DDTHH:mm:ssZ] routed to <provider>"`.
- **AC7 – Alert Trigger**  
  When `gateway_rate_limit_exhausted_total` reaches a configurable threshold (e.g., 100), an alert is emitted via a mock alerting service (e.g., console.warn or a stubbed notifier).

// tests/metrics.test.js
const { createMetricsRouter } = require('../src/metrics');
const request = require('supertest');
const express = require('express');

describe('Metrics Router', () => {
  let app, router;

  beforeAll(() => {
    router = createMetricsRouter();
    app = express();
    app.use(router);
  });

  test('GET /metrics returns Prometheus format', async () => {
    const res = await request(app).get('/metrics');
    expect(res.status).toBe(200);
    expect(res.headers['content-type']).toMatch(/text\/plain/);
    expect(res.text).toMatch(/gateway_rate_limit_exhausted_total/);
  });

  test('Rate limit counter increments on exhaustion', async () => {
    const mockProvider = { name: 'mock', rateLimit: 0 };
    await router.handleRequest(mockProvider); // internal helper
    const res = await request(app).get('/metrics');
    expect(res.text).toMatch(/gateway_rate_limit_exhausted_total 1/);
  });

  test('Error counter increments on provider error', async () => {
    const mockProvider = { name: 'mock', respondWithError: true };
    await router.handleRequest(mockProvider);
    const res = await request(app).get('/metrics');
    expect(res.text).toMatch(/gateway_request_errors_total 1/);
  });

  test('Latency histogram records latency', async () => {
    const mockProvider = { name: 'mock', latencyMs: 350 };
    await router.handleRequest(mockProvider);
    const res = await request(app).get('/metrics');
    expect(res.text).toMatch(/gateway_request_latency_seconds_bucket{le="0.5"} 1/);
  });

  test('/health returns 200 and ok status', async () => {
    const res = await request(app).get('/health');
    expect(res.status).toBe(200);
    expect(res.body).toEqual({ status: 'ok' });
  });
});
