2. Modify the `translate.go` file to include the following changes:

```go
package middleware

import (
    // ... existing imports
    "github.com/olekukonko/go-ole/ole"
)

// ... existing code

func Translate(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // Create an instance of the OLE automation server
        server, err := ole.NewServer("sapi.spLanguage", &ole.ServerOptions{
            AppID:   "Thai Gateway",
            Callback: func(msg ole.Message) {
                // Log the translated prompt for debugging
                translatedPrompt := string(msg.Data)
                log.Println("Translated prompt:", translatedPrompt)
            },
        })
        if err != nil {
            http.Error(w, "Failed to create OLE automation server", http.StatusInternalServerError)
            return
        }
        defer server.Release()

        // Set the language to Thai
        server.Call("SpeechSpeak", "Thai", r.URL.Path, r.Method, r.Header.Get("Content-Type"), r.Body)

        next.ServeHTTP(w, r)
    })
}

// ... existing code
```
