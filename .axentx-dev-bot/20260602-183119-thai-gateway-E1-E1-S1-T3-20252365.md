// /opt/axentx/thai-gateway/src/services/translator.ts

import { Provider } from '../providers';

interface TranslateRequest {
  prompt: string;
}

interface TranslateResponse {
  response_text: string;
  provider_used: string;
}

class Translator {
  private providers: Provider[];

  constructor(providers: Provider[]) {
    this.providers = providers;
  }

  async translate(request: TranslateRequest): Promise<TranslateResponse> {
    const provider = this.getLeastCongestedProvider();
    const response = await provider.translate(request.prompt);
    return {
      response_text: response,
      provider_used: provider.name,
    };
  }

  private getLeastCongestedProvider(): Provider {
    // Simple implementation to get the least-congested provider
    // In a real-world scenario, this would be more complex and consider various factors
    return this.providers.reduce((leastCongested, current) => {
      if (current.congestionLevel < leastCongested.congestionLevel) {
        return current;
      }
      return leastCongested;
    }, this.providers[0]);
  }
}

export { Translator };
```
