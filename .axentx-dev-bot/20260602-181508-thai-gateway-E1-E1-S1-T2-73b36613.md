### `src/translation/validator.py`

"""
Utility functions for validating Thai translations.
"""

def validate_prompt_translation(original: str, translated: str) -> bool:
    """
    Basic quality check for a Thai translation.

    * Rejects empty or whitespace‑only translations.
    * Rejects translations that are less than 50 % the length of the original
      (a crude proxy for missing content).
    * Rejects translations that contain the original English words
      (indicating a failed translation).

    Parameters
    ----------
    original : str
        The source prompt in English.
    translated : str
        The translated prompt in Thai.

    Returns
    -------
    bool
        True if the translation passes all checks, False otherwise.
    """
    if not translated or not translated.strip():
        return False

    if len(translated) < len(original) / 2:
        return False

    # Simple check for English words leaking into Thai
    english_words = set(original.lower().split())
    if any(word in translated.lower() for word in english_words):
        return False

    return True
```
