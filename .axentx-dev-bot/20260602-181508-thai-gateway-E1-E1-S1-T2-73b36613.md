type Provider struct {
	Name      string
	Quota     int

var (

		prometheus.GaugeOpts{
			Name: "provider_load",
			Help: "Current load of each AI provider",
		},
		[]string{"provider"},
	)
	routingDecisions = prometheus.NewCounterVec(
		prometheus.CounterOpts{
			Name: "routing_decisions_total",
			Help: "Total routing decisions made for each AI provider",

	)
)

	priorityProviders = getPriorityProviders()
	go func() {
		ticker := time.NewTicker(30 * time.Second)
		for range ticker.C {
			healthCheck()
		}
	}()
}

func getPriorityProviders() []string {
	providers := os.Getenv("PRIORITY_PROVIDERS")
	if providers == "" {
		return []string{}
	}
	return strings.Split(providers, ",")
}

func healthCheck() {
	// Implement health check logic here
	// Update Provider struct fields based on health check results
}

func balanceRequest(ctx context.Context, request *Request) (*Response, error) {
	providers := getProviders()
	sort.Slice(providers, func(i, j int) bool {
		providerI := providers[i]
		providerJ := providers[j]
		if contains(priorityProviders, providerI.Name) && !contains(priorityProviders, providerJ.Name) {
			return true
		}
		if providerI.Quota < providerJ.Quota {
			return true
		}
		if providerI.LastHealthCheck.Before(providerJ.LastHealthCheck) {
			return true
		}
		return false
	})

	providerMetrics.WithLabelValues(providers[0].Name).Set(float64(providers[0].Quota))
	routingDecisions.WithLabelValues(providers[0].Name).Inc()

	return routeRequest(ctx, request, providers[0])
}

func contains(slice []string, element string) bool {
	for _, a := range slice {
		if a == element {
			return true
		}
	}
	return false
}

	prometheus.MustRegister(providerMetrics)
	prometheus.MustRegister(routingDecisions)
