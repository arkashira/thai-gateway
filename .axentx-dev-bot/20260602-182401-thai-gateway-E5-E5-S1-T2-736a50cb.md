```javascript
// src/middleware/requestId.js

/**
 * Generates a unique request ID and attaches it to the request/response.
 * Logs the incoming request with the ID for tracing.
 */
const generateId = () => Math.random().toString(36).substr(2, 9);

module.exports = (req, res, next) => {
  const requestId = generateId();

  // Attach to request object for internal use
  req.id = requestId;

  // Set header for downstream services
  res.setHeader('X-Request-ID', requestId);

  // Log request with ID for tracing
  console.log(`[${new Date().toISOString()}] ${req.method} ${req.path} - RequestID: ${requestId}`);

  next();
};
```

**1. Acceptance criteria**  
- A UUID v4 is generated for every incoming HTTP request that does not already contain a request‑ID.  
- The generated ID is attached to `req.requestId` and is available to all downstream middleware/handlers.  
- The ID is propagated back to the client via the `X-Request-ID` response header.  
- Every log entry produced during the handling of a request includes the request‑ID (e.g., via a logger that reads `req.requestId`).  
- If the client supplies an `X-Request-ID` header, the middleware preserves that value, does not overwrite it, and still propagates it downstream.  
- The middleware does not modify any other request or response fields.  
- The middleware is idempotent: calling it twice on the same request does not change the ID or headers.
