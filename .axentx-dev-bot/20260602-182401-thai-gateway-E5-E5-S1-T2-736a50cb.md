## 2. Unit Tests (Jest – Node.js)

```js
// __tests__/requestId.test.js
const request = require('supertest');
const express = require('express');
const requestIdMiddleware = require('../src/middleware/requestId');

describe('requestId middleware', () => {
  let app;

  beforeEach(() => {
    app = express();
    app.use(requestIdMiddleware);
    app.get('/test', (req, res) => {
      res.json({ id: req.requestId });
    });
  });

  test('adds X-Request-ID when missing', async () => {
    const res = await request(app).get('/test');
    expect(res.headers['x-request-id']).toBeDefined();
    expect(res.body.id).toBe(res.headers['x-request-id']);
    expect(/^[0-9a-f]{8}-[0-9a-f]{4}-4[0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/.test(res.headers['x-request-id'])).toBe(true);
  });

  test('passes through client-provided X-Request-ID', async () => {
    const id = 'client-12345';
    const res = await request(app).get('/test').set('X-Request-ID', id);
    expect(res.headers['x-request-id']).toBe(id);
    expect(res.body.id).toBe(id);
  });

  test('does not modify other headers', async () => {
    const res = await request(app).get('/test').set('Accept', 'application/json');
    expect(res.headers['accept']).toBe('application/json');
  });

  test('requestId is available in downstream handlers', async () => {
    const res = await request(app).get('/test');
    expect(res.body.id).toBeDefined();
  });

  test('handles concurrent requests without ID collision', async () => {
    const promises = Array.from({ length: 1000 }, () =>
      request(app).get('/test')
    );
    const responses = await Promise.all(promises);
    const ids = responses.map(r => r.headers['x-request-id']);
    const uniqueIds = new Set(ids);
    expect(uniqueIds.size).toBe(ids.length);
  });
});
```
