# tests/test_provider_factory.py
import json
import yaml
import tempfile
import os
import pytest
from thai_gateway.providers.factory import ProviderFactory, ProviderConfigError

@pytest.fixture
def factory():
    return ProviderFactory()

def write_temp_file(content, suffix):
    fd, path = tempfile.mkstemp(suffix=suffix)
    with os.fdopen(fd, 'w') as f:
        f.write(content)
    return path

def test_load_yaml_providers(factory):
    yaml_content = yaml.dump([
        {"name": "alpha", "endpoint": "https://alpha.ai", "rate_limit": 50},
        {"name": "beta",  "endpoint": "https://beta.ai",  "rate_limit": 75}
    ])
    cfg_path = write_temp_file(yaml_content, ".yaml")
    providers = factory.load_providers(cfg_path)

    assert isinstance(providers, list)
    assert len(providers) == 2
    assert providers[0].name == "alpha"
    assert providers[0].endpoint == "https://alpha.ai"
    assert providers[0].rate_limit == 50

def test_load_json_providers(factory):
    json_content = json.dumps([
        {"name": "gamma", "endpoint": "https://gamma.ai", "rate_limit": 30}
    ])
    cfg_path = write_temp_file(json_content, ".json")
    providers = factory.load_providers(cfg_path)

    assert len(providers) == 1
    assert providers[0].name == "gamma"
    assert providers[0].rate_limit == 30

def test_missing_required_keys(factory):
    bad_yaml = yaml.dump([{"name": "delta", "endpoint": "https://delta.ai"}])  # missing rate_limit
    cfg_path = write_temp_file(bad_yaml, ".yaml")
    with pytest.raises(ProviderConfigError) as exc:
        factory.load_providers(cfg_path)
    assert "rate_limit" in str(exc.value)

def test_invalid_file_type(factory):
    bad_file = write_temp_file("just some text", ".txt")
    with pytest.raises(ProviderConfigError) as exc:
        factory.load_providers(bad_file)
    assert "Unsupported file type" in str(exc.value)

def test_idempotent_load(factory):
    yaml_content = yaml.dump([{"name": "epsilon", "endpoint": "https://epsilon.ai", "rate_limit": 20}])
    cfg_path = write_temp_file(yaml_content, ".yaml")
    first_load = factory.load_providers(cfg_path)
    second_load = factory.load_providers(cfg_path)
    assert first_load is not second_load  # new list each call
    assert first_load[0].name == second_load[0].name
