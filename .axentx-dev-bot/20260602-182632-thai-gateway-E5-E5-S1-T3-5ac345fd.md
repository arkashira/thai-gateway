### 1️⃣ Dockerfile – multi‑stage, language‑agnostic template  

```dockerfile
# ------------------------------------------------------------
# Multi‑stage Dockerfile for thai‑gateway
# ------------------------------------------------------------
# Stage 1 – Builder (install deps, compile if needed)
# ------------------------------------------------------------
ARG BASE_IMAGE=node:20-alpine   # default for a Node.js project
# If the project is Python‑based, override at build time:
#   docker build --build-arg BASE_IMAGE=python:3.11-slim -t thai-gateway .
FROM ${BASE_IMAGE} AS builder

# Create a non‑root user early (helps with permission caching)
RUN addgroup -S appgroup && adduser -S appuser -G appgroup
WORKDIR /app
USER appuser

# ---------- Dependency install ----------
# Node.js layout
COPY --chown=appuser:appgroup package*.json ./
RUN if [ -f package.json ]; then \
        npm ci --only=production; \
    fi

# Python layout
COPY --chown=appuser:appgroup requirements.txt ./
RUN if [ -f requirements.txt ]; then \
        pip install --no-cache-dir -r requirements.txt; \
    fi

# ---------- Source copy ----------
COPY --chown=appuser:appgroup . .

# If the project uses a build step (e.g., TypeScript, Babel, etc.)
# uncomment the line below:
# RUN npm run build

# ------------------------------------------------------------
# Stage 2 – Runtime (tiny image, non‑root)
# ------------------------------------------------------------
FROM ${BASE_IMAGE} AS runtime

# Re‑create the same non‑root user (keeps UID/GID consistent)
RUN addgroup -S appgroup && adduser -S appuser -G appgroup
USER appuser
WORKDIR /app

# Copy only the artefacts we need from the builder
COPY --from=builder /app /app

# Expose the default listening port – can be overridden at runtime
EXPOSE 3000

# Default environment (can be overridden with `docker run -e …`)
ENV NODE_ENV=production \
    PORT=3000

# Entrypoint – works for both Node and Python projects
#  * Node: expects an entry file at src/index.js (adjust if different)
#  * Python: expects a module called gateway.py with a `main()` function
CMD if [ -f src/index.js ]; then \
        node src/index.js; \
    elif [ -f gateway.py ]; then \
        python -m gateway; \
    else \
        echo "❌ No recognised entry point found (src/index.js or gateway.py)"; exit 1; \
    fi
```
