```go
package metrics

import (
	"fmt"
	"net/http"
	"time"

	"github.com/prometheus/client_golang/prometheus"
	"github.com/prometheus/client_golang/prometheus/promauto"
)

// providerRequestsTotal counts total requests per provider.
var providerRequestsTotal = promauto.NewCounterVec(
	prometheus.CounterOpts{
		Name: "thai_gateway_provider_requests_total",
		Help: "Total number of requests sent to each AI provider.",
	},
	[]string{"provider"},
)

// providerRequestsLatency records request latency per provider.
var providerRequestsLatency = promauto.NewHistogramVec(
	prometheus.HistogramOpts{
		Name:    "thai_gateway_provider_requests_latency_seconds",
		Help:    "Latency of requests to each AI provider (seconds).",
		Buckets: []float64{0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0},
	},
	[]string{"provider"},
)

// IncRequest increments the request counter for the given provider.
func IncRequest(provider string) {
	providerRequestsTotal.WithLabelValues(provider).Inc()
}

// ObserveLatency records the latency for the given provider.
func ObserveLatency(provider string, d time.Duration) {
	providerRequestsLatency.WithLabelValues(provider).Observe(d.Seconds())
}

// ExposeMetrics registers the `/metrics` HTTP handler.
// Call this once from your main initialization code.
func ExposeMetrics(mux *http.ServeMux) {
	mux.Handle("/metrics", promhttp.Handler())
}
```
