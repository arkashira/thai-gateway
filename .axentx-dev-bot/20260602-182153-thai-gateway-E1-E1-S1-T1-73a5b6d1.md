// /opt/axentx/thai-gateway/src/routes/ai.js

const express = require('express');
const { logger } = require('../utils/logger');
const { forwardRequest } = require('../utils/ai-proxy');

const router = express.Router();

router.post('/', async (req, res) => {
  try {
    const { provider, payload } = req.body;
    
    if (!provider || !payload) {
      return res.status(400).json({ error: 'Provider and payload are required' });
    }
    
    // Log request
    const logEntry = {
      provider,
      timestamp: new Date().toISOString(),
      status: 'processing',
      payload: JSON.stringify(payload)
    };
    logger.info('AI request received', logEntry);
    
    // Forward request to AI provider
    const response = await forwardRequest(provider, payload);
    
    // Log response status
    const responseLog = {
      provider,
      timestamp: new Date().toISOString(),
      status: 'completed',
      response: JSON.stringify(response)
    };
    logger.info('AI response received', responseLog);
    
    // Return the provider's response unchanged
    res.json(response);
  } catch (error) {
    logger.error('AI request failed', {
      provider: req.body.provider,
      error: error.message,
      timestamp: new Date().toISOString()
    });
    res.status(500).json({ error: 'Failed to process AI request' });
  }
});

module.exports = router;
```
