```go
// translator.go
package translate

import (
	"context"
	"encoding/json"
	"fmt"
	"log"

	"github.com/nlpcloud/nlpcloud-go"
)

// Translator struct
type Translator struct {
	client *nlpcloud.Client
}

// NewTranslator returns a new Translator instance
func NewTranslator(apiKey string) *Translator {
	client := nlpcloud.NewClient(apiKey)
	return &Translator{client: client}
}

// Translate translates the given prompt into Thai
func (t *Translator) Translate(prompt string) (string, error) {
	ctx := context.Background()
	translation, err := t.client.Translate(ctx, prompt, "th")
	if err != nil {
		return "", err
	}
	return translation.TranslatedText, nil
}

// TranslateJSONPayload translates the 'prompt' field in the given JSON payload
func (t *Translator) TranslateJSONPayload(payload []byte) (string, error) {
	var data map[string]string
	err := json.Unmarshal(payload, &data)
	if err != nil {
		return "", err
	}
	prompt, ok := data["prompt"]
	if !ok {
		return "", fmt.Errorf("missing 'prompt' field in JSON payload")
	}
	translatedPrompt, err := t.Translate(prompt)
	if err != nil {
		return "", err
	}
	return translatedPrompt, nil
```
```go
}
```

### 1. Acceptance Criteria
- **AC1**: The gateway accepts a JSON POST request with a top‑level `prompt` string field; any request missing this field returns HTTP 400.
- **AC2**: The `prompt` value is passed to the translation component, which returns a Thai‑language string; translation failures return HTTP 502.
- **AC3**: The translated Thai string is included in the outbound request body to the configured AI provider under the same `prompt` key.
- **AC4**: The translation component loads its API key (or model credentials) exclusively from an environment variable `THAI_TRANSLATE_API_KEY`; the process fails to start if the variable is unset.
- **AC5**: No secret (API key) appears in any source file, binary, or logs; the key is never logged at INFO or DEBUG level.
- **AC6**: The gateway logs a structured entry (`translation_success` or `translation_error`) with request ID, original language, and target language, but without the raw API key.
- **AC7**: End‑to‑end latency added by translation does not exceed 500 ms (95th percentile) under load of 100 RPS.
