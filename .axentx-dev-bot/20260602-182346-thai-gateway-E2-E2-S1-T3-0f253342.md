// src/main/route.rs
use axentx::provider::Provider;
use axentx::util::load_balancing::ProviderLoad;
use axentx::util::load_balancing::ProviderLoadComparator;

pub fn balance_provider(
    providers: &[Provider],
    request: &Request,
) -> Provider {
    // Find the provider with the lowest current load
    let mut min_load_provider = providers[0];
    let mut min_load = ProviderLoad::MAX;

    for provider in providers {
        let load = provider.load();
        if load < min_load {
            min_load_provider = provider;
            min_load = load;
        }
    }

    min_load_provider
}
