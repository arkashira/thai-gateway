// HealthCheck represents a health check for an AI provider.
type HealthCheck struct {
	Provider string
	Timeout  time.Duration
}

// NewHealthCheck returns a new HealthCheck instance.
func NewHealthCheck(provider string, timeout time.Duration) *HealthCheck {
	return &HealthCheck{
		Provider: provider,
		Timeout:  timeout,
	}
}

// RunHealthCheck runs a health check on the specified provider.
func (hc *HealthCheck) RunHealthCheck(ctx context.Context) error {
	// Create a new HTTP client.
	client := &http.Client{
		Timeout: hc.Timeout,
	}

	// Create a new request.
	req, err := http.NewRequest("GET", hc.Provider, nil)
	if err != nil {
		return err
	}

	// Set the context for the request.
	req = req.WithContext(ctx)

	// Send the request and get the response.
	resp, err := client.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()

	// Check if the response is successful.
	if resp.StatusCode >= 500 {
		return fmt.Errorf("health check failed for %s: %d", hc.Provider, resp.StatusCode)
	}

	return nil
}
