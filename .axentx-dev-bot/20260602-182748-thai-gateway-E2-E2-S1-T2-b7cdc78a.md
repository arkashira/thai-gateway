func selectProvider() *Provider {
	var selectedProvider *Provider
	highestQuota := int64(-1)
	roundRobinIndex := atomic.LoadInt64(&roundRobinCounter)

	for i, p := range Providers {
		if p.RemainingQuota > highestQuota {

			selectedProvider = &Providers[i]

			if int(roundRobinIndex) == i {

		}

	return selectedProvider
}

- **AC‑1**: The middleware selects the provider whose *remaining quota* (`totalQuota - usedQuota`) is the greatest at the moment of request handling.  
- **AC‑2**: When two or more providers have identical highest remaining quota, the middleware chooses among them using a round‑robin algorithm that cycles on each subsequent request.  
- **AC‑3**: The selected provider’s endpoint is used to forward the incoming HTTP request **without** altering any original request headers (including `Authorization`, `User‑Agent`, custom headers, etc.).  
- **AC‑4**: If **no providers** are configured, the middleware returns HTTP 502 Bad Gateway with a JSON error payload `{ "error": "no providers available" }`.  
- **AC‑5**: Provider selection is performed **concurrently safe** – multiple goroutines handling requests must not corrupt the round‑robin state.  
- **AC‑6**: The middleware logs the chosen provider identifier at INFO level and logs a warning if a tie‑break occurs.
