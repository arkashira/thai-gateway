// src/providers/formatters.js
const { MalformedRequestError } = require('./errors');

/**
 * Minimal schema validators for each provider.
 * These are intentionally lightweight – only the fields required by the gateway.
 */
const validators = {
    openai: (payload) => {
        // Expect { model, messages: [{role, content}], temperature? }
        if (!payload || typeof payload !== 'object') return false;
        if (!payload.model || typeof payload.model !== 'string') return false;
        if (!Array.isArray(payload.messages)) return false;
        if (!payload.messages.every(m => m.role && m.content)) return false;
        return true;
    },
    anthropic: (payload) => {
        // Expect { model, messages: [{role, content}], max_tokens? }
        if (!payload || typeof payload !== 'object') return false;
        if (!payload.model || typeof payload.model !== 'string') return false;
        if (!Array.isArray(payload.messages)) return false;
        if (!payload.messages.every(m => m.role && m.content)) return false;
        return true;
    },
};

/**
 * Provider-specific request formatters.
 * @param {string} provider - 'openai' | 'anthropic'
 * @param {object} payload - raw user payload
 * @returns {object} formatted request body
 * @throws {MalformedRequestError}
 */
function formatRequest(provider, payload) {
    const validator = validators[provider];
    if (!validator) {
        throw new Error(`Unsupported provider: ${provider}`);
    }

    if (!validator(payload)) {
        console.warn(
            `Malformed request for ${provider}:`,
            JSON.stringify(payload, null, 2)
        );
        throw new MalformedRequestError(provider, payload);
    }

    // Format according to provider schema
    switch (provider) {
        case 'openai':
            return {
                model: payload.model,
                messages: payload.messages,
                temperature: payload.temperature ?? 0.7,
            };
        case 'anthropic':
            return {
                model: payload.model,
                messages: payload.messages,
                max_tokens: payload.max_tokens ?? 1024,
            };
        default:
            // unreachable due to earlier guard
            return {};
    }
}

module.exports = { formatRequest, MalformedRequestError };
```
