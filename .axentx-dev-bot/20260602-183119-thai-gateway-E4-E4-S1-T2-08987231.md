```bash
#!/usr/bin/env bash
#
# Build the Thai-Gateway Docker image.
#
set -euo pipefail

IMAGE_NAME="thai-gateway"
TAG="latest"

echo "🔨 Building Docker image ${IMAGE_NAME}:${TAG} ..."
docker build -t "${IMAGE_NAME}:${TAG}" .

echo "✅ Build complete. Image ID:"
docker images --filter=reference="${IMAGE_NAME}:${TAG}" --format "{{.ID}}"

# Optional: push to a registry if $REGISTRY is set
if [[ -n "${REGISTRY:-}" ]]; then
  FULL_NAME="${REGISTRY}/${IMAGE_NAME}:${TAG}"

  docker tag "${IMAGE_NAME}:${TAG}" "${FULL_NAME}"
  docker push "${FULL_NAME}"
fi
