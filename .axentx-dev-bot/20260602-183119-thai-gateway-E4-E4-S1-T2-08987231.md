## 2. `Dockerfile`

```dockerfile
# /opt/axentx/thai-gateway/Dockerfile
# ---- Build stage ---------------------------------------------------------
FROM python:3.12-slim AS builder

# Install build‑time deps (if any)
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Copy only the files we need for the build
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy the rest of the source
COPY . .

# ---- Runtime stage -------------------------------------------------------
FROM python:3.12-slim

WORKDIR /app

# Copy built artefacts from builder
COPY --from=builder /app /app

# Copy the default config into a known location
COPY config.yaml /etc/thai-gateway/config.yaml

# Expose the API port
EXPOSE 8080

# Default command – the gateway entrypoint
CMD ["python", "-m", "thai_gateway.main"]
