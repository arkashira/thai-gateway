**1. Acceptance Criteria**
- The gateway reads a list of AI providers and their rate‑limit quotas from environment variables at startup.  
- For each provider, the gateway maintains a real‑time counter of remaining requests for the current window.  
- When a request arrives, the gateway selects the provider with the highest remaining quota and forwards the request to it.  
- If the selected provider’s remaining quota is 0, the gateway automatically falls back to the next provider with the highest remaining quota.  
- If **all** providers have exhausted their quota, the gateway returns HTTP 429 (Too Many Requests) with a descriptive error payload.  
- A `/metrics/rate_limits` endpoint returns a JSON object containing, for every provider: `provider_id`, `total_quota`, `remaining`, `window_reset_timestamp`.  
- The rate‑limit window duration is configurable per provider via an env var (e.g., `PROVIDER_<NAME>_WINDOW_SECONDS`).

// rate_limit_test.go
package ratelimit_test

import (
    "testing"
    "time"

    "github.com/stretchr/testify/assert"
    "github.com/axentx/thai-gateway/config"
    "github.com/axentx/thai-gateway/internal/ratelimit"
)

func TestLoadConfigFromEnv(t *testing.T) {
    t.Setenv("PROVIDER_OPENAI_QUOTA", "1000")
    t.Setenv("PROVIDER_OPENAI_WINDOW_SECONDS", "60")
    t.Setenv("PROVIDER_CLAUDE_QUOTA", "500")
    t.Setenv("PROVIDER_CLAUDE_WINDOW_SECONDS", "60")

    cfg, err := config.LoadRateLimitConfig()
    assert.NoError(t, err)
    assert.Len(t, cfg.Providers, 2)
    assert.Equal(t, 1000, cfg.Providers["openai"].Quota)
    assert.Equal(t, 60, cfg.Providers["openai"].WindowSeconds)
    assert.Equal(t, 500, cfg.Providers["claude"].Quota)
}

func TestRemainingDecrementsOnUse(t *testing.T) {
    rl := ratelimit.NewTracker(map[string]ratelimit.ProviderConfig{
        "openai": {Quota: 10, WindowSeconds: 60},
    })
    assert.Equal(t, 10, rl.Remaining("openai"))

    rl.Consume("openai")
    assert.Equal(t, 9, rl.Remaining("openai"))
}

func TestWindowReset(t *testing.T) {
    rl := ratelimit.NewTracker(map[string]ratelimit.ProviderConfig{
        "openai": {Quota: 2, WindowSeconds: 1},
    })
    rl.Consume("openai")
    rl.Consume("openai")
    assert.Equal(t, 0, rl.Remaining("openai"))

    // wait for window to expire
    time.Sleep(1100 * time.Millisecond)
    rl.CleanupExpiredWindows()
    assert.Equal(t, 2, rl.Remaining("openai"))
}

func TestSelectProviderHighestRemaining(t *testing.T) {
    rl := ratelimit.NewTracker(map[string]ratelimit.ProviderConfig{
        "openai": {Quota: 5, WindowSeconds: 60},
        "claude": {Quota: 10, WindowSeconds: 60},
    })
    // consume some from claude to make openai highest
    for i := 0; i < 8; i++ {
        rl.Consume("claude")
    }
    selected := rl.SelectProvider()
    assert.Equal(t, "openai", selected)
}

func TestFallbackWhenPrimaryExhausted(t *testing.T) {
    rl := ratelimit.NewTracker(map[string]ratelimit.ProviderConfig{
        "openai": {Quota: 1, WindowSeconds: 60},
        "claude": {Quota: 5, WindowSeconds: 60},
    })
    rl.Consume("openai") // exhaust openai
    selected := rl.SelectProvider()
    assert.Equal(t, "claude", selected)
}

func TestAllProvidersExhaustedReturnsEmpty(t *testing.T) {

        "openai": {Quota: 0, WindowSeconds: 60},
        "claude": {Quota: 0, WindowSeconds: 60},
    })
    selected := rl.SelectProvider()

}
