// /opt/axentx/thai-gateway/internal/translation/middleware.go
package translation

import (
    "bytes"
    "encoding/json"
    "log"
    "net/http"
    "time"

    "github.com/axentx/thai-gateway/config"
)

func TranslateMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        if r.Header.Get("language") == "th" {
            originalPrompt := r.FormValue("prompt")

            translatedPrompt, err := translateToThai(originalPrompt)
            if err != nil {
                log.Printf("Translation failed: %v. Falling back to original prompt.", err)
                translatedPrompt = originalPrompt
            }

            r.Form.Set("prompt", translatedPrompt)
        }
        next.ServeHTTP(w, r)
    })
}

func translateToThai(text string) (string, error) {
    client := &http.Client{
        Timeout: 500 * time.Millisecond,
    }

    requestBody, _ := json.Marshal(map[string]interface{}{
        "q":      text,
        "target": "th",
        "key":    config.TranslationAPIKey,
    })

    req, err := http.NewRequest("POST", config.TranslationURL, bytes.NewBuffer(requestBody))
    if err != nil {
        return "", err
    }
    req.Header.Set("Content-Type", "application/json")

    resp, err := client.Do(req)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return "", fmt.Errorf("failed to translate: %s", resp.Status)
    }

    var result map[string]interface{}
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return "", err
    }

    translations := result["data"].(map[string]interface{})["translations"].([]interface{})
    return translations[0].(map[string]interface{})["translatedText"].(string), nil
}
```
