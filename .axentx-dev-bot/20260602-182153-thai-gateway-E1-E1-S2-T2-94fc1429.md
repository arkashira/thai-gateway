// /opt/axentx/thai-gateway/src/services/router.js
const redis = require('../utils/redisClient');
const providers = require('../config/providers');

// Helper: get the next available provider based on remaining quota
async function getNextProvider() {
  const quotaKeys = providers.map(p => `provider:${p.name}:quota`);
  const quotas = await redis.mget(...quotaKeys);

  // Find first provider with quota > 0
  for (let i = 0; i < providers.length; i++) {
    const quota = parseInt(quotas[i], 10);
    if (!isNaN(quota) && quota > 0) {
      return providers[i];
    }
  }
  return null; // all exhausted
}

// Helper: decrement quota for a provider atomically
async function decrementQuota(provider) {
  const key = `provider:${provider.name}:quota`;
  // Use Redis DECR to avoid race conditions
  await redis.decr(key);
}

// Helper: get provider quota from Redis
async function getProviderQuota(provider) {
  const key = `provider:quota:${provider}`;
  const value = await redis.get(key);
  if (value) {
    return JSON.parse(value);
  }
  return { remaining: 0, ttl: 3600 };
}

// Helper: set provider quota in Redis
async function setProviderQuota(provider, quota) {
  const key = `provider:quota:${provider}`;
  await redis.set(key, JSON.stringify(quota), 'EX', 3600);
}

// Existing router logic
module.exports = async function router(req, res) {
  const provider = await getNextProvider();

  if (!provider) {
    // All providers exhausted – return 429
    res.set('Retry-After', '60'); // 60 seconds retry
    return res.status(429).json({ error: 'All providers exhausted. Please retry later.' });
  }

  // Get provider quota
  const quota = await getProviderQuota(provider.name);
  if (quota.remaining <= 0) {
    // provider is exhausted, return 429 with retry-after header
    res.status(429).header('Retry-After', 60).send('Too Many Requests');
    return;
  }

  // Decrement quota for the chosen provider
  await decrementQuota(provider);

  // Forward request to the chosen provider (simplified)
  try {
    const providerRes = await provider.client.send(req.body);
    return res.status(providerRes.status).json(providerRes.data);
  } catch (err) {
    // If provider fails, try next provider recursively
    return router(req, res);
  }
};

## review — reviewer @ 2026-06-03T03:30:56.212107Z

REJECT: The implementation contains multiple functional inconsistencies that would prevent correct operation:

- Inconsistent Redis key naming: `getNextProvider` uses `provider:${p.name}:quota`, while `decrementQuota` and `getProviderQuota` use different key patterns (`provider:${provider.name}:quota` vs `provider:quota:${provider}`).
- `decrementQuota` performs a simple `DECR` on a key that is expected to hold a JSON string, causing type mismatch and incorrect quota handling.
- `getProviderQuota` parses a JSON string but the decrement logic never updates that JSON structure, leading to stale quota data.
- Recursive call to `router(req, res)` on provider failure risks stack overflow and does not properly propagate errors or retry limits.
- The test suite references `setProviderQuota` from a different module (`../stores/redis`) while the implementation defines it in the same file, causing import mismatches.

These issues prevent the feature from working as intended and should be addressed before acceptance.

## 1. Acceptance Criteria

1. **Quota Persistence** – The gateway writes the remaining quota for each provider to Redis with a TTL that equals the provider’s quota window (seconds).  
2. **Automatic Provider Rotation** – When the current provider’s remaining quota would be exceeded by an incoming request, the gateway automatically selects the next provider in the configured rotation and retries the request.  
3. **Global Exhaustion Handling** – If all providers are exhausted, the gateway returns **HTTP 429** with a `Retry‑After` header set to the earliest remaining TTL among all providers.  
4. **Deterministic Order** – Provider selection follows the order defined in `config.providers` and never skips a provider that still has quota.  
5. **Audit Logging** – Every quota decrement, provider switch, and 429 response is logged with provider ID, remaining quota, and timestamp.
