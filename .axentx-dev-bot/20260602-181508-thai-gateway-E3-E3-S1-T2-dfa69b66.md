// Package ratelimit provides utilities for tracking and reporting
// cumulative rate limits across providers.
package ratelimit

import (
	"net/http"
	"strconv"
)

// header names used by the gateway
const (
	HeaderLimit   = "X-RateLimit-Limit"
	HeaderRemaining = "X-RateLimit-Remaining"
)

// writeRateLimitHeaders writes the current rate-limit state to the
// response headers. It expects the limiter to be stored in the request
// context under the key `LimiterContextKey` (defined in limiter.go).
func writeRateLimitHeaders(w http.ResponseWriter, r *http.Request) {
	limiter, ok := r.Context().Value(LimiterContextKey).(*Limiter)
	if !ok || limiter == nil {
		// No limiter in context – nothing to report
		return
	}

	// The limiter keeps the total limit and the remaining count.
	w.Header().Set(HeaderLimit, strconv.Itoa(limiter.TotalLimit))
	w.Header().Set(HeaderRemaining, strconv.Itoa(limiter.Remaining))
}
```

import (
	"net/http"
	"time"

	"github.com/axentx/thai-gateway/internal/ratelimit"
)

// RateLimitMiddleware enforces the cumulative RPM limit and injects
// rate-limit headers into every response.
func RateLimitMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		// Retrieve limiter from context (set by earlier middleware)
		limiter, ok := r.Context().Value(ratelimit.LimiterContextKey).(*ratelimit.Limiter)
		if !ok || limiter == nil {
			// If no limiter is present, just pass through
			next.ServeHTTP(w, r)
			return
		}

		// Attempt to acquire a token; if not available, reject with 429
		if !limiter.Allow() {
			http.Error(w, http.StatusText(http.StatusTooManyRequests), http.StatusTooManyRequests)
			return
		}

		// Defer header injection so that the handler can run
		defer ratelimit.WriteRateLimitHeaders(w, r)

		// Proceed to the next handler
		next.ServeHTTP(w, r)

		// After response, sleep to enforce the RPM window
		time.Sleep(limiter.Window)
	})
}
```
