package internal/ratelimit

import (
    "sync"
)

// RateLimit represents a rate limit configuration
type RateLimit struct {
    TotalRPM   int
    CurrentRPM int
    Lock       sync.RWMutex
}

// NewRateLimit returns a new rate limit configuration
func NewRateLimit(totalRPM int) *RateLimit {
    return &RateLimit{
        TotalRPM:   totalRPM,
        CurrentRPM: 0,
    }
}

// Update increments the current rate limit
func (rl *RateLimit) Update() {
    rl.Lock.Lock()
    defer rl.Lock.Unlock()
    rl.CurrentRPM++
}

// IsQuotaExhausted checks if the quota is exhausted
func (rl *RateLimit) IsQuotaExhausted() bool {
    rl.Lock.RLock()
    defer rl.Lock.RUnlock()
    return rl.CurrentRPM >= rl.TotalRPM
}

// Reset resets the current rate limit
func (rl *RateLimit) Reset() {
    rl.Lock.Lock()
    defer rl.Lock.Unlock()
    rl.CurrentRPM = 0
}
```
