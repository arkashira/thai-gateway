const axios = require('axios');
const { v4: uuidv4 } = require('uuid');

// Define a health check function for endpoints
async function healthCheck(endpoint) {
  try {
    const response = await axios.get(`${endpoint}/healthcheck`);
    return response.status === 200;
  } catch (error) {
    return false;
  }
}

// Define a function to get the next available endpoint
async function getNextAvailableEndpoint(endpoints) {
  for (const endpoint of endpoints) {
    if (await healthCheck(endpoint)) {
      return endpoint;
    }
  }
  throw new Error('No available endpoints');
}

// Define the circuit breaker function
async function circuitBreaker(endpoint, request) {
  const maxRetries = 3;
  const retryDelay = 500; // 500ms

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      const response = await axios.post(endpoint, request);
      return response;
    } catch (error) {
      if (attempt < maxRetries) {
        await new Promise((resolve) => setTimeout(resolve, retryDelay * (attempt + 1)));
      } else {
        // If all retries fail, throw an error
        throw new Error(`Failed to send request to ${endpoint} after ${maxRetries} retries`);
      }
    }
  }
}

// Define the main request handler function
async function handleRequest(endpoints, request) {
  const availableEndpoints = await Promise.all(endpoints.map(healthCheck));
  const availableEndpoint = await getNextAvailableEndpoint(endpoints);

  if (!availableEndpoint) {
    throw new Error('No available endpoints');
  }

  try {
    const response = await circuitBreaker(availableEndpoint, request);
    return response;
  } catch (error) {
    // If the circuit breaker fails, try the next available endpoint
    const nextEndpoint = await getNextAvailableEndpoint(endpoints);
    if (nextEndpoint) {
      return handleRequest([nextEndpoint], request);
    } else {
      throw error;
    }
  }
}

module.exports = handleRequest;
