+ gauge := prometheus.NewGaugeVec(
+ prometheus.GaugeOpts{
+ Namespace: "thai_gateway",
+ Subsystem: "provider",
+ Name:      "health",
+ Help:      "Health status of downstream providers (1 = healthy, 0 = unhealthy)",
+ },
+ []string{"provider"},
+ )
+ _ = prometheus.Register(gauge)
+
+ m := &Manager{
+ providers:    providers,
+ failureCount: make(map[string]int),
+ healthy:      make(map[string]bool),
+ gauge:        gauge,
+ }
+
+ for _, p := range providers {
+ m.healthy[p.ID()] = true
+ m.gauge.WithLabelValues(p.ID()).Set(1)
+ }
+ return m
+}

## 1. Acceptance Criteria
1. **Ping interval** – The health‑check scheduler invokes a ping to every registered provider **exactly every 30 seconds** (±1 s tolerance).  
2. **Failure threshold** – A provider that returns a failed ping **three consecutive times** is marked **Unhealthy**; a provider that subsequently returns a successful ping resets its failure counter and becomes **Healthy**.  
3. **Routing exclusion** – The router’s `SelectProvider()` function must never return a provider whose health status is **Unhealthy**.  
4. **Prometheus metric** – The metric `thai_gateway_provider_up{provider="<id>"}` is exported as **1** for Healthy and **0** for Unhealthy providers, refreshed after each ping cycle.  
5. **Cron reliability** – The health‑check cron (`RunHealthCheckCron`) must start without error, continue running indefinitely, and log any internal error without crashing the process.  
6. **Graceful shutdown** – When the application receives a context cancellation, the cron stops cleanly within 5 seconds.

func TestPingProvider_UpdatesHealthAndCounter(t *testing.T) {
	p := newMockProvider("p1", true) // will respond 200
	defer p.Close()

	// first successful ping → healthy, counter reset
	err := health.PingProvider(p)
	require.NoError(t, err)
	assert.Equal(t, provider.Healthy, p.Health())
	assert.Equal(t, 0, p.FailureCount())

	// simulate failure
	p.SetURL("http://invalid.invalid")
	err = health.PingProvider(p)
	require.Error(t, err)
	assert.Equal(t, 1, p.FailureCount())
	assert.Equal(t, provider.Healthy, p.Health()) // not yet unhealthy

	// second failure
	err = health.PingProvider(p)
	require.Error(t, err)
	assert.Equal(t, 2, p.FailureCount())
	assert.Equal(t, provider.Healthy, p.Health())

	// third failure → becomes Unhealthy
	err = health.PingProvider(p)
	require.Error(t, err)
	assert.Equal(t, 3, p.FailureCount())
	assert.Equal(t, provider.Unhealthy, p.Health())
}
