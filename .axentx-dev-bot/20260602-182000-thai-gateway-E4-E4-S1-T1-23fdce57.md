// health.go
package health

import (
	"context"
	"fmt"
	"log"
	"net/http"
	"time"

	"github.com/prometheus/client_golang/prometheus"
	"github.com/prometheus/client_golang/prometheus/promhttp"
)

// Provider represents an AI provider
type Provider struct {
	URL    string
	Healthy bool
	Failures int
}

// PingProvider pings a provider and updates its health status
func (p *Provider) PingProvider() {
	resp, err := http.Get(p.URL)
	if err != nil {
		p.Failures++
		p.Healthy = false
	} else {
		p.Failures = 0
		p.Healthy = true
		resp.Body.Close()
	}
}

// HealthCheckCron runs the health check cron job
func HealthCheckCron(providers []*Provider) {
	ticker := time.NewTicker(30 * time.Second)
	defer ticker.Stop()
	for {
		select {
		case <-ticker.C:
			for _, provider := range providers {
				provider.PingProvider()
				if provider.Failures >= 3 {
					provider.Healthy = false
				}
			}
		}
	}
}

// RegisterMetrics registers the health metrics with Prometheus
func RegisterMetrics(providers []*Provider) {
	healthMetric := prometheus.NewDesc(
		"provider_health",
		"Health status of the provider",
		[]string{"provider"},
		nil,
	)
	prometheus.MustRegister(prometheus.NewGaugeFunc(healthMetric, func() float64 {
		var healthy float64
		for _, provider := range providers {
			if provider.Healthy {
				healthy++
			}
		}
		return healthy
	}))
}

// ExcludeUnhealthyProviders excludes unhealthy providers from routing
func ExcludeUnhealthyProviders(providers []*Provider) []*Provider {
	var healthyProviders []*Provider
	for _, provider := range providers {
		if provider.Healthy {
			healthyProviders = append(healthyProviders, provider)
		}
	}
	return healthyProviders
}

func main() {
	providers := []*Provider{
		&Provider{URL: "http://provider1.com"},
		&Provider{URL: "http://provider2.com"},
	}
	go HealthCheckCron(providers)
	RegisterMetrics(providers)
	http.Handle("/metrics", promhttp.Handler())
	log.Fatal(http.ListenAndServe(":8080", nil))
}
```
