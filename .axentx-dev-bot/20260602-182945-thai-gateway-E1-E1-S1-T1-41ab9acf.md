package config

import (
	"encoding/json"
	"errors"
	"fmt"
	"io/ioutil"
	"log"
	"os"
	"time"

	"gopkg.in/yaml.v3"
)

// ProviderConfig represents a single provider entry.
type ProviderConfig struct {
	Name      string        `json:"name" yaml:"name"`
	URL       string        `json:"url" yaml:"url"`
	APIKey    string        `json:"api_key" yaml:"api_key"`
	RateLimit int           `json:"rate_limit" yaml:"rate_limit"` // requests per minute
	Timeout   time.Duration `json:"timeout" yaml:"timeout"`       // in seconds
}

// Config holds a slice of provider configs.
type Config struct {
	Providers []ProviderConfig `json:"providers" yaml:"providers"`
}

// LoadConfig reads a JSON or YAML file, unmarshals it into Config,
// validates each provider and logs a concise summary.
func LoadConfig(path string) (*Config, error) {
	if path == "" {
		return nil, errors.New("config path must not be empty")
	}

	data, err := ioutil.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("failed to read config file %s: %w", path, err)
	}

	cfg := &Config{}
	if err := unmarshalConfig(data, cfg); err != nil {
		return nil, err
	}

	if err := cfg.validate(); err != nil {
		return nil, err
	}

	logSummary(cfg)
	return cfg, nil
}

// unmarshalConfig detects JSON or YAML based on file extension.
func unmarshalConfig(data []byte, cfg *Config) error {
	if json.Valid(data) {
		return json.Unmarshal(data, cfg)
	}
	if err := yaml.Unmarshal(data, cfg); err != nil {
		return fmt.Errorf("failed to unmarshal config: %w", err)
	}
	return nil
}

// validate ensures each provider has required fields and sane values.
func (c *Config) validate() error {
	if len(c.Providers) == 0 {
		return errors.New("no providers defined in config")
	}
	for i, p := range c.Providers {
		if p.Name == "" {
			return fmt.Errorf("provider[%d] missing name", i)
		}
		if p.URL == "" {
			return fmt.Errorf("provider[%d] missing URL", i)
		}
		if p.APIKey == "" {
			return fmt.Errorf("provider[%d] missing API key", i)
		}
		if p.RateLimit <= 0 {
			return fmt.Errorf("provider[%d] must have positive rate limit", i)
		}
		if p.Timeout <= 0 {
			return fmt.Errorf("provider[%d] must have positive timeout", i)
		}
	}
	return nil
}

// logSummary prints a minimal but useful overview of the loaded config.
func logSummary(c *Config) {
	log.Printf("Loaded %d provider(s):", len(c.Providers))
	for _, p := range c.Providers {
		log.Printf("- %s: %s (rate %d/min, timeout %s)", p.Name, p.URL, p.RateLimit, p.Timeout)
	}
}

## 1. Acceptance Criteria
1. **Config Loading** – On startup the gateway must successfully read a configuration file (`providers.json` or `providers.yaml`) and populate an in‑memory list of providers with fields: `name`, `url`, `apiKey`, `rateLimit.requestsPerMinute`.
2. **Validation – Required Fields** – If any provider entry lacks `name`, `url`, or `apiKey`, the gateway aborts startup and logs an error specifying the missing field and provider index.
3. **Validation – URL Format** – Provider `url` must be a valid HTTP/HTTPS URL; otherwise startup aborts with a clear error.
4. **Validation – Rate‑Limit** – `rateLimit.requestsPerMinute` must be a positive integer; if omitted, defaults to `60`. Invalid values abort startup.
5. **Missing/Unreadable Config** – If the config file is missing, unreadable, or malformed (invalid JSON/YAML syntax), the gateway aborts with a descriptive error.
6. **Startup Logging** – After successful parsing, the gateway logs a single summary line:  
   `Registered X AI providers: [provider1, provider2, …]`.
7. **Security – No Sensitive Logging** – API keys must never be logged; only provider `name` is shown in logs.
