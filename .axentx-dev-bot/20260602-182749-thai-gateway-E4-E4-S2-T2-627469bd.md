class ErrorTracker:
    def __init__(self, threshold, duration, webhook_url):
        self.threshold = threshold
        self.duration = duration
        self.webhook_url = webhook_url
        self.error_count = 0
        self.total_requests = 0
        self.error_history = []
        
    def record_error(self):
        self.error_count += 1
        self.total_requests += 1
        self.error_history.append(True)
        
    def record_success(self):
        self.total_requests += 1
        self.error_history.append(False)
        
    def calculate_error_rate(self):
        if self.total_requests == 0:
            return 0
        return (self.error_count / self.total_requests) * 100
        
    def check_alert_condition(self):
        error_rate = self.calculate_error_rate()
        return error_rate > self.threshold and self._check_duration()
        
    def _check_duration(self):
        # Simple implementation - could be improved with time tracking
        return len(self.error_history) >= self.duration * 60
```
