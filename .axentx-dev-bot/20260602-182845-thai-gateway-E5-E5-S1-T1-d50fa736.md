```diff
+package metrics
 
+import (
+	"github.com/prometheus/client_golang/prometheus"
+	"net/http"
+)
+
+// Metrics struct holds the Prometheus metrics
+type Metrics struct {
+	RequestCount   *prometheus.CounterVec
+	RequestLatency *prometheus.HistogramVec
+	ProviderHealth *prometheus.GaugeVec
+}
+
+// NewMetrics creates and initializes the Prometheus metrics
+func NewMetrics() *Metrics {
+	// Create a new metrics instance
+	metrics := &Metrics{}
+
+	// Request count with provider label
+	metrics.RequestCount = prometheus.NewCounterVec(
+		prometheus.CounterOpts{
+			Name: "request_count",
+			Help: "Total number of requests processed",
+		},
+		[]string{"provider"},
+	)
+
+	// Request latency with provider label (in seconds)
+	metrics.RequestLatency = prometheus.NewHistogramVec(
+		prometheus.HistogramOpts{
+			Name:    "request_latency_seconds",
+			Help:    "Request latency distribution",
+			Buckets: prometheus.ExponentialBuckets(0.1, 2, 10),
+		},
+		[]string{"provider"},
+	)
+
+	// Provider health with provider label (0 = unhealthy, 1 = healthy)
+	metrics.ProviderHealth = prometheus.NewGaugeVec(
+		prometheus.GaugeOpts{
+			Name: "provider_health",
+			Help: "Health status of providers (0 = unhealthy, 1 = healthy)",
+		},
+		[]string{"provider"},
+	)
+
+	// Register all metrics with the default registry
+	prometheus.MustRegister(metrics.RequestCount)
+	prometheus.MustRegister(metrics.RequestLatency)
+	prometheus.MustRegister(metrics.ProviderHealth)
+
+	return metrics
+}
+
+// RegisterMetricsHandler registers the /metrics endpoint
+func RegisterMetricsHandler(m *Metrics, mux *http.ServeMux) {
+	// Expose metrics via /metrics endpoint
+	mux.Handle("/metrics", prometheus.Handler())
+}
+
+// IncrementRequestCount increments the request counter for a given provider
+func (m *Metrics) IncrementRequestCount(provider string) {
+	m.RequestCount.WithLabelValues(provider).Inc()
+}
+
+// RecordRequestLatency records the request latency for a given provider
+func (m *Metrics) RecordRequestLatency(provider string, latency float64) {
+	m.RequestLatency.WithLabelValues(provider).Observe(latency)
+}
+
+// SetProviderHealth sets the health status of a provider
+func (m *Metrics) SetProviderHealth(provider string, healthy bool) {
+	value := 0.0
+	if healthy {
+		value = 1.0
+	}
+	m.ProviderHealth.WithLabelValues(provider).Set(value)
```
```diff
+}
```

// metrics/metrics_test.go
package metrics

import (
	"net/http/httptest"
	"strings"
	"testing"
	"time"

	"github.com/prometheus/client_golang/prometheus"
	"github.com/prometheus/client_golang/prometheus/promhttp"
)

// helper to gather metrics from a registry
func gatherMetrics(reg *prometheus.Registry) []*prometheus.MetricFamily {
	mfs, err := reg.Gather()
	if err != nil {
		panic(err)
	}
	return mfs
}

// find metric family by name
func findMF(mfs []*prometheus.MetricFamily, name string) *prometheus.MetricFamily {
	for _, mf := range mfs {
		if mf.GetName() == name {
			return mf
		}
	}
	return nil
}

func TestMetricsEndpointReturns200AndCorrectHeader(t *testing.T) {
	handler := promhttp.Handler()
	req := httptest.NewRequest("GET", "/metrics", nil)
	w := httptest.NewRecorder()
	handler.ServeHTTP(w, req)

	if w.Code != 200 {
		t.Fatalf("expected status 200, got %d", w.Code)
	}
	ct := w.Header().Get("Content-Type")
	if !strings.Contains(ct, "text/plain") || !strings.Contains(ct, "version=0.0.4") {
		t.Fatalf("unexpected Content-Type: %s", ct)
	}
}

func TestRequestCountMetricIncrements(t *testing.T) {
	reg := prometheus.NewRegistry()
	counter := prometheus.NewCounterVec(
		prometheus.CounterOpts{Name: "request_count_total", Help: "Total requests"},
		[]string{"provider"},
	)
	reg.MustRegister(counter)

	// simulate 3 requests to provider "thai"
	for i := 0; i < 3; i++ {
		counter.WithLabelValues("thai").Inc()
	}

	mfs := gatherMetrics(reg)
	mf := findMF(mfs, "request_count_total")
	if mf == nil {
		t.Fatal("request_count_total not found")
	}
	val := mf.GetMetric()[0].GetCounter().GetValue()
	if val != 3 {
		t.Fatalf("expected counter 3, got %f", val)
	}
}

func TestRequestLatencyHistogram(t *testing.T) {
	reg := prometheus.NewRegistry()
	hist := prometheus.NewHistogramVec(
		prometheus.HistogramOpts{Name: "request_latency_seconds", Help: "Latency"},

	reg.MustRegister(hist)

	if metric.GetHistogram() == nil {

	}

	}
}

	mfs := gatherMetrics(reg)
	mf := findMF(mfs, "provider_health")
	if mf == nil {
		t.Fatal("provider_health not found")
	}
	for _, m := range m
