class RateLimiter {
    constructor(maxRequestsPerMinute) {
        this.maxRequests = maxRequestsPerMinute;
        this.windowSize = 60; // 60 seconds
        this.requests = new Array(this.windowSize).fill(0);
        this.index = 0;
    }

    canMakeRequest() {
        const totalRequests = this.requests.reduce((acc, val) => acc + val, 0);
        return totalRequests < this.maxRequests;
    }

    recordRequest() {
        this.requests[this.index] += 1;
        this.index = (this.index + 1) % this.windowSize;
        this.requests[this.index] = 0;
    }
}

const rateLimiter = new RateLimiter(1000);

module.exports = {
    checkRateLimit: () => {
        if (!rateLimiter.canMakeRequest()) {
            throw new Error('Rate limit exceeded');
        }
        rateLimiter.recordRequest();
    }
};

**Sliding‑Window Rate Limiter (JavaScript / Node.js)**  

```ts
/**
 * SlidingWindowRateLimiter
 *
 * Implements a fixed‑window sliding‑window algorithm that tracks the number
 * of requests per second over a configurable window (default 60 s).
 *
 * The implementation uses a circular buffer where each slot represents
 * one second.  The buffer is advanced by a single background timer
 * that fires every 1000 ms.  Each request increments the current slot.
 * When the timer fires, the slot that is now outside the window is
 * cleared and the index is advanced.  The total count is maintained
 * in O(1) time.
 *
 * This implementation is fully deterministic, thread‑safe in a single
 * Node.js process, and can be easily adapted to a distributed
 * environment by persisting the buffer state in a shared store.
 */

export class SlidingWindowRateLimiter {
  /** number of seconds in the sliding window */
  private readonly windowSize: number;
  /** maximum number of requests allowed in the window */
  private readonly maxRequests: number;

  /** circular buffer: one element per second */
  private readonly slots: number[];
  /** index of the current second in the buffer */
  private currentIndex: number = 0;
  /** total number of requests in the current window */
  private total: number = 0;

  /** timer that advances the window every second */
  private readonly timer: NodeJS.Timeout;

  /**
   * @param maxRequests maximum allowed requests in the window
   * @param windowSize  window length in seconds (default 60)
   */
  constructor(maxRequests: number, windowSize: number = 60) {
    if (maxRequests <= 0) throw new Error('maxRequests must be > 0');
    if (windowSize <= 0) throw new Error('windowSize must be > 0');

    this.maxRequests = maxRequests;
    this.windowSize = windowSize;
    this.slots = new Array(windowSize).fill(0);

    // Advance the window every second
    this.timer = setInterval(() => this.advanceWindow(), 1000);
    // Ensure the timer does not keep the process alive
    this.timer.unref();
  }

  /** Stop the internal timer (useful for tests / shutdown) */
  public stop(): void {
    clearInterval(this.timer);
  }

  /**
   * Record a request at the current time.
   *
   * @returns true if the request is allowed, false if the limit is exceeded
   */
  public recordRequest(): boolean {
    // If the request would exceed the limit, reject it
    if (this.total >= this.maxRequests) {
      return false;
    }

    // Increment current slot and total
    this.slots[this.currentIndex] += 1;
    this.total += 1;
    return true;
  }

  /**
   * Advance the sliding window by one second.
   *
   * This method clears the slot that is now outside the window,
   * updates the total, and moves the current index forward.
   */
  private advanceWindow(): void {
    // Move to the next slot
    this.currentIndex = (this.currentIndex + 1) % this.windowSize;

    // Subtract the count that is sliding out of the window
    const expiredCount = this.slots[this.currentIndex];
    this.total -= expiredCount;

    // Reset the slot for the new second
    this.slots[this.currentIndex] = 0;
  }

  /**
   * Returns the current number of requests in the window.
   */
  public getCurrentCount(): number {
    return this.total;
  }

  /**
   * Returns the remaining quota for the current window.
   */
  public getRemainingQuota(): number {
    return Math.max(this.maxRequests - this.total, 0);
  }
}
```
