  ]
}

```

### Nginx

See `docs/nginx_reverse_proxy.conf` for an example configuration.

```nginx
server {
    listen 80;
    location / {

        proxy_set_header X-Real-IP $remote_addr;
    }
}
```

    ProxyPass / http://localhost:8080/

</VirtualHost>
